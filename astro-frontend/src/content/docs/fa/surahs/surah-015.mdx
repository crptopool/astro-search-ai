<Aside type="quran" title="1">
الٓر ۚ تِلْكَ ءَايَٰتُ ٱلْكِتَٰبِ وَقُرْءَانٍۢ مُّبِينٍۢ
</Aside>

<Aside type="reference" title="1">
الر، این آیات کتاب، و قرآن مبین است.
</Aside>

---

<Aside type="quran" title="2">
رُّبَمَا يَوَدُّ ٱلَّذِينَ كَفَرُوا۟ لَوْ كَانُوا۟ مُسْلِمِينَ
</Aside>

<Aside type="reference" title="2">
کافران (هنگامی که آثار شوم اعمال خود را ببینند،) چه بسا آرزو می‌کنند که ای کاش مسلمان بودند!
</Aside>

---

<Aside type="quran" title="3">
ذَرْهُمْ يَأْكُلُوا۟ وَيَتَمَتَّعُوا۟ وَيُلْهِهِمُ ٱلْأَمَلُ ۖ فَسَوْفَ يَعْلَمُونَ
</Aside>

<Aside type="reference" title="3">
بگذار آنها بخورند، و بهره گیرند، و آرزوها آنان را غافل سازد؛ ولی بزودی خواهند فهمید!
</Aside>

---

<Aside type="quran" title="4">
وَمَآ أَهْلَكْنَا مِن قَرْيَةٍ إِلَّا وَلَهَا كِتَابٌۭ مَّعْلُومٌۭ
</Aside>

<Aside type="reference" title="4">
ما اهل هیچ شهر و دیاری را هلاک نکردیم مگر اینکه اجل معیّن (و زمان تغییر ناپذیری) داشتند!
</Aside>

---

<Aside type="quran" title="5">
مَّا تَسْبِقُ مِنْ أُمَّةٍ أَجَلَهَا وَمَا يَسْتَـْٔخِرُونَ
</Aside>

<Aside type="reference" title="5">
هیچ گروهی از اجل خود پیشی نمی‌گیرد؛ و از آن عقب نخواهد افتاد!
</Aside>

---

<Aside type="quran" title="6">
وَقَالُوا۟ يَٰٓأَيُّهَا ٱلَّذِى نُزِّلَ عَلَيْهِ ٱلذِّكْرُ إِنَّكَ لَمَجْنُونٌۭ
</Aside>

<Aside type="reference" title="6">
و گفتند: «ای کسی که «ذکر» [= قرآن‌] بر او نازل شده، مسلماً تو دیوانه‌ای!
</Aside>

---

<Aside type="quran" title="7">
لَّوْ مَا تَأْتِينَا بِٱلْمَلَٰٓئِكَةِ إِن كُنتَ مِنَ ٱلصَّٰدِقِينَ
</Aside>

<Aside type="reference" title="7">
اگر راست می‌گویی، چرا فرشتگان را نزد ما نمی‌زوری؟!»
</Aside>

---

<Aside type="quran" title="8">
مَا نُنَزِّلُ ٱلْمَلَٰٓئِكَةَ إِلَّا بِٱلْحَقِّ وَمَا كَانُوٓا۟ إِذًۭا مُّنظَرِينَ
</Aside>

<Aside type="reference" title="8">
(امّا اینها باید بدانند) ما فرشتگان را، جز بحق، نازل نمی‌کنیم،؛و هرگاه نازل شوند، دیگر به اینها مهلت داده نمی‌شود (؛و در صورت انکار، به عذاب الهی نابود می‌گردند)!
</Aside>

---

<Aside type="quran" title="9">
إِنَّا نَحْنُ نَزَّلْنَا ٱلذِّكْرَ وَإِنَّا لَهُۥ لَحَٰفِظُونَ
</Aside>

<Aside type="reference" title="9">
ما قرآن را نازل کردیم؛ و ما بطور قطع نگهدار آنیم!
</Aside>

---

<Aside type="quran" title="10">
وَلَقَدْ أَرْسَلْنَا مِن قَبْلِكَ فِى شِيَعِ ٱلْأَوَّلِينَ
</Aside>

<Aside type="reference" title="10">

</Aside>

---

<Aside type="quran" title="11">
وَمَا يَأْتِيهِم مِّن رَّسُولٍ إِلَّا كَانُوا۟ بِهِۦ يَسْتَهْزِءُونَ
</Aside>

<Aside type="reference" title="11">

</Aside>

---

<Aside type="quran" title="12">

<Aside type="reference" title="12">

</Aside>

---

<Aside type="quran" title="13">
لَا يُؤْمِنُونَ بِهِۦ ۖ وَقَدْ خَلَتْ سُنَّةُ ٱلْأَوَّلِينَ
</Aside>

<Aside type="reference" title="13">

</Aside>

<Aside type="quran" title="14">
وَلَوْ فَتَحْنَا عَلَيْهِم بَابًۭا مِّنَ ٱلسَّمَآءِ فَظَلُّوا۟ فِيهِ يَعْرُجُونَ
</Aside>

<Aside type="reference" title="14">

<Aside type="quran" title="15">

</Aside>

<Aside type="reference" title="15">

</Aside>

---

<Aside type="quran" title="16">
وَلَقَدْ جَعَلْنَا فِى ٱلسَّمَآءِ بُرُوجًۭا وَزَيَّنَّٰهَا لِلنَّٰظِرِينَ
</Aside>

<Aside type="reference" title="16">

</Aside>

<Aside type="quran" title="17">

</Aside>

<Aside type="reference" title="18">
مگر آن کس که استراق سمع کند (و دزدانه گوش فرا دهد) که «شهاب مبین» او را تعقیب می‌کند (و می‌راند)

<Aside type="quran" title="19">

</Aside>

<Aside type="reference" title="19">

</Aside>

<Aside type="quran" title="20">

<Aside type="reference" title="20">

</Aside>

<Aside type="quran" title="21">

</Aside>

<Aside type="reference" title="21">

<Aside type="quran" title="22">

<Aside type="reference" title="22">

</Aside>

<Aside type="quran" title="23">
