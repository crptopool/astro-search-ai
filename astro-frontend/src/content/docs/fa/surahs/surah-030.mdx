<Aside type="reference" title="3">

<Aside type="quran" title="4">
فِى بِضْعِ سِنِينَ ۗ لِلَّهِ ٱلْأَمْرُ مِن قَبْلُ وَمِنۢ بَعْدُ ۚ وَيَوْمَئِذٍۢ يَفْرَحُ ٱلْمُؤْمِنُونَ
</Aside>

<Aside type="reference" title="4">
در چند سال همه کارها از آن خداست؛ چه قبل و چه بعد (از این شکست و پیروزی)؛ و در آن روز، مؤمنان (بخاطر پیروزی دیگری) خوشحال خواهند شد...
</Aside>

<Aside type="quran" title="5">
بِنَصْرِ ٱللَّهِ ۚ يَنصُرُ مَن يَشَآءُ ۖ وَهُوَ ٱلْعَزِيزُ ٱلرَّحِيمُ
</Aside>

<Aside type="reference" title="5">
به سبب یاری خداوند؛ و او هر کس را بخواهد یاری می‌دهد؛ و او صاحب قدرت و رحیم است!
</Aside>

---

<Aside type="quran" title="6">
وَعْدَ ٱللَّهِ ۖ لَا يُخْلِفُ ٱللَّهُ وَعْدَهُۥ وَلَٰكِنَّ أَكْثَرَ ٱلنَّاسِ لَا يَعْلَمُونَ
</Aside>

<Aside type="reference" title="6">
این وعده‌ای است که خدا کرده؛ و خداوند هرگز از وعده‌اش تخلّف نمی‌کند؛ ولی بیشتر مردم نمی‌دانند!
</Aside>

---

<Aside type="quran" title="7">
يَعْلَمُونَ ظَٰهِرًۭا مِّنَ ٱلْحَيَوٰةِ ٱلدُّنْيَا وَهُمْ عَنِ ٱلْءَاخِرَةِ هُمْ غَٰفِلُونَ
</Aside>

<Aside type="reference" title="7">
آنها فقط ظاهری از زندگی دنیا را می‌دانند، و از آخرت (و پایان کار) غافلند!
</Aside>

---

<Aside type="quran" title="8">
أَوَلَمْ يَتَفَكَّرُوا۟ فِىٓ أَنفُسِهِم ۗ مَّا خَلَقَ ٱللَّهُ ٱلسَّمَٰوَٰتِ وَٱلْأَرْضَ وَمَا بَيْنَهُمَآ إِلَّا بِٱلْحَقِّ وَأَجَلٍۢ مُّسَمًّۭى ۗ وَإِنَّ كَثِيرًۭا مِّنَ ٱلنَّاسِ بِلِقَآئِ رَبِّهِمْ لَكَٰفِرُونَ
</Aside>

<Aside type="reference" title="8">
آیا آنان با خود نیندیشیدند که خداوند، آسمانها و زمین و آنچه را میان آن دو است جز بحق و برای زمان معیّنی نیافریده است؟! ولی بسیاری از مردم (رستاخیز و) لقای پروردگارشان را منکرند!
</Aside>

---

<Aside type="quran" title="9">
أَوَلَمْ يَسِيرُوا۟ فِى ٱلْأَرْضِ فَيَنظُرُوا۟ كَيْفَ كَانَ عَٰقِبَةُ ٱلَّذِينَ مِن قَبْلِهِمْ ۚ كَانُوٓا۟ أَشَدَّ مِنْهُمْ قُوَّةًۭ وَأَثَارُوا۟ ٱلْأَرْضَ وَعَمَرُوهَآ أَكْثَرَ مِمَّا عَمَرُوهَا وَجَآءَتْهُمْ رُسُلُهُم بِٱلْبَيِّنَٰتِ ۖ فَمَا كَانَ ٱللَّهُ لِيَظْلِمَهُمْ وَلَٰكِن كَانُوٓا۟ أَنفُسَهُمْ يَظْلِمُونَ
</Aside>

<Aside type="reference" title="9">
آیا در زمین گردش نکردند تا ببینند عاقبت کسانی که قبل از آنان بودند چگونه بود؟! آنها نیرومندتر از اینان بودند، و زمین را (برای زراعت و آبادی)بیش از اینان دگرگون ساختند و آباد کردند، و پیامبرانشان با دلایل روشن به سراغشان آمدند (امّا آنها انکار کردند و کیفر خود را دیدند)؛ خداوند هرگز به آنان ستم نکرد، آنها به خودشان ستم می‌کردند!
</Aside>

---

<Aside type="quran" title="10">
ثُمَّ كَانَ عَٰقِبَةَ ٱلَّذِينَ أَسَٰٓـُٔوا۟ ٱلسُّوٓأَىٰٓ أَن كَذَّبُوا۟ بِـَٔايَٰتِ ٱللَّهِ وَكَانُوا۟ بِهَا يَسْتَهْزِءُونَ
</Aside>

<Aside type="reference" title="10">
سپس سرانجام کسانی که اعمال بد مرتکب شدند به جایی رسید که آیات خدا را تکذیب کردند و آن را به مسخره گرفتند!
</Aside>

---

<Aside type="quran" title="11">
ٱللَّهُ يَبْدَؤُا۟ ٱلْخَلْقَ ثُمَّ يُعِيدُهُۥ ثُمَّ إِلَيْهِ تُرْجَعُونَ
</Aside>

<Aside type="reference" title="11">
خداوند آفرینش را آغاز می‌کند، سپس آن را بازمی‌گرداند، سپس شما را بسوی او باز می‌گردانند!
</Aside>

---

<Aside type="quran" title="12">
وَيَوْمَ تَقُومُ ٱلسَّاعَةُ يُبْلِسُ ٱلْمُجْرِمُونَ
</Aside>

<Aside type="reference" title="12">
آن روز که قیامت برپا می‌شود، مجرمان در نومیدی و غم و اندوه فرو می‌روند!
</Aside>

---

<Aside type="quran" title="13">
وَلَمْ يَكُن لَّهُم مِّن شُرَكَآئِهِمْ شُفَعَٰٓؤُا۟ وَكَانُوا۟ بِشُرَكَآئِهِمْ كَٰفِرِينَ
</Aside>

<Aside type="reference" title="13">
و برای آنان شفیعانی از معبودانشان نخواهد بود، و نسبت به معبودهایی که آنها را همتای خدا قرار داده بودند کافر می‌شوند!
</Aside>

---

<Aside type="quran" title="14">
وَيَوْمَ تَقُومُ ٱلسَّاعَةُ يَوْمَئِذٍۢ يَتَفَرَّقُونَ
</Aside>

<Aside type="reference" title="14">
آن روز که قیامت برپا می‌گردد، (مردم) از هم جدا می‌شوند؛
</Aside>

---

<Aside type="quran" title="15">
فَأَمَّا ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّٰلِحَٰتِ فَهُمْ فِى رَوْضَةٍۢ يُحْبَرُونَ
</Aside>

<Aside type="reference" title="15">
امّا آنان که ایمان آورده و اعمال صالح انجام دادند، در باغی از بهشت شاد و مسرور خواهند بود.
</Aside>

---

<Aside type="quran" title="16">
وَأَمَّا ٱلَّذِينَ كَفَرُوا۟ وَكَذَّبُوا۟ بِـَٔايَٰتِنَا وَلِقَآئِ ٱلْءَاخِرَةِ فَأُو۟لَٰٓئِكَ فِى ٱلْعَذَابِ مُحْضَرُونَ
</Aside>

<Aside type="reference" title="16">
و امّا آنان که به آیات ما و لقای آخرت کافر شدند، در عذاب الهی احضار می‌شوند.
</Aside>

---

<Aside type="quran" title="17">
فَسُبْحَٰنَ ٱللَّهِ حِينَ تُمْسُونَ وَحِينَ تُصْبِحُونَ
</Aside>

<Aside type="reference" title="17">
منزّه است خداوند به هنگامی که شام می‌کنید و صبح می‌کنید؛
</Aside>

---

<Aside type="quran" title="18">
وَلَهُ ٱلْحَمْدُ فِى ٱلسَّمَٰوَٰتِ وَٱلْأَرْضِ وَعَشِيًّۭا وَحِينَ تُظْهِرُونَ
</Aside>

<Aside type="reference" title="18">
و حمد و ستایش مخصوص اوست در آسمان و زمین، و به هنگام عصر و هنگامی که ظهر می‌کنید.
</Aside>

---

<Aside type="quran" title="19">
يُخْرِجُ ٱلْحَىَّ مِنَ ٱلْمَيِّتِ وَيُخْرِجُ ٱلْمَيِّتَ مِنَ ٱلْحَىِّ وَيُحْىِ ٱلْأَرْضَ بَعْدَ مَوْتِهَا ۚ وَكَذَٰلِكَ تُخْرَجُونَ
</Aside>

<Aside type="reference" title="19">
او زنده را از مرده بیرون می‌آورد، و مرده را از زنده، و زمین را پس از مردنش حیات می‌بخشد، و به همین گونه روز قیامت (از گورها) بیرون آورده می‌شوید!
</Aside>

---

<Aside type="quran" title="20">
وَمِنْ ءَايَٰتِهِۦٓ أَنْ خَلَقَكُم مِّن تُرَابٍۢ ثُمَّ إِذَآ أَنتُم بَشَرٌۭ تَنتَشِرُونَ
</Aside>

<Aside type="reference" title="20">
از نشانه‌های او این است که شما را از خاک آفرید، سپس بناگاه انسانهایی شدید و در روی زمین گسترش یافتید!
</Aside>

---

<Aside type="quran" title="21">
وَمِنْ ءَايَٰتِهِۦٓ أَنْ خَلَقَ لَكُم مِّنْ أَنفُسِكُمْ أَزْوَٰجًۭا لِّتَسْكُنُوٓا۟ إِلَيْهَا وَجَعَلَ بَيْنَكُم مَّوَدَّةًۭ وَرَحْمَةً ۚ إِنَّ فِى ذَٰلِكَ لَءَايَٰتٍۢ لِّقَوْمٍۢ يَتَفَكَّرُونَ
</Aside>

<Aside type="reference" title="21">
و از نشانه‌های او اینکه همسرانی از جنس خودتان برای شما آفرید تا در کنار آنان آرامش یابید، و در میانتان مودّت و رحمت قرار داد؛ در این نشانه‌هایی است برای گروهی که تفکّر می‌کنند!
</Aside>

---

<Aside type="quran" title="22">
وَمِنْ ءَايَٰتِهِۦ خَلْقُ ٱلسَّمَٰوَٰتِ وَٱلْأَرْضِ وَٱخْتِلَٰفُ أَلْسِنَتِكُمْ وَأَلْوَٰنِكُمْ ۚ إِنَّ فِى ذَٰلِكَ لَءَايَٰتٍۢ لِّلْعَٰلِمِينَ
</Aside>

<Aside type="reference" title="22">
و از آیات او آفرینش آسمانها و زمین، و تفاوت زبانها و رنگهای شماست؛ در این نشانه‌هایی است برای عالمان!
</Aside>

---

<Aside type="quran" title="23">
وَمِنْ ءَايَٰتِهِۦ مَنَامُكُم بِٱلَّيْلِ وَٱلنَّهَارِ وَٱبْتِغَآؤُكُم مِّن فَضْلِهِۦٓ ۚ إِنَّ فِى ذَٰلِكَ لَءَايَٰتٍۢ لِّقَوْمٍۢ يَسْمَعُونَ
</Aside>

<Aside type="reference" title="23">
و از نشانه‌های او خواب شما در شب و روز است و تلاش و کوششتان برای بهره‌گیری از فضل پروردگار (و تأمین معاش)؛ در این امور نشانه‌هایی است برای آنان که گوش شنوا دارند!
</Aside>

---

<Aside type="quran" title="24">
وَمِنْ ءَايَٰتِهِۦ يُرِيكُمُ ٱلْبَرْقَ خَوْفًۭا وَطَمَعًۭا وَيُنَزِّلُ مِنَ ٱلسَّمَآءِ مَآءًۭ فَيُحْىِۦ بِهِ ٱلْأَرْضَ بَعْدَ مَوْتِهَآ ۚ إِنَّ فِى ذَٰلِكَ لَءَايَٰتٍۢ لِّقَوْمٍۢ يَعْقِلُونَ
</Aside>

<Aside type="reference" title="24">
و از آیات او این است که برق و رعد را به شما نشان می‌دهد که هم مایه ترس و هم امید است (ترس از صاعقه، و امید به نزول باران)، و از آسمان آبی فرو می‌فرستد که زمین را بعد از مردنش بوسیله آن زنده می‌کند؛ در این نشانه‌هایی است برای جمعیّتی که می‌اندیشند!
</Aside>

---

<Aside type="quran" title="25">
وَمِنْ ءَايَٰتِهِۦٓ أَن تَقُومَ ٱلسَّمَآءُ وَٱلْأَرْضُ بِأَمْرِهِۦ ۚ ثُمَّ إِذَا دَعَاكُمْ دَعْوَةًۭ مِّنَ ٱلْأَرْضِ إِذَآ أَنتُمْ تَخْرُجُونَ
</Aside>

<Aside type="reference" title="25">
و از آیات او این است که آسمان و زمین به فرمان او برپاست؛ سپس هنگامی که شما را (در قیامت) از زمین فراخواند، ناگهان همه خارج می‌شوید (و در صحنه محشر حضور می‌یابید)!
</Aside>

---

<Aside type="quran" title="26">
وَلَهُۥ مَن فِى ٱلسَّمَٰوَٰتِ وَٱلْأَرْضِ ۖ كُلٌّۭ لَّهُۥ قَٰنِتُونَ
</Aside>

<Aside type="reference" title="26">
و از آن اوست تمام کسانی که در آسمانها و زمین‌اند و همگی در برابر او خاضع و مطیع‌اند!
</Aside>

---

<Aside type="quran" title="27">
وَهُوَ ٱلَّذِى يَبْدَؤُا۟ ٱلْخَلْقَ ثُمَّ يُعِيدُهُۥ وَهُوَ أَهْوَنُ عَلَيْهِ ۚ وَلَهُ ٱلْمَثَلُ ٱلْأَعْلَىٰ فِى ٱلسَّمَٰوَٰتِ وَٱلْأَرْضِ ۚ وَهُوَ ٱلْعَزِيزُ ٱلْحَكِيمُ
</Aside>

<Aside type="reference" title="27">
او کسی است که آفرینش را آغاز می‌کند، سپس آن را بازمی‌گرداند، و این کار برای او آسانتر می‌باشد؛ و برای اوست توصیف برتر در آسمانها و زمین؛ و اوست توانمند و حکیم!
</Aside>

---

<Aside type="quran" title="28">
ضَرَبَ لَكُم مَّثَلًۭا مِّنْ أَنفُسِكُمْ ۖ هَل لَّكُم مِّن مَّا مَلَكَتْ أَيْمَٰنُكُم مِّن شُرَكَآءَ فِى مَا رَزَقْنَٰكُمْ فَأَنتُمْ فِيهِ سَوَآءٌۭ تَخَافُونَهُمْ كَخِيفَتِكُمْ أَنفُسَكُمْ ۚ كَذَٰلِكَ نُفَصِّلُ ٱلْءَايَٰتِ لِقَوْمٍۢ يَعْقِلُونَ
</Aside>

<Aside type="reference" title="28">
خداوند مثالی از خودتان، برای شما زده است: آیا (اگر مملوک و برده‌ای داشته باشید)، این برده‌های شما هرگز در روزیهایی که به شما داده‌ایم شریک شما می‌باشند؛ آنچنان که هر دو مساوی بوده و از تصرّف مستقل و بدون اجازه آنان بیم داشته باشید، آن گونه که در مورد شرکای آزاد خود بیم دارید؟! اینچنین آیات خود را برای کسانی که تعقّل می‌کنند شرح می‌دهیم.
</Aside>

<Aside type="quran" title="29">
بَلِ ٱتَّبَعَ ٱلَّذِينَ ظَلَمُوٓا۟ أَهْوَآءَهُم بِغَيْرِ عِلْمٍۢ ۖ فَمَن يَهْدِى مَنْ أَضَلَّ ٱللَّهُ ۖ وَمَا لَهُم مِّن نَّٰصِرِينَ
</Aside>

<Aside type="reference" title="29">
ولی ظالمان بدون علم و آگاهی، از هوی و هوسهای خود پیروی کردند! پس چه کسی می‌تواند آنان را که خدا گمراه کرده است هدایت کند؟! و برای آنها هیچ یاوری نخواهد بود!

<Aside type="quran" title="56">
وَقَالَ ٱلَّذِينَ أُوتُوا۟ ٱلْعِلْمَ وَٱلْإِيمَٰنَ لَقَدْ لَبِثْتُمْ فِى كِتَٰبِ ٱللَّهِ إِلَىٰ يَوْمِ ٱلْبَعْثِ ۖ فَهَٰذَا يَوْمُ ٱلْبَعْثِ وَلَٰكِنَّكُمْ كُنتُمْ لَا تَعْلَمُونَ
</Aside>

<Aside type="reference" title="56">

</Aside>
