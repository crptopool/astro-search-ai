<Aside type="quran" title="3">
نَتْلُوا۟ عَلَيْكَ مِن نَّبَإِ مُوسَىٰ وَفِرْعَوْنَ بِٱلْحَقِّ لِقَوْمٍۢ يُؤْمِنُونَ
</Aside>

<Aside type="reference" title="3">
ما از داستان موسی و فرعون بحقّ بر تو می‌خوانیم، برای گروهی که (طالب حقّند و) ایمان می‌آورند!
</Aside>

---

<Aside type="quran" title="4">
إِنَّ فِرْعَوْنَ عَلَا فِى ٱلْأَرْضِ وَجَعَلَ أَهْلَهَا شِيَعًۭا يَسْتَضْعِفُ طَآئِفَةًۭ مِّنْهُمْ يُذَبِّحُ أَبْنَآءَهُمْ وَيَسْتَحْىِۦ نِسَآءَهُمْ ۚ إِنَّهُۥ كَانَ مِنَ ٱلْمُفْسِدِينَ
</Aside>

<Aside type="reference" title="4">
فرعون در زمین برتری‌جویی کرد، و اهل آن را به گروه‌های مختلفی تقسیم نمود؛ گروهی را به ضعف و ناتوانی می‌کشاند، پسرانشان را سر می‌برید و زنانشان را (برای کنیزی و خدمت) زنده نگه می‌داشت؛ او به یقین از مفسدان بود!
</Aside>

---

<Aside type="quran" title="5">
وَنُرِيدُ أَن نَّمُنَّ عَلَى ٱلَّذِينَ ٱسْتُضْعِفُوا۟ فِى ٱلْأَرْضِ وَنَجْعَلَهُمْ أَئِمَّةًۭ وَنَجْعَلَهُمُ ٱلْوَٰرِثِينَ
</Aside>

<Aside type="reference" title="5">
ما می‌خواهیم بر مستضعفان زمین منّت نهیم و آنان را پیشوایان و وارثان روی زمین قرار دهیم!
</Aside>

---

<Aside type="quran" title="6">
وَنُمَكِّنَ لَهُمْ فِى ٱلْأَرْضِ وَنُرِىَ فِرْعَوْنَ وَهَٰمَٰنَ وَجُنُودَهُمَا مِنْهُم مَّا كَانُوا۟ يَحْذَرُونَ
</Aside>

<Aside type="reference" title="6">
و حکومتشان را در زمین پابرجا سازیم؛ و به فرعون و هامان و لشکریانشان، آنچه را از آنها [= بنی اسرائیل‌] بیم داشتند نشان دهیم!
</Aside>

---

<Aside type="quran" title="7">
وَأَوْحَيْنَآ إِلَىٰٓ أُمِّ مُوسَىٰٓ أَنْ أَرْضِعِيهِ ۖ فَإِذَا خِفْتِ عَلَيْهِ فَأَلْقِيهِ فِى ٱلْيَمِّ وَلَا تَخَافِى وَلَا تَحْزَنِىٓ ۖ إِنَّا رَآدُّوهُ إِلَيْكِ وَجَاعِلُوهُ مِنَ ٱلْمُرْسَلِينَ
</Aside>

<Aside type="reference" title="7">
ما به مادر موسی الهام کردیم که: «او را شیر ده؛ و هنگامی که بر او ترسیدی، وی را در دریا(ی نیل) بیفکن؛ و نترس و غمگین مباش، که ما او را به تو بازمی‌گردانیم، و او را از رسولان قرار می‌دهیم!»
</Aside>

---

<Aside type="quran" title="8">
فَٱلْتَقَطَهُۥٓ ءَالُ فِرْعَوْنَ لِيَكُونَ لَهُمْ عَدُوًّۭا وَحَزَنًا ۗ إِنَّ فِرْعَوْنَ وَهَٰمَٰنَ وَجُنُودَهُمَا كَانُوا۟ خَٰطِـِٔينَ
</Aside>

<Aside type="reference" title="8">
(هنگامی که مادر بفرمان خدا او را به دریا افکند) خاندان فرعون او را از آب گرفتند، تا سرانجام دشمن آنان و مایه اندوهشان گردد! مسلّماً فرعون و هامان و لشکریانشان خطاکار بودند.
</Aside>

---

<Aside type="quran" title="9">
وَقَالَتِ ٱمْرَأَتُ فِرْعَوْنَ قُرَّتُ عَيْنٍۢ لِّى وَلَكَ ۖ لَا تَقْتُلُوهُ عَسَىٰٓ أَن يَنفَعَنَآ أَوْ نَتَّخِذَهُۥ وَلَدًۭا وَهُمْ لَا يَشْعُرُونَ
</Aside>

<Aside type="reference" title="9">
همسر فرعون (چون دید آنها قصد کشتن کودک را دارند) گفت: «نور چشم من و توست! او را نکشید شاید برای ما مفید باشد، یا او را بعنوان پسر خود برگزینیم!» و آنها نمی‌فهمیدند (که دشمن اصلی خود را در آغوش خویش می‌پرورانند)!
</Aside>

<Aside type="quran" title="10">
وَأَصْبَحَ فُؤَادُ أُمِّ مُوسَىٰ فَٰرِغًا ۖ إِن كَادَتْ لَتُبْدِى بِهِۦ لَوْلَآ أَن رَّبَطْنَا عَلَىٰ قَلْبِهَا لِتَكُونَ مِنَ ٱلْمُؤْمِنِينَ
</Aside>

<Aside type="reference" title="10">
(سرانجام) قلب مادر موسی از همه چیز (جز یاد فرزندش) تهی گشت؛ و اگر دل او را (بوسیله ایمان و امید) محکم نکرده بودیم، نزدیک بود مطلب را افشا کند!
</Aside>

<Aside type="quran" title="11">
وَقَالَتْ لِأُخْتِهِۦ قُصِّيهِ ۖ فَبَصُرَتْ بِهِۦ عَن جُنُبٍۢ وَهُمْ لَا يَشْعُرُونَ
</Aside>

<Aside type="reference" title="11">
و (مادر موسی) به خواهر او گفت: «وضع حال او را پیگیری کن!» او نیز از دور ماجرا را مشاهده کرد در حالی که آنان بی‌خبر بودند.
</Aside>

<Aside type="quran" title="12">
۞ وَحَرَّمْنَا عَلَيْهِ ٱلْمَرَاضِعَ مِن قَبْلُ فَقَالَتْ هَلْ أَدُلُّكُمْ عَلَىٰٓ أَهْلِ بَيْتٍۢ يَكْفُلُونَهُۥ لَكُمْ وَهُمْ لَهُۥ نَٰصِحُونَ
</Aside>

<Aside type="reference" title="12">
ما همه زنان شیرده را از پیش بر او حرام کردیم (تا تنها به آغوش مادر بازگردد)؛ و خواهرش (که بیتابی مأموران را برای پیدا کردن دایه مشاهده کرد) گفت: «آیا شما را به خانواده‌ای راهنمایی کنم که می‌توانند این نوزاد را برای شما کفالت کنند و خیرخواه او باشند؟!»
</Aside>

<Aside type="quran" title="13">
فَرَدَدْنَٰهُ إِلَىٰٓ أُمِّهِۦ كَىْ تَقَرَّ عَيْنُهَا وَلَا تَحْزَنَ وَلِتَعْلَمَ أَنَّ وَعْدَ ٱللَّهِ حَقٌّۭ وَلَٰكِنَّ أَكْثَرَهُمْ لَا يَعْلَمُونَ
</Aside>

<Aside type="reference" title="13">
ما او را به مادرش بازگرداندیم تا چشمش روشن شود و غمگین نباشد و بداند که وعده الهی حق است؛ ولی بیشتر آنان نمی‌دانند!
</Aside>

<Aside type="quran" title="14">
وَلَمَّا بَلَغَ أَشُدَّهُۥ وَٱسْتَوَىٰٓ ءَاتَيْنَٰهُ حُكْمًۭا وَعِلْمًۭا ۚ وَكَذَٰلِكَ نَجْزِى ٱلْمُحْسِنِينَ
</Aside>

<Aside type="reference" title="14">
و هنگامی که (موسی) نیرومند و کامل شد، حکمت و دانش به او دادیم؛ و این گونه نیکوکاران را جزا می‌دهیم!
</Aside>

<Aside type="quran" title="15">
وَدَخَلَ ٱلْمَدِينَةَ عَلَىٰ حِينِ غَفْلَةٍۢ مِّنْ أَهْلِهَا فَوَجَدَ فِيهَا رَجُلَيْنِ يَقْتَتِلَانِ هَٰذَا مِن شِيعَتِهِۦ وَهَٰذَا مِنْ عَدُوِّهِۦ ۖ فَٱسْتَغَٰثَهُ ٱلَّذِى مِن شِيعَتِهِۦ عَلَى ٱلَّذِى مِنْ عَدُوِّهِۦ فَوَكَزَهُۥ مُوسَىٰ فَقَضَىٰ عَلَيْهِ ۖ قَالَ هَٰذَا مِنْ عَمَلِ ٱلشَّيْطَٰنِ ۖ إِنَّهُۥ عَدُوٌّۭ مُّضِلٌّۭ مُّبِينٌۭ
</Aside>

<Aside type="reference" title="15">
او به هنگامی که اهل شهر در غفلت بودند وارد شهر شد؛ ناگهان دو مرد را دید که به جنگ و نزاع مشغولند؛ یکی از پیروان او بود (و از بنی اسرائیل)، و دیگری از دشمنانش، آن که از پیروان او بود در برابر دشمنش از وی تقاضای کمک نمود؛ موسی مشت محکمی بر سینه او زد و کار او را ساخت (و بر زمین افتاد و مرد)؛ موسی گفت: «این (نزاع شما) از عمل شیطان بود، که او دشمن و گمراه‌کننده آشکاری است»
</Aside>

<Aside type="quran" title="16">
قَالَ رَبِّ إِنِّى ظَلَمْتُ نَفْسِى فَٱغْفِرْ لِى فَغَفَرَ لَهُۥٓ ۚ إِنَّهُۥ هُوَ ٱلْغَفُورُ ٱلرَّحِيمُ
</Aside>

<Aside type="reference" title="16">
(سپس) عرض کرد: «پروردگارا! من به خویشتن ستم کردم؛ مرا ببخش!» خداوند او را بخشید، که او غفور و رحیم است!
</Aside>

<Aside type="quran" title="17">
قَالَ رَبِّ بِمَآ أَنْعَمْتَ عَلَىَّ فَلَنْ أَكُونَ ظَهِيرًۭا لِّلْمُجْرِمِينَ
</Aside>

<Aside type="reference" title="17">
عرض کرد: «پروردگارا! بشکرانه نعمتی که به من دادی، هرگز پشتیبان مجرمان نخواهم بود!»
</Aside>

<Aside type="quran" title="18">
فَأَصْبَحَ فِى ٱلْمَدِينَةِ خَآئِفًۭا يَتَرَقَّبُ فَإِذَا ٱلَّذِى ٱسْتَنصَرَهُۥ بِٱلْأَمْسِ يَسْتَصْرِخُهُۥ ۚ قَالَ لَهُۥ مُوسَىٰٓ إِنَّكَ لَغَوِىٌّۭ مُّبِينٌۭ
</Aside>

<Aside type="reference" title="18">
موسی در شهر ترسان بود و هر لحظه در انتظار حادثه‌ای (و در جستجوی اخبار)؛ ناگهان دید همان کسی که دیروز از او یاری طلبیده بود فریاد می‌زند و از او کمک می‌خواهد، موسی به او گفت: «تو آشکارا انسان (ماجراجو و) گمراهی هستی!»
</Aside>

<Aside type="quran" title="19">
فَلَمَّآ أَنْ أَرَادَ أَن يَبْطِشَ بِٱلَّذِى هُوَ عَدُوٌّۭ لَّهُمَا قَالَ يَٰمُوسَىٰٓ أَتُرِيدُ أَن تَقْتُلَنِى كَمَا قَتَلْتَ نَفْسًۢا بِٱلْأَمْسِ ۖ إِن تُرِيدُ إِلَّآ أَن تَكُونَ جَبَّارًۭا فِى ٱلْأَرْضِ وَمَا تُرِيدُ أَن تَكُونَ مِنَ ٱلْمُصْلِحِينَ
</Aside>

<Aside type="reference" title="19">
و هنگامی که خواست با کسی که دشمن هر دوی آنها بود درگیر شود و با قدرت مانع او گردد، (فریادش بلند شد،) گفت: «ای موسی می‌خواهی! مرا بکشی همان گونه که دیروز انسانی را کشتی؟! تو فقط می‌خواهی جبّاری در روی زمین باشی، و نمی‌خواهی از مصلحان باشی!»
</Aside>

<Aside type="quran" title="20">
وَجَآءَ رَجُلٌۭ مِّنْ أَقْصَا ٱلْمَدِينَةِ يَسْعَىٰ قَالَ يَٰمُوسَىٰٓ إِنَّ ٱلْمَلَأَ يَأْتَمِرُونَ بِكَ لِيَقْتُلُوكَ فَٱخْرُجْ إِنِّى لَكَ مِنَ ٱلنَّٰصِحِينَ

<Aside type="reference" title="20">

</Aside>

<Aside type="quran" title="21">

</Aside>

<Aside type="reference" title="21">

<Aside type="quran" title="22">

</Aside>

<Aside type="reference" title="22">

</Aside>

<Aside type="quran" title="23">
وَلَمَّا وَرَدَ مَآءَ مَدْيَنَ وَجَدَ عَلَيْهِ أُمَّةًۭ مِّنَ ٱلنَّاسِ يَسْقُونَ وَوَجَدَ مِن دُونِهِمُ ٱمْرَأَتَيْنِ تَذُودَانِ ۖ قَالَ مَا خَطْبُكُمَا ۖ قَالَتَا لَا نَسْقِى حَتَّىٰ يُصْدِرَ ٱلرِّعَآءُ ۖ وَأَبُونَا شَيْخٌۭ كَبِيرٌۭ
</Aside>

<Aside type="reference" title="23">
و هنگامی که به (چاه) آب مدین رسید، گروهی از مردم را در آنجا دید که چهارپایان خود را سیراب می‌کنند؛ و در کنار آنان دو زن را دید که مراقب گوسفندان خویشند (و به چاه نزدیک نمی‌شوند؛ موسی) به آن دو گفت: «کار شما چیست؟ (چرا گوسفندان خود را آب نمی‌دهید؟!)» گفتند: «ما آنها را آب نمی‌دهیم تا چوپانها همگی خارج شوند؛ و پدر ما پیرمرد کهنسالی است (و قادر بر این کارها نیست.)!»
</Aside>

<Aside type="quran" title="24">
