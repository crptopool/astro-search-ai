<Aside type="reference" title="3">

<Aside type="quran" title="4">
وَفِى خَلْقِكُمْ وَمَا يَبُثُّ مِن دَآبَّةٍ ءَايَٰتٌۭ لِّقَوْمٍۢ يُوقِنُونَ
</Aside>

<Aside type="reference" title="4">
و نیز در آفرینش شما و جنبندگانی که (در سراسر زمین) پراکنده ساخته، نشانه‌هایی است برای جمعیّتی که اهل یقینند.
</Aside>

---

<Aside type="quran" title="5">
وَٱخْتِلَٰفِ ٱلَّيْلِ وَٱلنَّهَارِ وَمَآ أَنزَلَ ٱللَّهُ مِنَ ٱلسَّمَآءِ مِن رِّزْقٍۢ فَأَحْيَا بِهِ ٱلْأَرْضَ بَعْدَ مَوْتِهَا وَتَصْرِيفِ ٱلرِّيَٰحِ ءَايَٰتٌۭ لِّقَوْمٍۢ يَعْقِلُونَ
</Aside>

<Aside type="reference" title="5">
و نیز در آمد و شد شب و روز، و رزق (و بارانی) که خداوند از آسمان نازل کرده و بوسیله آن زمین را بعد از مردنش حیات بخشیده و همچنین در وزش بادها، نشانه‌های روشنی است برای گروهی که اهل تفکّرند!
</Aside>

---

<Aside type="quran" title="6">
تِلْكَ ءَايَٰتُ ٱللَّهِ نَتْلُوهَا عَلَيْكَ بِٱلْحَقِّ ۖ فَبِأَىِّ حَدِيثٍۭ بَعْدَ ٱللَّهِ وَءَايَٰتِهِۦ يُؤْمِنُونَ
</Aside>

<Aside type="reference" title="6">
اینها آیات خداوند است که ما آن را بحقّ بر تو تلاوت می‌کنیم؛ اگر آنها به این آیات ایمان نیاورند، به کدام سخن بعد از سخن خدا و آیاتش ایمان می‌آورند؟!
</Aside>

<Aside type="quran" title="8">
يَسْمَعُ ءَايَٰتِ ٱللَّهِ تُتْلَىٰ عَلَيْهِ ثُمَّ يُصِرُّ مُسْتَكْبِرًۭا كَأَن لَّمْ يَسْمَعْهَا ۖ فَبَشِّرْهُ بِعَذَابٍ أَلِيمٍۢ
</Aside>

<Aside type="reference" title="8">
که پیوسته آیات خدا را می‌شنود که بر او تلاوت می‌شود، امّا از روی تکبّر اصرار بر مخالفت دارد؛ گویی اصلاً آن را هیچ نشنیده است؛ چنین کسی را به عذابی دردناک بشارت ده!
</Aside>

---

<Aside type="quran" title="9">
وَإِذَا عَلِمَ مِنْ ءَايَٰتِنَا شَيْـًٔا ٱتَّخَذَهَا هُزُوًا ۚ أُو۟لَٰٓئِكَ لَهُمْ عَذَابٌۭ مُّهِينٌۭ
</Aside>

<Aside type="reference" title="9">
و هرگاه از بعضی آیات ما آگاه شود، آن را به باد استهزا می‌گیرد؛ برای آنان عذاب خوارکننده‌ای است!
</Aside>

---

<Aside type="quran" title="10">
مِّن وَرَآئِهِمْ جَهَنَّمُ ۖ وَلَا يُغْنِى عَنْهُم مَّا كَسَبُوا۟ شَيْـًۭٔا وَلَا مَا ٱتَّخَذُوا۟ مِن دُونِ ٱللَّهِ أَوْلِيَآءَ ۖ وَلَهُمْ عَذَابٌ عَظِيمٌ
</Aside>

<Aside type="reference" title="10">
و پشت سرشان دوزخ است؛ و هرگز آنچه را به دست آورده‌اند آنها را (از عذاب الهی) رهایی نمی‌بخشد، و نه اولیایی که غیر از خدا برای خود برگزیدند (مایه نجاتشان خواهند بود)؛ و عذاب بزرگی برای آنهاست!
</Aside>

---

<Aside type="quran" title="11">
هَٰذَا هُدًۭى ۖ وَٱلَّذِينَ كَفَرُوا۟ بِـَٔايَٰتِ رَبِّهِمْ لَهُمْ عَذَابٌۭ مِّن رِّجْزٍ أَلِيمٌ
</Aside>

<Aside type="reference" title="11">
این (قرآن) مایه هدایت است، و کسانی که به آیات پروردگارشان کافر شدند، عذابی سخت و دردناک دارند!
</Aside>

---

<Aside type="quran" title="12">
۞ ٱللَّهُ ٱلَّذِى سَخَّرَ لَكُمُ ٱلْبَحْرَ لِتَجْرِىَ ٱلْفُلْكُ فِيهِ بِأَمْرِهِۦ وَلِتَبْتَغُوا۟ مِن فَضْلِهِۦ وَلَعَلَّكُمْ تَشْكُرُونَ
</Aside>

<Aside type="reference" title="12">
خداوند همان کسی است که دریا را مسخّر شما کرد تا کشتیها بفرمانش در آن حرکت کنند و بتوانند از فضل او بهره گیرید، و شاید شکر نعمتهایش را بجا آورید!
</Aside>

---

<Aside type="quran" title="13">
وَسَخَّرَ لَكُم مَّا فِى ٱلسَّمَٰوَٰتِ وَمَا فِى ٱلْأَرْضِ جَمِيعًۭا مِّنْهُ ۚ إِنَّ فِى ذَٰلِكَ لَءَايَٰتٍۢ لِّقَوْمٍۢ يَتَفَكَّرُونَ
</Aside>

<Aside type="reference" title="13">
او آنچه در آسمانها و آنچه در زمین است همه را از سوی خودش مسخّر شما ساخته؛ در این نشانه‌های (مهمّی) است برای کسانی که اندیشه می‌کنند!
</Aside>

---

<Aside type="quran" title="14">
قُل لِّلَّذِينَ ءَامَنُوا۟ يَغْفِرُوا۟ لِلَّذِينَ لَا يَرْجُونَ أَيَّامَ ٱللَّهِ لِيَجْزِىَ قَوْمًۢا بِمَا كَانُوا۟ يَكْسِبُونَ
</Aside>

<Aside type="reference" title="14">
به مؤمنان بگو: «کسانی را که امید به ایّام اللّه [= روز رستاخیز] ندارند مورد عفو قرار دهند تا خداوند هر قومی را به اعمالی که انجام می‌دادند جزا دهد»!
</Aside>

---

<Aside type="quran" title="15">
مَنْ عَمِلَ صَٰلِحًۭا فَلِنَفْسِهِۦ ۖ وَمَنْ أَسَآءَ فَعَلَيْهَا ۖ ثُمَّ إِلَىٰ رَبِّكُمْ تُرْجَعُونَ
</Aside>

<Aside type="reference" title="15">
هر کس کار شایسته‌ای بجا آورد، برای خود بجا آورده است؛ و کسی که کار بد می‌کند، به زیان خود اوست؛ سپس همه شما به سوی پروردگارتان بازگردانده می‌شوید!
</Aside>

<Aside type="quran" title="16">
وَلَقَدْ ءَاتَيْنَا بَنِىٓ إِسْرَٰٓءِيلَ ٱلْكِتَٰبَ وَٱلْحُكْمَ وَٱلنُّبُوَّةَ وَرَزَقْنَٰهُم مِّنَ ٱلطَّيِّبَٰتِ وَفَضَّلْنَٰهُمْ عَلَى ٱلْعَٰلَمِينَ
</Aside>

<Aside type="reference" title="16">
ما بنی اسرائیل را کتاب (آسمانی) و حکومت و نبّوت بخشیدیم و از روزیهای پاکیزه به آنها عطا کردیم و آنان را بر جهانیان (و مردم عصر خویش) برتری بخشیدیم؛
</Aside>

<Aside type="quran" title="17">
وَءَاتَيْنَٰهُم بَيِّنَٰتٍۢ مِّنَ ٱلْأَمْرِ ۖ فَمَا ٱخْتَلَفُوٓا۟ إِلَّا مِنۢ بَعْدِ مَا جَآءَهُمُ ٱلْعِلْمُ بَغْيًۢا بَيْنَهُمْ ۚ إِنَّ رَبَّكَ يَقْضِى بَيْنَهُمْ يَوْمَ ٱلْقِيَٰمَةِ فِيمَا كَانُوا۟ فِيهِ يَخْتَلِفُونَ
</Aside>

<Aside type="reference" title="17">
و دلایل روشنی از امر نبّوت و شریعت در اختیارشان قرار دادیم؛ آنها اختلاف نکردند مگر بعد از علم و آگاهی؛ و این اختلاف بخاطر ستم و برتری‌جویی آنان بود؛ امّا پروردگارت روز قیامت در میان آنها در آنچه اختلاف داشتند داوری می‌کند.
</Aside>

---

<Aside type="quran" title="18">
ثُمَّ جَعَلْنَٰكَ عَلَىٰ شَرِيعَةٍۢ مِّنَ ٱلْأَمْرِ فَٱتَّبِعْهَا وَلَا تَتَّبِعْ أَهْوَآءَ ٱلَّذِينَ لَا يَعْلَمُونَ
</Aside>

<Aside type="reference" title="18">
سپس تو را بر شریعت و آیین حقّی قرار دادیم؛ از آن پیروی کن و از هوسهای کسانی که آگاهی ندارند پیروی مکن!
</Aside>

<Aside type="quran" title="19">

</Aside>

<Aside type="reference" title="19">
آنها هرگز نمی‌توانند تو را در برابر خداوند بی‌نیاز کنند (و از عذابش برهانند)؛ و ظالمان یار و یاور یکدیگرند، امّا خداوند یار و یاور پرهیزگاران است!
</Aside>

<Aside type="reference" title="20">

<Aside type="quran" title="21">

<Aside type="quran" title="23">

<Aside type="reference" title="23">

</Aside>

<Aside type="quran" title="24">

<Aside type="quran" title="25">

<Aside type="reference" title="25">

وَتَرَىٰ كُلَّ أُمَّةٍۢ جَاثِيَةًۭ ۚ كُلُّ أُمَّةٍۢ تُدْعَىٰٓ إِلَىٰ كِتَٰبِهَا ٱلْيَوْمَ تُجْزَوْنَ مَا كُنتُمْ تَعْمَلُونَ
</Aside>

<Aside type="reference" title="28">
در آن روز هر امّتی را می‌بینی (که از شدّت ترس و وحشت) بر زانو نشسته؛ هر امّتی بسوی کتابش خوانده می‌شود، و (به آنها می‌گویند:) امروز جزای آنچه را انجام می‌دادید به شما می‌دهند!
</Aside>

---

<Aside type="quran" title="29">
هَٰذَا كِتَٰبُنَا يَنطِقُ عَلَيْكُم بِٱلْحَقِّ ۚ إِنَّا كُنَّا نَسْتَنسِخُ مَا كُنتُمْ تَعْمَلُونَ
</Aside>

<Aside type="reference" title="29">
این کتاب ما است که بحق با شما سخن می‌گوید (و اعمال شما را بازگو می‌کند)؛ ما آنچه را انجام می‌دادید می‌نوشتیم!
</Aside>

---

<Aside type="quran" title="30">
فَأَمَّا ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّٰلِحَٰتِ فَيُدْخِلُهُمْ رَبُّهُمْ فِى رَحْمَتِهِۦ ۚ ذَٰلِكَ هُوَ ٱلْفَوْزُ ٱلْمُبِينُ
</Aside>

<Aside type="reference" title="30">
امّا کسانی که ایمان آوردند و اعمال صالح انجام دادند، پروردگارشان آنها را در رحمت خود وارد می‌کند؛ این همان پیروزی بزرگ است!
</Aside>

---

<Aside type="quran" title="31">
وَأَمَّا ٱلَّذِينَ كَفَرُوٓا۟ أَفَلَمْ تَكُنْ ءَايَٰتِى تُتْلَىٰ عَلَيْكُمْ فَٱسْتَكْبَرْتُمْ وَكُنتُمْ قَوْمًۭا مُّجْرِمِينَ
</Aside>

<Aside type="reference" title="31">
امّا کسانی که کافر شدند (به آنها گفته می‌شود:) مگر آیات من بر شما خوانده نمی‌شد و شما استکبار کردید و قوم مجرمی بودید؟!
</Aside>

---

<Aside type="quran" title="32">
وَإِذَا قِيلَ إِنَّ وَعْدَ ٱللَّهِ حَقٌّۭ وَٱلسَّاعَةُ لَا رَيْبَ فِيهَا قُلْتُم مَّا نَدْرِى مَا ٱلسَّاعَةُ إِن نَّظُنُّ إِلَّا ظَنًّۭا وَمَا نَحْنُ بِمُسْتَيْقِنِينَ
</Aside>

<Aside type="reference" title="32">
و هنگامی که گفته می‌شد: «وعده خداوند حقّ است، و در قیامت هیچ شکّی نیست» ، شما می‌گفتید: «ما نمی‌دانیم قیامت چیست؟ ما تنها گمانی در این باره داریم، و به هیچ‌وجه یقین نداریم!»
</Aside>

---

<Aside type="quran" title="33">
وَبَدَا لَهُمْ سَيِّـَٔاتُ مَا عَمِلُوا۟ وَحَاقَ بِهِم مَّا كَانُوا۟ بِهِۦ يَسْتَهْزِءُونَ
</Aside>

<Aside type="reference" title="33">
و بدیهای اعمالشان برای آنان آشکار می‌شود، و سرانجام آنچه را استهزا می‌کردند آنها را فرامی‌گیرد!
</Aside>

---

<Aside type="quran" title="34">
وَقِيلَ ٱلْيَوْمَ نَنسَىٰكُمْ كَمَا نَسِيتُمْ لِقَآءَ يَوْمِكُمْ هَٰذَا وَمَأْوَىٰكُمُ ٱلنَّارُ وَمَا لَكُم مِّن نَّٰصِرِينَ
</Aside>

<Aside type="reference" title="34">
و به آنها گفته می‌شود: «امروز شما را فراموش می‌کنیم همان‌گونه که شما دیدار امروزتان را فراموش کردید؛ و جایگاه شما دوزخ است و هیچ یاوری ندارید!
</Aside>

---

<Aside type="quran" title="35">
ذَٰلِكُم بِأَنَّكُمُ ٱتَّخَذْتُمْ ءَايَٰتِ ٱللَّهِ هُزُوًۭا وَغَرَّتْكُمُ ٱلْحَيَوٰةُ ٱلدُّنْيَا ۚ فَٱلْيَوْمَ لَا يُخْرَجُونَ مِنْهَا وَلَا هُمْ يُسْتَعْتَبُونَ
</Aside>

<Aside type="reference" title="35">
این بخاطر آن است که شما آیات خدا را به مسخره گرفتید و زندگی دنیا شما را فریب داد! «امروز نه آنان را از دوزخ بیرون می‌آورند، و نه هیچ‌گونه عذری از آنها پذیرفته می‌شود!
</Aside>

---

<Aside type="quran" title="36">
فَلِلَّهِ ٱلْحَمْدُ رَبِّ ٱلسَّمَٰوَٰتِ وَرَبِّ ٱلْأَرْضِ رَبِّ ٱلْعَٰلَمِينَ
</Aside>

<Aside type="reference" title="36">
پس حمد و ستایش مخصوص خداست، پروردگار آسمانها و پروردگار زمین و پروردگار همه جهانیان!
</Aside>

<Aside type="quran" title="37">
وَلَهُ ٱلْكِبْرِيَآءُ فِى ٱلسَّمَٰوَٰتِ وَٱلْأَرْضِ ۖ وَهُوَ ٱلْعَزِيزُ ٱلْحَكِيمُ
</Aside>

<Aside type="reference" title="37">
و برای اوست کبریا و عظمت در آسمانها و زمین، و اوست عزیز و حکیم!
</Aside>
