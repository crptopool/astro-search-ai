<Aside type="quran" title="1">
تَبَٰرَكَ ٱلَّذِى بِيَدِهِ ٱلْمُلْكُ وَهُوَ عَلَىٰ كُلِّ شَىْءٍۢ قَدِيرٌ
</Aside>

<Aside type="reference" title="1">
پربرکت و زوال‌ناپذیر است کسی که حکومت جهان هستی به دست اوست، و او بر هر چیز تواناست.
</Aside>

---

<Aside type="quran" title="2">
ٱلَّذِى خَلَقَ ٱلْمَوْتَ وَٱلْحَيَوٰةَ لِيَبْلُوَكُمْ أَيُّكُمْ أَحْسَنُ عَمَلًۭا ۚ وَهُوَ ٱلْعَزِيزُ ٱلْغَفُورُ
</Aside>

<Aside type="reference" title="2">
آن کس که مرگ و حیات را آفرید تا شما را بیازماید که کدام یک از شما بهتر عمل می‌کنید، و او شکست‌ناپذیر و بخشنده است.
</Aside>

---

<Aside type="quran" title="3">
ٱلَّذِى خَلَقَ سَبْعَ سَمَٰوَٰتٍۢ طِبَاقًۭا ۖ مَّا تَرَىٰ فِى خَلْقِ ٱلرَّحْمَٰنِ مِن تَفَٰوُتٍۢ ۖ فَٱرْجِعِ ٱلْبَصَرَ هَلْ تَرَىٰ مِن فُطُورٍۢ
</Aside>

<Aside type="reference" title="3">
همان کسی که هفت آسمان را بر فراز یکدیگر آفرید؛ در آفرینش خداوند رحمان هیچ تضادّ و عیبی نمی‌بینی! بار دیگر نگاه کن، آیا هیچ شکاف و خللی مشاهده می‌کنی؟!
</Aside>

---

<Aside type="quran" title="4">
ثُمَّ ٱرْجِعِ ٱلْبَصَرَ كَرَّتَيْنِ يَنقَلِبْ إِلَيْكَ ٱلْبَصَرُ خَاسِئًۭا وَهُوَ حَسِيرٌۭ
</Aside>

<Aside type="reference" title="4">
بار دیگر (به عالم هستی) نگاه کن، سرانجام چشمانت (در جستجوی خلل و نقصان ناکام مانده) به سوی تو باز می‌گردد در حالی که خسته و ناتوان است!
</Aside>

---

<Aside type="quran" title="5">
وَلَقَدْ زَيَّنَّا ٱلسَّمَآءَ ٱلدُّنْيَا بِمَصَٰبِيحَ وَجَعَلْنَٰهَا رُجُومًۭا لِّلشَّيَٰطِينِ ۖ وَأَعْتَدْنَا لَهُمْ عَذَابَ ٱلسَّعِيرِ
</Aside>

<Aside type="reference" title="5">
ما آسمان پایین (نزدیک) را با چراغهای فروزانی زینت بخشیدیم، و آنها [= شهابها ] را تیرهایی برای شیاطین قرار دادیم، و برای آنان عذاب آتش فروزان فراهم ساختیم!
</Aside>

---

<Aside type="quran" title="6">
وَلِلَّذِينَ كَفَرُوا۟ بِرَبِّهِمْ عَذَابُ جَهَنَّمَ ۖ وَبِئْسَ ٱلْمَصِيرُ
</Aside>

<Aside type="reference" title="6">
و برای کسانی که به پروردگارشان کافر شدند عذاب جهنّم است، و بد فرجامی است!
</Aside>

---

<Aside type="quran" title="7">
إِذَآ أُلْقُوا۟ فِيهَا سَمِعُوا۟ لَهَا شَهِيقًۭا وَهِىَ تَفُورُ
</Aside>

<Aside type="reference" title="7">
هنگامی که در آن افکنده شوند صدای وحشتناکی از آن می‌شنوند، و این در حالی است که پیوسته می‌جوشد!
</Aside>

---

<Aside type="quran" title="8">
تَكَادُ تَمَيَّزُ مِنَ ٱلْغَيْظِ ۖ كُلَّمَآ أُلْقِىَ فِيهَا فَوْجٌۭ سَأَلَهُمْ خَزَنَتُهَآ أَلَمْ يَأْتِكُمْ نَذِيرٌۭ
</Aside>

<Aside type="reference" title="8">
نزدیک است (دوزخ) از شدّت غضب پاره پاره شود؛ هر زمان که گروهی در آن افکنده می‌شوند، نگهبانان دوزخ از آنها می‌پرسند: «مگر بیم‌دهنده الهی به سراغ شما نیامد؟!»
</Aside>

---

<Aside type="quran" title="9">
قَالُوا۟ بَلَىٰ قَدْ جَآءَنَا نَذِيرٌۭ فَكَذَّبْنَا وَقُلْنَا مَا نَزَّلَ ٱللَّهُ مِن شَىْءٍ إِنْ أَنتُمْ إِلَّا فِى ضَلَٰلٍۢ كَبِيرٍۢ
</Aside>

<Aside type="reference" title="9">
می‌گویند: «آری، بیم‌دهنده به سراغ ما آمد، ولی ما او را تکذیب کردیم و گفتیم: خداوند هرگز چیزی نازل نکرده، و شما در گمراهی بزرگی هستید!»
</Aside>

---

<Aside type="quran" title="10">
وَقَالُوا۟ لَوْ كُنَّا نَسْمَعُ أَوْ نَعْقِلُ مَا كُنَّا فِىٓ أَصْحَٰبِ ٱلسَّعِيرِ
</Aside>

<Aside type="reference" title="10">
و می‌گویند: «اگر ما گوش شنوا داشتیم یا تعقّل می‌کردیم، در میان دوزخیان نبودیم!»
</Aside>

---

<Aside type="quran" title="11">
فَٱعْتَرَفُوا۟ بِذَنۢبِهِمْ فَسُحْقًۭا لِّأَصْحَٰبِ ٱلسَّعِيرِ
</Aside>

<Aside type="reference" title="11">
اینجاست که به گناه خود اعتراف می‌کنند؛ دور باشند دوزخیان از رحمت خدا!
</Aside>

---

<Aside type="quran" title="12">
إِنَّ ٱلَّذِينَ يَخْشَوْنَ رَبَّهُم بِٱلْغَيْبِ لَهُم مَّغْفِرَةٌۭ وَأَجْرٌۭ كَبِيرٌۭ
</Aside>

<Aside type="reference" title="12">
(امّا) کسانی که از پروردگارشان در نهان می‌ترسند، مسلّماً آمرزش و پاداش بزرگی دارند!
</Aside>

---

<Aside type="quran" title="13">
وَأَسِرُّوا۟ قَوْلَكُمْ أَوِ ٱجْهَرُوا۟ بِهِۦٓ ۖ إِنَّهُۥ عَلِيمٌۢ بِذَاتِ ٱلصُّدُورِ
</Aside>

<Aside type="reference" title="13">
گفتار خود را پنهان کنید یا آشکار (تفاوتی نمی‌کند)، او به آنچه در سینه‌هاست آگاه است!
</Aside>

---

<Aside type="quran" title="14">
أَلَا يَعْلَمُ مَنْ خَلَقَ وَهُوَ ٱللَّطِيفُ ٱلْخَبِيرُ
</Aside>

<Aside type="reference" title="14">
آیا آن کسی که موجودات را آفریده از حال آنها آگاه نیست؟! در حالی که او (از اسرار دقیق) باخبر و آگاه است!
</Aside>

---

<Aside type="quran" title="15">
هُوَ ٱلَّذِى جَعَلَ لَكُمُ ٱلْأَرْضَ ذَلُولًۭا فَٱمْشُوا۟ فِى مَنَاكِبِهَا وَكُلُوا۟ مِن رِّزْقِهِۦ ۖ وَإِلَيْهِ ٱلنُّشُورُ
</Aside>

<Aside type="reference" title="15">
او کسی است که زمین را برای شما رام کرد، بر شانه‌های آن راه بروید و از روزیهای خداوند بخورید؛ و بازگشت و اجتماع همه به سوی اوست!
</Aside>

---

<Aside type="quran" title="16">
ءَأَمِنتُم مَّن فِى ٱلسَّمَآءِ أَن يَخْسِفَ بِكُمُ ٱلْأَرْضَ فَإِذَا هِىَ تَمُورُ
</Aside>

<Aside type="reference" title="16">
آیا خود را از عذاب کسی که حاکم بر آسمان است در امان می‌دانید که دستور دهد زمین بشکافد و شما را فرو برد و به لرزش خود ادامه دهد؟!
</Aside>

---

<Aside type="quran" title="17">
أَمْ أَمِنتُم مَّن فِى ٱلسَّمَآءِ أَن يُرْسِلَ عَلَيْكُمْ حَاصِبًۭا ۖ فَسَتَعْلَمُونَ كَيْفَ نَذِيرِ
</Aside>

<Aside type="reference" title="17">
یا خود را از عذاب خداوند آسمان در امان می‌دانید که تندبادی پر از سنگریزه بر شما فرستد؟! و بزودی خواهید دانست تهدیدهای من چگونه است!
</Aside>

---

<Aside type="quran" title="18">
وَلَقَدْ كَذَّبَ ٱلَّذِينَ مِن قَبْلِهِمْ فَكَيْفَ كَانَ نَكِيرِ
</Aside>

<Aside type="reference" title="18">
کسانی که پیش از آنان بودند (آیات الهی را) تکذیب کردند، امّا (ببین) مجازات من چگونه بود!
</Aside>

---

<Aside type="quran" title="19">
أَوَلَمْ يَرَوْا۟ إِلَى ٱلطَّيْرِ فَوْقَهُمْ صَٰٓفَّٰتٍۢ وَيَقْبِضْنَ ۚ مَا يُمْسِكُهُنَّ إِلَّا ٱلرَّحْمَٰنُ ۚ إِنَّهُۥ بِكُلِّ شَىْءٍۭ بَصِيرٌ
</Aside>

<Aside type="reference" title="19">
آیا به پرندگانی که بالای سرشان است، و گاه بالهای خود را گسترده و گاه جمع می‌کنند، نگاه نکردند؟! جز خداوند رحمان کسی آنها را بر فراز آسمان نگه نمی‌دارد، چرا که او به هر چیز بیناست!
</Aside>

---

<Aside type="quran" title="20">
أَمَّنْ هَٰذَا ٱلَّذِى هُوَ جُندٌۭ لَّكُمْ يَنصُرُكُم مِّن دُونِ ٱلرَّحْمَٰنِ ۚ إِنِ ٱلْكَٰفِرُونَ إِلَّا فِى غُرُورٍ
</Aside>

<Aside type="reference" title="20">
آیا این کسی که لشکر شماست می‌تواند شما را در برابر خداوند یاری دهد؟ ولی کافران تنها گرفتار فریبند!
</Aside>

---

<Aside type="quran" title="21">
أَمَّنْ هَٰذَا ٱلَّذِى يَرْزُقُكُمْ إِنْ أَمْسَكَ رِزْقَهُۥ ۚ بَل لَّجُّوا۟ فِى عُتُوٍّۢ وَنُفُورٍ
</Aside>

<Aside type="reference" title="21">
یا آن کسی که شما را روزی می‌دهد، اگر روزیش را بازدارد (چه کسی می‌تواند نیاز شما را تأمین کند)؟! ولی آنها در سرکشی و فرار از حقیقت لجاجت می‌ورزند!
</Aside>

---

<Aside type="quran" title="22">
أَفَمَن يَمْشِى مُكِبًّا عَلَىٰ وَجْهِهِۦٓ أَهْدَىٰٓ أَمَّن يَمْشِى سَوِيًّا عَلَىٰ صِرَٰطٍۢ مُّسْتَقِيمٍۢ
</Aside>

<Aside type="reference" title="22">
آیا کسی که به رو افتاده حرکت می‌کند به هدایت نزدیکتر است یا کسی که راست‌قامت در صراط مستقیم گام برمی‌دارد؟!
</Aside>

---

<Aside type="quran" title="23">
قُلْ هُوَ ٱلَّذِىٓ أَنشَأَكُمْ وَجَعَلَ لَكُمُ ٱلسَّمْعَ وَٱلْأَبْصَٰرَ وَٱلْأَفْـِٔدَةَ ۖ قَلِيلًۭا مَّا تَشْكُرُونَ
</Aside>

<Aside type="reference" title="23">
بگو: «او کسی است که شما را آفرید و برای شما گوش و چشم و قلب قرار داد؛ امّا کمتر سپاسگزاری می‌کنید!»
</Aside>

---

<Aside type="quran" title="24">
قُلْ هُوَ ٱلَّذِى ذَرَأَكُمْ فِى ٱلْأَرْضِ وَإِلَيْهِ تُحْشَرُونَ
</Aside>

<Aside type="reference" title="24">
بگو: «او کسی است که شما را در زمین آفرید و به سوی او محشور می‌شوید!»
</Aside>

---

<Aside type="quran" title="25">
وَيَقُولُونَ مَتَىٰ هَٰذَا ٱلْوَعْدُ إِن كُنتُمْ صَٰدِقِينَ
</Aside>

<Aside type="reference" title="25">
آنها می‌گویند: «اگر راست می‌گویید این وعده قیامت چه زمانی است؟!»
</Aside>

---

<Aside type="quran" title="26">
قُلْ إِنَّمَا ٱلْعِلْمُ عِندَ ٱللَّهِ وَإِنَّمَآ أَنَا۠ نَذِيرٌۭ مُّبِينٌۭ
</Aside>

<Aside type="reference" title="26">
بگو: «علم آن تنها نزد خداست؛ و من فقط بیم‌دهنده آشکاری هستم!»
</Aside>

---

<Aside type="quran" title="27">
فَلَمَّا رَأَوْهُ زُلْفَةًۭ سِيٓـَٔتْ وُجُوهُ ٱلَّذِينَ كَفَرُوا۟ وَقِيلَ هَٰذَا ٱلَّذِى كُنتُم بِهِۦ تَدَّعُونَ
</Aside>

<Aside type="reference" title="27">
هنگامی که آن (وعده الهی) را از نزدیک می‌بینند، صورت کافران زشت و سیاه می‌گردد، و به آنها گفته می‌شود: «این همان چیزی است که تقاضای آن را داشتید»!
</Aside>

---

<Aside type="quran" title="28">
قُلْ أَرَءَيْتُمْ إِنْ أَهْلَكَنِىَ ٱللَّهُ وَمَن مَّعِىَ أَوْ رَحِمَنَا فَمَن يُجِيرُ ٱلْكَٰفِرِينَ مِنْ عَذَابٍ أَلِيمٍۢ
</Aside>

<Aside type="reference" title="28">
بگو: «به من خبر دهید اگر خداوند مرا و تمام کسانی را که با من هستند هلاک کند، یا مورد ترحّم قرار دهد، چه کسی کافران را از عذاب دردناک پناه می‌دهد؟!»
</Aside>

---

<Aside type="quran" title="29">
قُلْ هُوَ ٱلرَّحْمَٰنُ ءَامَنَّا بِهِۦ وَعَلَيْهِ تَوَكَّلْنَا ۖ فَسَتَعْلَمُونَ مَنْ هُوَ فِى ضَلَٰلٍۢ مُّبِينٍۢ
</Aside>

<Aside type="reference" title="29">
بگو: «او خداوند رحمان است، ما به او ایمان آورده و بر او توکّل کرده‌ایم؛ و بزودی می‌دانید چه کسی در گمراهی آشکار است!»
</Aside>

---

<Aside type="quran" title="30">
قُلْ أَرَءَيْتُمْ إِنْ أَصْبَحَ مَآؤُكُمْ غَوْرًۭا فَمَن يَأْتِيكُم بِمَآءٍۢ مَّعِينٍۭ
</Aside>

<Aside type="reference" title="30">
بگو: «به من خبر دهید اگر آبهای (سرزمین) شما در زمین فرو رود، چه کسی می‌تواند آب جاری و گوارا در دسترس شما قرار دهد؟!»
</Aside>
