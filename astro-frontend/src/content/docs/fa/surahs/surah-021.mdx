<Aside type="quran" title="1">
ٱقْتَرَبَ لِلنَّاسِ حِسَابُهُمْ وَهُمْ فِى غَفْلَةٍۢ مُّعْرِضُونَ
</Aside>

<Aside type="reference" title="1">
حساب مردم به آنان نزدیک شده، در حالی که در غفلتند و روی گردانند!
</Aside>

---

<Aside type="quran" title="2">
مَا يَأْتِيهِم مِّن ذِكْرٍۢ مِّن رَّبِّهِم مُّحْدَثٍ إِلَّا ٱسْتَمَعُوهُ وَهُمْ يَلْعَبُونَ
</Aside>

<Aside type="reference" title="2">
هیچ یادآوری تازه‌ای از طرف پروردگارشان برای آنها نمی‌آید، مگر آنکه با بازی (و شوخی) به آن گوش می‌دهند!
</Aside>

---

<Aside type="quran" title="3">
لَاهِيَةًۭ قُلُوبُهُمْ ۗ وَأَسَرُّوا۟ ٱلنَّجْوَى ٱلَّذِينَ ظَلَمُوا۟ هَلْ هَٰذَآ إِلَّا بَشَرٌۭ مِّثْلُكُمْ ۖ أَفَتَأْتُونَ ٱلسِّحْرَ وَأَنتُمْ تُبْصِرُونَ
</Aside>

<Aside type="reference" title="3">
این در حالی است که دلهایشان در لهو و بی‌خبری فرو رفته است! و ستمگران پنهانی نجوا کردند (و گفتند): «آیا جز این است که او بشری همانند شماست؟! آیا به سراغ سحر می‌روید، با اینکه (چشم دارید و) می‌بینید؟!
</Aside>

---

<Aside type="quran" title="4">
قَالَ رَبِّى يَعْلَمُ ٱلْقَوْلَ فِى ٱلسَّمَآءِ وَٱلْأَرْضِ ۖ وَهُوَ ٱلسَّمِيعُ ٱلْعَلِيمُ
</Aside>

<Aside type="reference" title="4">
(پیامبر) گفت: «پروردگارم همه سخنان را، چه در آسمان باشد و چه در زمین، می‌داند؛ و او شنوا و داناست!»
</Aside>

---

<Aside type="quran" title="5">
بَلْ قَالُوٓا۟ أَضْغَٰثُ أَحْلَٰمٍۭ بَلِ ٱفْتَرَىٰهُ بَلْ هُوَ شَاعِرٌۭ فَلْيَأْتِنَا بِـَٔايَةٍۢ كَمَآ أُرْسِلَ ٱلْأَوَّلُونَ
</Aside>

<Aside type="reference" title="5">
آنها گفتند: «(آنچه محمّد (ص) آورده وحی نیست؛) بلکه خوابهایی آشفته است! اصلاً آن را بدروغ به خدا بسته؛ نه، بلکه او یک شاعر است! (اگر راست می‌گوید) باید معجزه‌ای برای ما بیاورد؛ همان گونه که پیامبران پیشین (با معجزات) فرستاده شدند!»
</Aside>

---

<Aside type="quran" title="6">
مَآ ءَامَنَتْ قَبْلَهُم مِّن قَرْيَةٍ أَهْلَكْنَٰهَآ ۖ أَفَهُمْ يُؤْمِنُونَ
</Aside>

<Aside type="reference" title="6">
تمام آبادیهایی که پیش از اینها هلاک کردیم (تقاضای معجزات گوناگون کردند،و خواسته آنان عملی شد، ولی) هرگز ایمان نیاوردند؛ آیا اینها ایمان می‌آورند؟!
</Aside>

---

<Aside type="quran" title="7">
وَمَآ أَرْسَلْنَا قَبْلَكَ إِلَّا رِجَالًۭا نُّوحِىٓ إِلَيْهِمْ ۖ فَسْـَٔلُوٓا۟ أَهْلَ ٱلذِّكْرِ إِن كُنتُمْ لَا تَعْلَمُونَ
</Aside>

<Aside type="reference" title="7">
ما پیش از تو، جز مردانی که به آنان وحی می‌کردیم، نفرستادیم! (همه انسان بودند، و از جنس بشر!) اگر نمی‌دانید، از آگاهان بپرسید.
</Aside>

---

<Aside type="quran" title="8">
وَمَا جَعَلْنَٰهُمْ جَسَدًۭا لَّا يَأْكُلُونَ ٱلطَّعَامَ وَمَا كَانُوا۟ خَٰلِدِينَ
</Aside>

<Aside type="reference" title="8">
آنان را پیکرهایی که غذا نخورند قرار ندادیم! عمر جاویدان هم نداشتند!
</Aside>

---

<Aside type="quran" title="9">
ثُمَّ صَدَقْنَٰهُمُ ٱلْوَعْدَ فَأَنجَيْنَٰهُمْ وَمَن نَّشَآءُ وَأَهْلَكْنَا ٱلْمُسْرِفِينَ
</Aside>

<Aside type="reference" title="9">
سپس وعده‌ای را که به آنان داده بودیم، وفا کردیم! آنها و هر کس را که می‌خواستیم (از چنگ دشمنانشان) نجات دادیم؛ و مسرفان را هلاک نمودیم!
</Aside>

---

<Aside type="quran" title="10">
لَقَدْ أَنزَلْنَآ إِلَيْكُمْ كِتَٰبًۭا فِيهِ ذِكْرُكُمْ ۖ أَفَلَا تَعْقِلُونَ
</Aside>

<Aside type="reference" title="10">
ما بر شما کتابی نازل کردیم که وسیله تذکّر (و بیداری) شما در آن است! آیا نمی‌فهمید؟!
</Aside>

---

<Aside type="quran" title="11">
وَكَمْ قَصَمْنَا مِن قَرْيَةٍۢ كَانَتْ ظَالِمَةًۭ وَأَنشَأْنَا بَعْدَهَا قَوْمًا ءَاخَرِينَ
</Aside>

<Aside type="reference" title="11">
چه بسیار آبادیهای ستمگری را در هم شکستیم؛ و بعد از آنها، قوم دیگری روی کار آوردیم!
</Aside>

---

<Aside type="quran" title="12">
فَلَمَّآ أَحَسُّوا۟ بَأْسَنَآ إِذَا هُم مِّنْهَا يَرْكُضُونَ
</Aside>

<Aside type="reference" title="12">
هنگامی که عذاب ما را احساس کردند، ناگهان پا به فرار گذاشتند!
</Aside>

---

<Aside type="quran" title="13">
لَا تَرْكُضُوا۟ وَٱرْجِعُوٓا۟ إِلَىٰ مَآ أُتْرِفْتُمْ فِيهِ وَمَسَٰكِنِكُمْ لَعَلَّكُمْ تُسْـَٔلُونَ
</Aside>

<Aside type="reference" title="13">
(گفتیم:) فرار نکنید؛ و به زندگی پر ناز و نعمت، و به مسکنهای پر زرق و برقتان بازگردید! شاید (سائلان بیایند و) از شما تقاضا کنند (شما هم آنان را محروم بازگردانید)!
</Aside>

---

<Aside type="quran" title="14">
قَالُوا۟ يَٰوَيْلَنَآ إِنَّا كُنَّا ظَٰلِمِينَ
</Aside>

<Aside type="reference" title="14">
گفتند: «ای وای بر ما! به یقین ما ستمگر بودیم!»
</Aside>

---

<Aside type="quran" title="15">
فَمَا زَالَت تِّلْكَ دَعْوَىٰهُمْ حَتَّىٰ جَعَلْنَٰهُمْ حَصِيدًا خَٰمِدِينَ
</Aside>

<Aside type="reference" title="15">
و همچنان این سخن را تکرار می‌کردند، تا آنها را درو کرده و خاموش ساختیم!
</Aside>

---

<Aside type="quran" title="16">
وَمَا خَلَقْنَا ٱلسَّمَآءَ وَٱلْأَرْضَ وَمَا بَيْنَهُمَا لَٰعِبِينَ
</Aside>

<Aside type="reference" title="16">
ما آسمان و زمین، و آنچه را در میان آنهاست از روی بازی نیافریدیم!
</Aside>

---

<Aside type="quran" title="17">
لَوْ أَرَدْنَآ أَن نَّتَّخِذَ لَهْوًۭا لَّٱتَّخَذْنَٰهُ مِن لَّدُنَّآ إِن كُنَّا فَٰعِلِينَ
</Aside>

<Aside type="reference" title="17">
(بفرض محال) اگر می‌خواستیم سرگرمی انتخاب کنیم، چیزی متناسب خود انتخاب می‌کردیم!
</Aside>

---

<Aside type="quran" title="18">
بَلْ نَقْذِفُ بِٱلْحَقِّ عَلَى ٱلْبَٰطِلِ فَيَدْمَغُهُۥ فَإِذَا هُوَ زَاهِقٌۭ ۚ وَلَكُمُ ٱلْوَيْلُ مِمَّا تَصِفُونَ
</Aside>

<Aside type="reference" title="18">
بلکه ما حق را بر سر باطل می‌کوبیم تا آن را هلاک سازد؛ و این گونه، باطل محو و نابود می‌شود! امّا وای بر شما از توصیفی که (درباره خدا و هدف آفرینش)می‌کنید!
</Aside>

---

<Aside type="quran" title="19">
وَلَهُۥ مَن فِى ٱلسَّمَٰوَٰتِ وَٱلْأَرْضِ ۚ وَمَنْ عِندَهُۥ لَا يَسْتَكْبِرُونَ عَنْ عِبَادَتِهِۦ وَلَا يَسْتَحْسِرُونَ
</Aside>

<Aside type="reference" title="19">
از آن اوست آنان که در آسمانها و زمینند! و آنها که نزد اویند [= فرشتگان‌] هیچ گاه از عبادتش استکبار نمی‌ورزند، و هرگز خسته نمی‌شوند.
</Aside>

---

<Aside type="quran" title="20">
يُسَبِّحُونَ ٱلَّيْلَ وَٱلنَّهَارَ لَا يَفْتُرُونَ
</Aside>

<Aside type="reference" title="20">
(تمام) شب و روز را تسبیح می‌گویند؛ و سست نمی‌گردند.
</Aside>

---

<Aside type="quran" title="21">
أَمِ ٱتَّخَذُوٓا۟ ءَالِهَةًۭ مِّنَ ٱلْأَرْضِ هُمْ يُنشِرُونَ
</Aside>

<Aside type="reference" title="21">
آیا آنها خدایانی از زمین برگزیدند که (خلق می‌کنند و) منتشر می‌سازند؟!
</Aside>

---

<Aside type="quran" title="22">
لَوْ كَانَ فِيهِمَآ ءَالِهَةٌ إِلَّا ٱللَّهُ لَفَسَدَتَا ۚ فَسُبْحَٰنَ ٱللَّهِ رَبِّ ٱلْعَرْشِ عَمَّا يَصِفُونَ
</Aside>

<Aside type="reference" title="22">
اگر در آسمان و زمین، جز «اللّه» خدایان دیگری بود، فاسد می‌شدند (و نظام جهان به هم می‌خورد)! منزه است خداوند پروردگار عرش، از توصیفی که آنها می‌کنند!
</Aside>

---

<Aside type="quran" title="23">
لَا يُسْـَٔلُ عَمَّا يَفْعَلُ وَهُمْ يُسْـَٔلُونَ
</Aside>

<Aside type="reference" title="23">
هیچ کس نمی‌تواند بر کار او خرده بگیرد؛ ولی در کارهای آنها، جای سؤال و ایراد است!
</Aside>

---

<Aside type="quran" title="24">
أَمِ ٱتَّخَذُوا۟ مِن دُونِهِۦٓ ءَالِهَةًۭ ۖ قُلْ هَاتُوا۟ بُرْهَٰنَكُمْ ۖ هَٰذَا ذِكْرُ مَن مَّعِىَ وَذِكْرُ مَن قَبْلِى ۗ بَلْ أَكْثَرُهُمْ لَا يَعْلَمُونَ ٱلْحَقَّ ۖ فَهُم مُّعْرِضُونَ
</Aside>

<Aside type="reference" title="24">
آیا آنها معبودانی جز خدا برگزیدند؟! بگو: «دلیلتان را بیاورید! این سخن کسانی است که با من هستند، و سخن کسانی [= پیامبرانی‌] است که پیش از من بودند!» امّا بیشتر آنها حق را نمی‌دانند؛ و به همین دلیل (از آن) روی گردانند.
</Aside>

---

<Aside type="quran" title="25">
وَمَآ أَرْسَلْنَا مِن قَبْلِكَ مِن رَّسُولٍ إِلَّا نُوحِىٓ إِلَيْهِ أَنَّهُۥ لَآ إِلَٰهَ إِلَّآ أَنَا۠ فَٱعْبُدُونِ
</Aside>

<Aside type="reference" title="25">
ما پیش از تو هیچ پیامبری را نفرستادیم مگر اینکه به او وحی کردیم که: «معبودی جز من نیست؛ پس تنها مرا پرستش کنید.»
</Aside>

---

<Aside type="quran" title="26">
وَقَالُوا۟ ٱتَّخَذَ ٱلرَّحْمَٰنُ وَلَدًۭا ۗ سُبْحَٰنَهُۥ ۚ بَلْ عِبَادٌۭ مُّكْرَمُونَ
</Aside>

<Aside type="reference" title="26">
آنها گفتند: «خداوند رحمان فرزندی برای خود انتخاب کرده است»! او منزه است (از این عیب و نقص)؛ آنها [= فرشتگان‌] بندگان شایسته اویند.
</Aside>

---

<Aside type="quran" title="27">
لَا يَسْبِقُونَهُۥ بِٱلْقَوْلِ وَهُم بِأَمْرِهِۦ يَعْمَلُونَ
</Aside>

<Aside type="reference" title="27">
هرگز در سخن بر او پیشی نمی‌گیرند؛ و (پیوسته) به فرمان او عمل می‌کنند.
</Aside>

---

<Aside type="quran" title="28">
يَعْلَمُ مَا بَيْنَ أَيْدِيهِمْ وَمَا خَلْفَهُمْ وَلَا يَشْفَعُونَ إِلَّا لِمَنِ ٱرْتَضَىٰ وَهُم مِّنْ خَشْيَتِهِۦ مُشْفِقُونَ
</Aside>

<Aside type="reference" title="28">
او اعمال امروز و آینده و اعمال گذشته آنها را می‌داند؛ و آنها جز برای کسی که خدا راضی (به شفاعت برای او) است شفاعت نمی‌کنند؛ و از ترس او بیمناکند.
</Aside>

---

<Aside type="quran" title="29">
۞ وَمَن يَقُلْ مِنْهُمْ إِنِّىٓ إِلَٰهٌۭ مِّن دُونِهِۦ فَذَٰلِكَ نَجْزِيهِ جَهَنَّمَ ۚ كَذَٰلِكَ نَجْزِى ٱلظَّٰلِمِينَ
</Aside>

<Aside type="reference" title="29">
و هر کس از آنها بگوید: «من جز خدا، معبودی دیگرم»، کیفر او را جهنم می‌دهیم! و ستمگران را این گونه کیفر خواهیم داد.
</Aside>

---

<Aside type="quran" title="30">
أَوَلَمْ يَرَ ٱلَّذِينَ كَفَرُوٓا۟ أَنَّ ٱلسَّمَٰوَٰتِ وَٱلْأَرْضَ كَانَتَا رَتْقًۭا فَفَتَقْنَٰهُمَا ۖ وَجَعَلْنَا مِنَ ٱلْمَآءِ كُلَّ شَىْءٍ حَىٍّ ۖ أَفَلَا يُؤْمِنُونَ
</Aside>

<Aside type="reference" title="30">
آیا کافران ندیدند که آسمانها و زمین به هم پیوسته بودند، و ما آنها را از یکدیگر باز کردیم؛ و هر چیز زنده‌ای را از آب قرار دادیم؟! آیا ایمان نمی‌آورند؟!
</Aside>

---

<Aside type="quran" title="31">
وَجَعَلْنَا فِى ٱلْأَرْضِ رَوَٰسِىَ أَن تَمِيدَ بِهِمْ وَجَعَلْنَا فِيهَا فِجَاجًۭا سُبُلًۭا لَّعَلَّهُمْ يَهْتَدُونَ
</Aside>

<Aside type="reference" title="31">
و در زمین، کوه‌های ثابت و پابرجایی قرار دادیم، مبادا آنها را بلرزاند! و در آن، درّه‌ها و راه‌هایی قرار دادیم تا هدایت شوند!
</Aside>

---

<Aside type="quran" title="32">
وَجَعَلْنَا ٱلسَّمَآءَ سَقْفًۭا مَّحْفُوظًۭا ۖ وَهُمْ عَنْ ءَايَٰتِهَا مُعْرِضُونَ
</Aside>

<Aside type="reference" title="32">
و آسمان را سقف محفوظی قرار دادیم؛ ولی آنها از آیات آن روی‌گردانند.
</Aside>

---

<Aside type="quran" title="33">
وَهُوَ ٱلَّذِى خَلَقَ ٱلَّيْلَ وَٱلنَّهَارَ وَٱلشَّمْسَ وَٱلْقَمَرَ ۖ كُلٌّۭ فِى فَلَكٍۢ يَسْبَحُونَ
</Aside>

<Aside type="reference" title="33">
او کسی است که شب و روز و خورشید و ماه را آفرید؛ هر یک در مداری در حرکتند!
</Aside>
