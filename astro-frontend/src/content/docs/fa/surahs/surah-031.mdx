<Aside type="reference" title="4">

<Aside type="quran" title="5">
أُو۟لَٰٓئِكَ عَلَىٰ هُدًۭى مِّن رَّبِّهِمْ ۖ وَأُو۟لَٰٓئِكَ هُمُ ٱلْمُفْلِحُونَ

<Aside type="quran" title="6">
وَمِنَ ٱلنَّاسِ مَن يَشْتَرِى لَهْوَ ٱلْحَدِيثِ لِيُضِلَّ عَن سَبِيلِ ٱللَّهِ بِغَيْرِ عِلْمٍۢ وَيَتَّخِذَهَا هُزُوًا ۚ أُو۟لَٰٓئِكَ لَهُمْ عَذَابٌۭ مُّهِينٌۭ
</Aside>

<Aside type="reference" title="6">
و بعضی از مردم سخنان بیهوده را می‌خرند تا مردم را از روی نادانی، از راه خدا گمراه سازند و آیات الهی را به استهزا گیرند؛ برای آنان عذابی خوارکننده است!
</Aside>

<Aside type="quran" title="7">
وَإِذَا تُتْلَىٰ عَلَيْهِ ءَايَٰتُنَا وَلَّىٰ مُسْتَكْبِرًۭا كَأَن لَّمْ يَسْمَعْهَا كَأَنَّ فِىٓ أُذُنَيْهِ وَقْرًۭا ۖ فَبَشِّرْهُ بِعَذَابٍ أَلِيمٍ
</Aside>

<Aside type="reference" title="7">
و هنگامی که آیات ما بر او خوانده می‌شود، مستکبرانه روی برمی‌گرداند، گویی آن را نشنیده است؛ گویی اصلاً گوشهایش سنگین است! او را به عذابی دردناک بشارت ده!
</Aside>

---

<Aside type="quran" title="8">
إِنَّ ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّٰلِحَٰتِ لَهُمْ جَنَّٰتُ ٱلنَّعِيمِ
</Aside>

<Aside type="reference" title="8">
(ولی) کسانی که ایمان آورده و اعمال صالح انجام داده‌اند، باغهای پرنعمت بهشت از آن آنهاست؛
</Aside>

---

<Aside type="quran" title="9">
خَٰلِدِينَ فِيهَا ۖ وَعْدَ ٱللَّهِ حَقًّۭا ۚ وَهُوَ ٱلْعَزِيزُ ٱلْحَكِيمُ
</Aside>

<Aside type="reference" title="9">
جاودانه در آن خواهند ماند؛ این وعده حتمی الهی است؛ و اوست عزیز و حکیم (شکست‌ناپذیر و دانا).
</Aside>

---

<Aside type="quran" title="10">
خَلَقَ ٱلسَّمَٰوَٰتِ بِغَيْرِ عَمَدٍۢ تَرَوْنَهَا ۖ وَأَلْقَىٰ فِى ٱلْأَرْضِ رَوَٰسِىَ أَن تَمِيدَ بِكُمْ وَبَثَّ فِيهَا مِن كُلِّ دَآبَّةٍۢ ۚ وَأَنزَلْنَا مِنَ ٱلسَّمَآءِ مَآءًۭ فَأَنۢبَتْنَا فِيهَا مِن كُلِّ زَوْجٍۢ كَرِيمٍ
</Aside>

<Aside type="reference" title="10">
(او) آسمانها را بدون ستونی که آن را ببینید آفرید، و در زمین کوه‌هایی افکند تا شما را نلرزاند (و جایگاه شما آرام باشد) و از هر گونه جنبنده‌ای روی آن منتشر ساخت؛ و از آسمان آبی نازل کردیم و بوسیله آن در روی زمین انواع گوناگونی از جفتهای گیاهان پر ارزش رویاندیم.
</Aside>

<Aside type="quran" title="11">
هَٰذَا خَلْقُ ٱللَّهِ فَأَرُونِى مَاذَا خَلَقَ ٱلَّذِينَ مِن دُونِهِۦ ۚ بَلِ ٱلظَّٰلِمُونَ فِى ضَلَٰلٍۢ مُّبِينٍۢ
</Aside>

<Aside type="reference" title="11">
این آفرینش خداست؛ امّا به من نشان دهید معبودانی غیر او چه چیز را آفریده‌اند؟! ولی ظالمان در گمراهی آشکارند.
</Aside>

<Aside type="quran" title="12">
وَلَقَدْ ءَاتَيْنَا لُقْمَٰنَ ٱلْحِكْمَةَ أَنِ ٱشْكُرْ لِلَّهِ ۚ وَمَن يَشْكُرْ فَإِنَّمَا يَشْكُرُ لِنَفْسِهِۦ ۖ وَمَن كَفَرَ فَإِنَّ ٱللَّهَ غَنِىٌّ حَمِيدٌۭ
</Aside>

<Aside type="reference" title="12">
ما به لقمان حکمت دادیم؛ (و به او گفتیم:) شکر خدا را بجای آور هر کس شکرگزاری کند، تنها به سود خویش شکر کرده؛ و آن کس که کفران کند، (زیانی به خدا نمی‌رساند)؛ چرا که خداوند بی‌نیاز و ستوده است.
</Aside>

---

<Aside type="quran" title="13">
وَإِذْ قَالَ لُقْمَٰنُ لِٱبْنِهِۦ وَهُوَ يَعِظُهُۥ يَٰبُنَىَّ لَا تُشْرِكْ بِٱللَّهِ ۖ إِنَّ ٱلشِّرْكَ لَظُلْمٌ عَظِيمٌۭ
</Aside>

<Aside type="reference" title="13">
(به خاطر بیاور) هنگامی را که لقمان به فرزندش -در حالی که او را موعظه می‌کرد- گفت: «پسرم! چیزی را همتای خدا قرار مده که شرک، ظلم بزرگی است.»
</Aside>

---

<Aside type="quran" title="14">
وَوَصَّيْنَا ٱلْإِنسَٰنَ بِوَٰلِدَيْهِ حَمَلَتْهُ أُمُّهُۥ وَهْنًا عَلَىٰ وَهْنٍۢ وَفِصَٰلُهُۥ فِى عَامَيْنِ أَنِ ٱشْكُرْ لِى وَلِوَٰلِدَيْكَ إِلَىَّ ٱلْمَصِيرُ
</Aside>

<Aside type="reference" title="14">
و ما به انسان درباره پدر و مادرش سفارش کردیم؛ مادرش او را با ناتوانی روی ناتوانی حمل کرد (به هنگام بارداری هر روز رنج و ناراحتی تازه‌ای را متحمّل می‌شد)، و دوران شیرخوارگی او در دو سال پایان می‌یابد؛ (آری به او توصیه کردم) که برای من و برای پدر و مادرت شکر بجا آور که بازگشت (همه شما)به سوی من است!
</Aside>

<Aside type="quran" title="15">
وَإِن جَٰهَدَاكَ عَلَىٰٓ أَن تُشْرِكَ بِى مَا لَيْسَ لَكَ بِهِۦ عِلْمٌۭ فَلَا تُطِعْهُمَا ۖ وَصَاحِبْهُمَا فِى ٱلدُّنْيَا مَعْرُوفًۭا ۖ وَٱتَّبِعْ سَبِيلَ مَنْ أَنَابَ إِلَىَّ ۚ ثُمَّ إِلَىَّ مَرْجِعُكُمْ فَأُنَبِّئُكُم بِمَا كُنتُمْ تَعْمَلُونَ
</Aside>

<Aside type="reference" title="15">
و هرگاه آن دو، تلاش کنند که تو چیزی را همتای من قرار دهی، که از آن آگاهی نداری (بلکه می‌دانی باطل است)، از ایشان اطاعت مکن، ولی با آن دو، در دنیا به طرز شایسته‌ای رفتار کن؛ و از راه کسانی پیروی کن که توبه‌کنان به سوی من آمده‌اند؛ سپس بازگشت همه شما به سوی من است و من شما را از آنچه عمل می‌کردید آگاه می‌کنم.
</Aside>

<Aside type="quran" title="16">
يَٰبُنَىَّ إِنَّهَآ إِن تَكُ مِثْقَالَ حَبَّةٍۢ مِّنْ خَرْدَلٍۢ فَتَكُن فِى صَخْرَةٍ أَوْ فِى ٱلسَّمَٰوَٰتِ أَوْ فِى ٱلْأَرْضِ يَأْتِ بِهَا ٱللَّهُ ۚ إِنَّ ٱللَّهَ لَطِيفٌ خَبِيرٌۭ
</Aside>

<Aside type="reference" title="16">
پسرم! اگر به اندازه سنگینی دانه خردلی (کار نیک یا بد) باشد، و در دل سنگی یا در (گوشه‌ای از) آسمانها و زمین قرار گیرد، خداوند آن را (در قیامت برای حساب) می‌آورد؛ خداوند دقیق و آگاه است!
</Aside>

---

<Aside type="quran" title="17">
يَٰبُنَىَّ أَقِمِ ٱلصَّلَوٰةَ وَأْمُرْ بِٱلْمَعْرُوفِ وَٱنْهَ عَنِ ٱلْمُنكَرِ وَٱصْبِرْ عَلَىٰ مَآ أَصَابَكَ ۖ إِنَّ ذَٰلِكَ مِنْ عَزْمِ ٱلْأُمُورِ
</Aside>

<Aside type="reference" title="17">
پسرم! نماز را برپا دار، و امر به معروف و نهی از منکر کن، و در برابر مصایبی که به تو می‌رسد شکیبا باش که این از کارهای مهمّ است!
</Aside>

<Aside type="quran" title="18">
وَلَا تُصَعِّرْ خَدَّكَ لِلنَّاسِ وَلَا تَمْشِ فِى ٱلْأَرْضِ مَرَحًا ۖ إِنَّ ٱللَّهَ لَا يُحِبُّ كُلَّ مُخْتَالٍۢ فَخُورٍۢ
</Aside>

<Aside type="reference" title="18">
(پسرم!) با بی‌اعتنایی از مردم روی مگردان، و مغرورانه بر زمین راه مرو که خداوند هیچ متکبّر مغروری را دوست ندارد.
</Aside>

<Aside type="quran" title="19">
وَٱقْصِدْ فِى مَشْيِكَ وَٱغْضُضْ مِن صَوْتِكَ ۚ إِنَّ أَنكَرَ ٱلْأَصْوَٰتِ لَصَوْتُ ٱلْحَمِيرِ
</Aside>

<Aside type="reference" title="19">
(پسرم!) در راه‌رفتن، اعتدال را رعایت کن؛ از صدای خود بکاه (و هرگز فریاد مزن) که زشت‌ترین صداها صدای خران است.
</Aside>

<Aside type="quran" title="20">
أَلَمْ تَرَوْا۟ أَنَّ ٱللَّهَ سَخَّرَ لَكُم مَّا فِى ٱلسَّمَٰوَٰتِ وَمَا فِى ٱلْأَرْضِ وَأَسْبَغَ عَلَيْكُمْ نِعَمَهُۥ ظَٰهِرَةًۭ وَبَاطِنَةًۭ ۗ وَمِنَ ٱلنَّاسِ مَن يُجَٰدِلُ فِى ٱللَّهِ بِغَيْرِ عِلْمٍۢ وَلَا هُدًۭى وَلَا كِتَٰبٍۢ مُّنِيرٍۢ
</Aside>

<Aside type="reference" title="20">
آیا ندیدید خداوند آنچه را در آسمانها و زمین است مسخّر شما کرده، و نعمتهای آشکار و پنهان خود را به طور فراوان بر شما ارزانی داشته است؟! ولی بعضی از مردم بدون هیچ دانش و هدایت و کتاب روشنگری درباره خدا مجادله می‌کنند!
</Aside>

<Aside type="quran" title="21">
وَإِذَا قِيلَ لَهُمُ ٱتَّبِعُوا۟ مَآ أَنزَلَ ٱللَّهُ قَالُوا۟ بَلْ نَتَّبِعُ مَا وَجَدْنَا عَلَيْهِ ءَابَآءَنَآ ۚ أَوَلَوْ كَانَ ٱلشَّيْطَٰنُ يَدْعُوهُمْ إِلَىٰ عَذَابِ ٱلسَّعِيرِ
</Aside>

<Aside type="reference" title="21">
و هنگامی که به آنان گفته شود: «از آنچه خدا نازل کرده پیروی کنید!»، می‌گویند: «نه، بلکه ما از چیزی پیروی می‌کنیم که پدران خود را بر آن یافتیم!»آیا حتّی اگر شیطان آنان را دعوت به عذاب آتش فروزان کند (باز هم تبعیّت می‌کنند)؟!
</Aside>

<Aside type="quran" title="22">
۞ وَمَن يُسْلِمْ وَجْهَهُۥٓ إِلَى ٱللَّهِ وَهُوَ مُحْسِنٌۭ فَقَدِ ٱسْتَمْسَكَ بِٱلْعُرْوَةِ ٱلْوُثْقَىٰ ۗ وَإِلَى ٱللَّهِ عَٰقِبَةُ ٱلْأُمُورِ
</Aside>

<Aside type="reference" title="22">

</Aside>

<Aside type="quran" title="23">
وَمَن كَفَرَ فَلَا يَحْزُنكَ كُفْرُهُۥٓ ۚ إِلَيْنَا مَرْجِعُهُمْ فَنُنَبِّئُهُم بِمَا عَمِلُوٓا۟ ۚ إِنَّ ٱللَّهَ عَلِيمٌۢ بِذَاتِ ٱلصُّدُورِ
</Aside>

<Aside type="reference" title="23">
و کسی که کافر شود، کفر او تو را غمگین نسازد؛ بازگشت همه آنان به سوی ماست و ما آنها را از اعمالی که انجام داده‌اند (و نتایج شوم آن) آگاه خواهیم ساخت؛ خداوند به آنچه درون سینه‌هاست آگاه است.
</Aside>

<Aside type="quran" title="24">
نُمَتِّعُهُمْ قَلِيلًۭا ثُمَّ نَضْطَرُّهُمْ إِلَىٰ عَذَابٍ غَلِيظٍۢ
</Aside>

<Aside type="reference" title="24">
ما اندکی آنها را از متاع دنیا بهره‌مند می‌کنیم، سپس آنها را به تحمّل عذاب شدیدی وادار می‌سازیم!
</Aside>

<Aside type="quran" title="25">
وَلَئِن سَأَلْتَهُم مَّنْ خَلَقَ ٱلسَّمَٰوَٰتِ وَٱلْأَرْضَ لَيَقُولُنَّ ٱللَّهُ ۚ قُلِ ٱلْحَمْدُ لِلَّهِ ۚ بَلْ أَكْثَرُهُمْ لَا يَعْلَمُونَ
</Aside>

<Aside type="reference" title="25">

</Aside>

<Aside type="quran" title="26">

</Aside>

<Aside type="reference" title="26">
آنچه در آسمانها و زمین است از آن خداست، چرا که خداوند بی‌نیاز و شایسته ستایش است!
</Aside>

<Aside type="reference" title="29">

<Aside type="quran" title="32">

<Aside type="reference" title="32">
