<Aside type="quran" title="1">
لَآ أُقْسِمُ بِيَوْمِ ٱلْقِيَٰمَةِ
</Aside>

<Aside type="reference" title="1">
سوگند به روز قیامت،
</Aside>

---

<Aside type="quran" title="2">
وَلَآ أُقْسِمُ بِٱلنَّفْسِ ٱللَّوَّامَةِ
</Aside>

<Aside type="reference" title="2">
و سوگند به (نفس لوّامه و) وجدان بیدار و ملامتگر (که رستاخیز حقّ است)!
</Aside>

---

<Aside type="quran" title="3">
أَيَحْسَبُ ٱلْإِنسَٰنُ أَلَّن نَّجْمَعَ عِظَامَهُۥ
</Aside>

<Aside type="reference" title="3">
آیا انسان می‌پندارد که هرگز استخوانهای او را جمع نخواهیم کرد؟!
</Aside>

---

<Aside type="quran" title="4">
بَلَىٰ قَٰدِرِينَ عَلَىٰٓ أَن نُّسَوِّىَ بَنَانَهُۥ
</Aside>

<Aside type="reference" title="4">
آری قادریم که (حتّی خطوط سر) انگشتان او را موزون و مرتّب کنیم!
</Aside>

---

<Aside type="quran" title="5">
بَلْ يُرِيدُ ٱلْإِنسَٰنُ لِيَفْجُرَ أَمَامَهُۥ
</Aside>

<Aside type="reference" title="5">
(انسان شک در معاد ندارد) بلکه او می‌خواهد (آزاد باشد و بدون ترس از دادگاه قیامت) در تمام عمر گناه کند!
</Aside>

---

<Aside type="quran" title="6">
يَسْـَٔلُ أَيَّانَ يَوْمُ ٱلْقِيَٰمَةِ
</Aside>

<Aside type="reference" title="6">
(از این رو) می‌پرسد: «قیامت کی خواهد بود»!
</Aside>

---

<Aside type="quran" title="7">
فَإِذَا بَرِقَ ٱلْبَصَرُ
</Aside>

<Aside type="reference" title="7">
(بگو:) در آن هنگام که چشمها از شدّت وحشت به گردش درآید،
</Aside>

---

<Aside type="quran" title="8">
وَخَسَفَ ٱلْقَمَرُ
</Aside>

<Aside type="reference" title="8">
و ماه بی‌نور گردد،
</Aside>

---

<Aside type="quran" title="9">
وَجُمِعَ ٱلشَّمْسُ وَٱلْقَمَرُ
</Aside>

<Aside type="reference" title="9">
و خورشید و ماه یک جا جمع شوند،
</Aside>

---

<Aside type="quran" title="10">
يَقُولُ ٱلْإِنسَٰنُ يَوْمَئِذٍ أَيْنَ ٱلْمَفَرُّ
</Aside>

<Aside type="reference" title="10">
آن روز انسان می‌گوید: «راه فرار کجاست؟!»
</Aside>

---

<Aside type="quran" title="11">
كَلَّا لَا وَزَرَ
</Aside>

<Aside type="reference" title="11">
هرگز چنین نیست، راه فرار و پناهگاهی وجود ندارد!
</Aside>

---

<Aside type="quran" title="12">
إِلَىٰ رَبِّكَ يَوْمَئِذٍ ٱلْمُسْتَقَرُّ
</Aside>

<Aside type="reference" title="12">
آن روز قرارگاه نهایی تنها بسوی پروردگار تو است؛
</Aside>

---

<Aside type="quran" title="13">
يُنَبَّؤُا۟ ٱلْإِنسَٰنُ يَوْمَئِذٍۭ بِمَا قَدَّمَ وَأَخَّرَ
</Aside>

<Aside type="reference" title="13">
و در آن روز انسان را از تمام کارهایی که از پیش یا پس فرستاده آگاه می‌کنند!
</Aside>

---

<Aside type="quran" title="14">
بَلِ ٱلْإِنسَٰنُ عَلَىٰ نَفْسِهِۦ بَصِيرَةٌۭ
</Aside>

<Aside type="reference" title="14">
بلکه انسان خودش از وضع خود آگاه است،
</Aside>

---

<Aside type="quran" title="15">
وَلَوْ أَلْقَىٰ مَعَاذِيرَهُۥ
</Aside>

<Aside type="reference" title="15">
هر چند (در ظاهر) برای خود عذرهایی بتراشد!
</Aside>

---

<Aside type="quran" title="16">
لَا تُحَرِّكْ بِهِۦ لِسَانَكَ لِتَعْجَلَ بِهِۦٓ
</Aside>

<Aside type="reference" title="16">
زبانت را بخاطر عجله برای خواندن آن [= قرآن‌] حرکت مده،
</Aside>

---

<Aside type="quran" title="17">
إِنَّ عَلَيْنَا جَمْعَهُۥ وَقُرْءَانَهُۥ
</Aside>

<Aside type="reference" title="17">
چرا که جمع‌کردن و خواندن آن بر عهده ماست!
</Aside>

---

<Aside type="quran" title="18">
فَإِذَا قَرَأْنَٰهُ فَٱتَّبِعْ قُرْءَانَهُۥ
</Aside>

<Aside type="reference" title="18">
پس هر گاه آن را خواندیم، از خواندن آن پیروی کن!
</Aside>

---

<Aside type="quran" title="19">
ثُمَّ إِنَّ عَلَيْنَا بَيَانَهُۥ
</Aside>

<Aside type="reference" title="19">
سپس بیان و (توضیح) آن (نیز) بر عهده ماست!
</Aside>

---

<Aside type="quran" title="20">
كَلَّا بَلْ تُحِبُّونَ ٱلْعَاجِلَةَ
</Aside>

<Aside type="reference" title="20">
چنین نیست که شما می‌پندارید (و دلایل معاد را کافی نمی‌دانید)؛ بلکه شما دنیای زودگذر را دوست دارید (و هوسرانی بی‌قید و شرط را)!
</Aside>

---

<Aside type="quran" title="21">
وَتَذَرُونَ ٱلْءَاخِرَةَ
</Aside>

<Aside type="reference" title="21">
و آخرت را رها می‌کنید!
</Aside>

---

<Aside type="quran" title="22">
وُجُوهٌۭ يَوْمَئِذٍۢ نَّاضِرَةٌ
</Aside>

<Aside type="reference" title="22">
(آری) در آن روز صورتهایی شاداب و مسرور است،
</Aside>

---

<Aside type="quran" title="23">
إِلَىٰ رَبِّهَا نَاظِرَةٌۭ
</Aside>

<Aside type="reference" title="23">
و به پروردگارش می‌نگرد!
</Aside>

---

<Aside type="quran" title="24">
وَوُجُوهٌۭ يَوْمَئِذٍۭ بَاسِرَةٌۭ
</Aside>

<Aside type="reference" title="24">
و در آن روز صورتهایی عبوس و در هم کشیده است،
</Aside>

---

<Aside type="quran" title="25">
تَظُنُّ أَن يُفْعَلَ بِهَا فَاقِرَةٌۭ
</Aside>

<Aside type="reference" title="25">
زیرا می‌داند عذابی در پیش دارد که پشت را در هم می‌شکند!
</Aside>

---

<Aside type="quran" title="26">
كَلَّآ إِذَا بَلَغَتِ ٱلتَّرَاقِىَ
</Aside>

<Aside type="reference" title="26">
چنین نیست (که انسان می‌پندارد! او ایمان نمی‌آورد) تا موقعی که جان به گلوگاهش رسد،
</Aside>

---

<Aside type="quran" title="27">
وَقِيلَ مَنْ ۜ رَاقٍۢ
</Aside>

<Aside type="reference" title="27">
و گفته شود: «آیا کسی هست که (این بیمار را از مرگ) نجات دهد؟!»
</Aside>

---

<Aside type="quran" title="28">
وَظَنَّ أَنَّهُ ٱلْفِرَاقُ
</Aside>

<Aside type="reference" title="28">
و به جدائی از دنیا یقین پیدا کند،
</Aside>

---

<Aside type="quran" title="29">
وَٱلْتَفَّتِ ٱلسَّاقُ بِٱلسَّاقِ
</Aside>

<Aside type="reference" title="29">
و ساق پاها (از سختی جان‌دادن) به هم بپیچد!
</Aside>

---

<Aside type="quran" title="30">
إِلَىٰ رَبِّكَ يَوْمَئِذٍ ٱلْمَسَاقُ
</Aside>

<Aside type="reference" title="30">
(آری) در آن روز مسیر همه بسوی (دادگاه) پروردگارت خواهد بود!
</Aside>

---

<Aside type="quran" title="31">
فَلَا صَدَّقَ وَلَا صَلَّىٰ
</Aside>

<Aside type="reference" title="31">
(در آن روز گفته می‌شود:) او هرگز ایمان نیاورد و نماز نخواند،
</Aside>

---

<Aside type="quran" title="32">
وَلَٰكِن كَذَّبَ وَتَوَلَّىٰ
</Aside>

<Aside type="reference" title="32">
بلکه تکذیب کرد و روی‌گردان شد،
</Aside>

---

<Aside type="quran" title="33">
ثُمَّ ذَهَبَ إِلَىٰٓ أَهْلِهِۦ يَتَمَطَّىٰٓ
</Aside>

<Aside type="reference" title="33">
سپس بسوی خانواده خود بازگشت در حالی که متکبّرانه قدم برمی‌داشت!
</Aside>

---

<Aside type="quran" title="34">
أَوْلَىٰ لَكَ فَأَوْلَىٰ
</Aside>

<Aside type="reference" title="34">
(با این اعمال) عذاب الهی برای تو شایسته‌تر است، شایسته‌تر!
</Aside>

---

<Aside type="quran" title="35">
ثُمَّ أَوْلَىٰ لَكَ فَأَوْلَىٰٓ
</Aside>

<Aside type="reference" title="35">
سپس عذاب الهی برای تو شایسته‌تر است، شایسته‌تر!
</Aside>

---

<Aside type="quran" title="36">
أَيَحْسَبُ ٱلْإِنسَٰنُ أَن يُتْرَكَ سُدًى
</Aside>

<Aside type="reference" title="36">
آیا انسان گمان می‌کند بی‌هدف رها می‌شود؟!
</Aside>

---

<Aside type="quran" title="37">
أَلَمْ يَكُ نُطْفَةًۭ مِّن مَّنِىٍّۢ يُمْنَىٰ
</Aside>

<Aside type="reference" title="37">
آیا او نطفه‌ای از منی که در رحم ریخته می‌شود نبود؟!
</Aside>

---

<Aside type="quran" title="38">
ثُمَّ كَانَ عَلَقَةًۭ فَخَلَقَ فَسَوَّىٰ
</Aside>

<Aside type="reference" title="38">
سپس بصورت خون بسته در آمد، و خداوند او را آفرید و موزون ساخت،
</Aside>

---

<Aside type="quran" title="39">
فَجَعَلَ مِنْهُ ٱلزَّوْجَيْنِ ٱلذَّكَرَ وَٱلْأُنثَىٰٓ
</Aside>

<Aside type="reference" title="39">
و از او دو زوج مرد و زن آفرید!
</Aside>

---

<Aside type="quran" title="40">
أَلَيْسَ ذَٰلِكَ بِقَٰدِرٍ عَلَىٰٓ أَن يُحْۦِىَ ٱلْمَوْتَىٰ
</Aside>

<Aside type="reference" title="40">
آیا چنین کسی قادر نیست که مردگان را زنده کند؟!
</Aside>
