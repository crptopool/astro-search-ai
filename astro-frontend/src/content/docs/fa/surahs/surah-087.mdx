<Aside type="quran" title="1">
سَبِّحِ ٱسْمَ رَبِّكَ ٱلْأَعْلَى
</Aside>

<Aside type="reference" title="1">
منزّه شمار نام پروردگار بلندمرتبه‌ات را!
</Aside>

---

<Aside type="quran" title="2">
ٱلَّذِى خَلَقَ فَسَوَّىٰ
</Aside>

<Aside type="reference" title="2">
همان خداوندی که آفرید و منظم کرد،
</Aside>

---

<Aside type="quran" title="3">
وَٱلَّذِى قَدَّرَ فَهَدَىٰ
</Aside>

<Aside type="reference" title="3">
و همان که اندازه‌گیری کرد و هدایت نمود،
</Aside>

---

<Aside type="quran" title="4">
وَٱلَّذِىٓ أَخْرَجَ ٱلْمَرْعَىٰ
</Aside>

<Aside type="reference" title="4">
و آن کس را که چراگاه را به وجود آورد،
</Aside>

---

<Aside type="quran" title="5">
فَجَعَلَهُۥ غُثَآءً أَحْوَىٰ
</Aside>

<Aside type="reference" title="5">
سپس آن را خشک و تیره قرار داد!
</Aside>

---

<Aside type="quran" title="6">
سَنُقْرِئُكَ فَلَا تَنسَىٰٓ
</Aside>

<Aside type="reference" title="6">
ما بزودی (قرآن را) بر تو می‌خوانیم و هرگز فراموش نخواهی کرد،
</Aside>

---

<Aside type="quran" title="7">
إِلَّا مَا شَآءَ ٱللَّهُ ۚ إِنَّهُۥ يَعْلَمُ ٱلْجَهْرَ وَمَا يَخْفَىٰ
</Aside>

<Aside type="reference" title="7">
مگر آنچه را خدا بخواهد، که او آشکار و نهان را می‌داند!
</Aside>

---

<Aside type="quran" title="8">
وَنُيَسِّرُكَ لِلْيُسْرَىٰ
</Aside>

<Aside type="reference" title="8">
و ما تو را برای انجام هر کار خیر آماده می‌کنیم!
</Aside>

---

<Aside type="quran" title="9">
فَذَكِّرْ إِن نَّفَعَتِ ٱلذِّكْرَىٰ
</Aside>

<Aside type="reference" title="9">
پس تذکّر ده اگر تذکّر مفید باشد!
</Aside>

---

<Aside type="quran" title="10">
سَيَذَّكَّرُ مَن يَخْشَىٰ
</Aside>

<Aside type="reference" title="10">
و بزودی کسی که از خدا می‌ترسد متذکّر می‌شود.
</Aside>

---

<Aside type="quran" title="11">
وَيَتَجَنَّبُهَا ٱلْأَشْقَى
</Aside>

<Aside type="reference" title="11">
امّا بدبخت‌ترین افراد از آن دوری می‌گزیند،
</Aside>

---

<Aside type="quran" title="12">
ٱلَّذِى يَصْلَى ٱلنَّارَ ٱلْكُبْرَىٰ
</Aside>

<Aside type="reference" title="12">
همان کسی که در آتش بزرگ وارد می‌شود،
</Aside>

---

<Aside type="quran" title="13">
ثُمَّ لَا يَمُوتُ فِيهَا وَلَا يَحْيَىٰ
</Aside>

<Aside type="reference" title="13">
سپس در آن آتش نه می‌میرد و نه زنده می‌شود!
</Aside>

---

<Aside type="quran" title="14">
قَدْ أَفْلَحَ مَن تَزَكَّىٰ
</Aside>

<Aside type="reference" title="14">
به یقین کسی که پاکی جست (و خود را تزکیه کرد)، رستگار شد.
</Aside>

---

<Aside type="quran" title="15">
وَذَكَرَ ٱسْمَ رَبِّهِۦ فَصَلَّىٰ
</Aside>

<Aside type="reference" title="15">
و (آن که) نام پروردگارش را یاد کرد سپس نماز خواند!
</Aside>

---

<Aside type="quran" title="16">
بَلْ تُؤْثِرُونَ ٱلْحَيَوٰةَ ٱلدُّنْيَا
</Aside>

<Aside type="reference" title="16">
ولی شما زندگی دنیا را مقدم می‌دارید،
</Aside>

---

<Aside type="quran" title="17">
وَٱلْءَاخِرَةُ خَيْرٌۭ وَأَبْقَىٰٓ
</Aside>

<Aside type="reference" title="17">
در حالی که آخرت بهتر و پایدارتر است!
</Aside>

---

<Aside type="quran" title="18">
إِنَّ هَٰذَا لَفِى ٱلصُّحُفِ ٱلْأُولَىٰ
</Aside>

<Aside type="reference" title="18">
این دستورها در کتب آسمانی پیشین (نیز) آمده است،
</Aside>

---

<Aside type="quran" title="19">
صُحُفِ إِبْرَٰهِيمَ وَمُوسَىٰ
</Aside>

<Aside type="reference" title="19">
در کتب ابراهیم و موسی.
</Aside>
