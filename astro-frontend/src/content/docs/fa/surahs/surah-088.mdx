<Aside type="quran" title="1">
هَلْ أَتَىٰكَ حَدِيثُ ٱلْغَٰشِيَةِ
</Aside>

<Aside type="reference" title="1">
آیا داستان غاشیه [= روز قیامت که حوادث وحشتناکش همه را می‌پوشاند] به تو رسیده است؟!
</Aside>

---

<Aside type="quran" title="2">
وُجُوهٌۭ يَوْمَئِذٍ خَٰشِعَةٌ
</Aside>

<Aside type="reference" title="2">
چهره‌هایی در آن روز خاشع و ذلّت‌بارند،
</Aside>

---

<Aside type="quran" title="3">
عَامِلَةٌۭ نَّاصِبَةٌۭ
</Aside>

<Aside type="reference" title="3">
آنها که پیوسته عمل کرده و خسته شده‌اند (و نتیجه‌ای عایدشان نشده است)،
</Aside>

---

<Aside type="quran" title="4">
تَصْلَىٰ نَارًا حَامِيَةًۭ
</Aside>

<Aside type="reference" title="4">
و در آتش سوزان وارد می‌گردند؛
</Aside>

---

<Aside type="quran" title="5">
تُسْقَىٰ مِنْ عَيْنٍ ءَانِيَةٍۢ
</Aside>

<Aside type="reference" title="5">
از چشمه‌ای بسیار داغ به آنان می‌نوشانند؛
</Aside>

---

<Aside type="quran" title="6">
لَّيْسَ لَهُمْ طَعَامٌ إِلَّا مِن ضَرِيعٍۢ
</Aside>

<Aside type="reference" title="6">
غذائی جز از ضَریع [= خار خشک تلخ و بدبو] ندارند؛
</Aside>

---

<Aside type="quran" title="7">
لَّا يُسْمِنُ وَلَا يُغْنِى مِن جُوعٍۢ
</Aside>

<Aside type="reference" title="7">
غذایی که نه آنها را فربه می‌کند و نه از گرسنگی می‌رهاند!
</Aside>

---

<Aside type="quran" title="8">
وُجُوهٌۭ يَوْمَئِذٍۢ نَّاعِمَةٌۭ
</Aside>

<Aside type="reference" title="8">
چهره‌هایی در آن روز شاداب و باطراوتند،
</Aside>

---

<Aside type="quran" title="9">
لِّسَعْيِهَا رَاضِيَةٌۭ
</Aside>

<Aside type="reference" title="9">
و از سعی و تلاش خود خشنودند،
</Aside>

---

<Aside type="quran" title="10">
فِى جَنَّةٍ عَالِيَةٍۢ
</Aside>

<Aside type="reference" title="10">
در بهشتی عالی جای دارند،
</Aside>

---

<Aside type="quran" title="11">
لَّا تَسْمَعُ فِيهَا لَٰغِيَةًۭ
</Aside>

<Aside type="reference" title="11">
که در آن هیچ سخن لغو و بیهوده‌ای نمی‌شنوند!
</Aside>

---

<Aside type="quran" title="12">
فِيهَا عَيْنٌۭ جَارِيَةٌۭ
</Aside>

<Aside type="reference" title="12">
در آن چشمه‌ای جاری است،
</Aside>

---

<Aside type="quran" title="13">
فِيهَا سُرُرٌۭ مَّرْفُوعَةٌۭ
</Aside>

<Aside type="reference" title="13">
در آن تختهای زیبای بلندی است،
</Aside>

---

<Aside type="quran" title="14">
وَأَكْوَابٌۭ مَّوْضُوعَةٌۭ
</Aside>

<Aside type="reference" title="14">
و قدحهایی (که در کنار این چشمه) نهاده،
</Aside>

---

<Aside type="quran" title="15">
وَنَمَارِقُ مَصْفُوفَةٌۭ
</Aside>

<Aside type="reference" title="15">
و بالشها و پشتیهای صف‌داده شده،
</Aside>

---

<Aside type="quran" title="16">
وَزَرَابِىُّ مَبْثُوثَةٌ
</Aside>

<Aside type="reference" title="16">
و فرشهای فاخر گسترده!
</Aside>

---

<Aside type="quran" title="17">
أَفَلَا يَنظُرُونَ إِلَى ٱلْإِبِلِ كَيْفَ خُلِقَتْ
</Aside>

<Aside type="reference" title="17">
آیا آنان به شتر نمی‌نگرند که چگونه آفریده شده است؟!
</Aside>

---

<Aside type="quran" title="18">
وَإِلَى ٱلسَّمَآءِ كَيْفَ رُفِعَتْ
</Aside>

<Aside type="reference" title="18">
و به آسمان نگاه نمی‌کنند که چگونه برافراشته شده؟!
</Aside>

---

<Aside type="quran" title="19">
وَإِلَى ٱلْجِبَالِ كَيْفَ نُصِبَتْ
</Aside>

<Aside type="reference" title="19">
و به کوه‌ها که چگونه در جای خود نصب گردیده!
</Aside>

---

<Aside type="quran" title="20">
وَإِلَى ٱلْأَرْضِ كَيْفَ سُطِحَتْ
</Aside>

<Aside type="reference" title="20">
و به زمین که چگونه گسترده و هموار گشته است؟!
</Aside>

---

<Aside type="quran" title="21">
فَذَكِّرْ إِنَّمَآ أَنتَ مُذَكِّرٌۭ
</Aside>

<Aside type="reference" title="21">
پس تذکّر ده که تو فقط تذکّر دهنده‌ای!
</Aside>

---

<Aside type="quran" title="22">
لَّسْتَ عَلَيْهِم بِمُصَيْطِرٍ
</Aside>

<Aside type="reference" title="22">
تو سلطه‌گر بر آنان نیستی که (بر ایمان) مجبورشان کنی،
</Aside>

---

<Aside type="quran" title="23">
إِلَّا مَن تَوَلَّىٰ وَكَفَرَ
</Aside>

<Aside type="reference" title="23">
مگر کسی که پشت کند و کافر شود،
</Aside>

---

<Aside type="quran" title="24">
فَيُعَذِّبُهُ ٱللَّهُ ٱلْعَذَابَ ٱلْأَكْبَرَ
</Aside>

<Aside type="reference" title="24">
که خداوند او را به عذاب بزرگ مجازات می‌کند!
</Aside>

---

<Aside type="quran" title="25">
إِنَّ إِلَيْنَآ إِيَابَهُمْ
</Aside>

<Aside type="reference" title="25">
به یقین بازگشت (همه) آنان به سوی ماست،
</Aside>

---

<Aside type="quran" title="26">
ثُمَّ إِنَّ عَلَيْنَا حِسَابَهُم
</Aside>

<Aside type="reference" title="26">
و مسلّماً حسابشان (نیز) با ماست!
</Aside>
