<Aside type="quran" title="1">
قٓ ۚ وَٱلْقُرْءَانِ ٱلْمَجِيدِ
</Aside>

<Aside type="reference" title="1">
ق، سوگند به قرآن مجید (که قیامت و رستاخیز حقّ است)!
</Aside>

---

<Aside type="quran" title="2">
بَلْ عَجِبُوٓا۟ أَن جَآءَهُم مُّنذِرٌۭ مِّنْهُمْ فَقَالَ ٱلْكَٰفِرُونَ هَٰذَا شَىْءٌ عَجِيبٌ
</Aside>

<Aside type="reference" title="2">
آنها تعجّب کردند که پیامبری انذارگر از میان خودشان آمده؛ و کافران گفتند: «این چیز عجیبی است!
</Aside>

---

<Aside type="quran" title="3">
أَءِذَا مِتْنَا وَكُنَّا تُرَابًۭا ۖ ذَٰلِكَ رَجْعٌۢ بَعِيدٌۭ
</Aside>

<Aside type="reference" title="3">
آیا هنگامی که مُردیم و خاک شدیم (دوباره به زندگی بازمی‌گردیم)؟! این بازگشتی بعید است!»
</Aside>

---

<Aside type="quran" title="4">
قَدْ عَلِمْنَا مَا تَنقُصُ ٱلْأَرْضُ مِنْهُمْ ۖ وَعِندَنَا كِتَٰبٌ حَفِيظٌۢ
</Aside>

<Aside type="reference" title="4">
ولی ما می‌دانیم آنچه را زمین از بدن آنها می‌کاهد؛ و نزد ما کتابی است که همه چیز در آن محفوظ است!
</Aside>

---

<Aside type="quran" title="5">
بَلْ كَذَّبُوا۟ بِٱلْحَقِّ لَمَّا جَآءَهُمْ فَهُمْ فِىٓ أَمْرٍۢ مَّرِيجٍ
</Aside>

<Aside type="reference" title="5">
آنها حقّ را هنگامی که به سراغشان آمد تکذیب کردند؛ از این رو پیوسته در کار پراکنده خود متحیّرند!
</Aside>

---

<Aside type="quran" title="6">
أَفَلَمْ يَنظُرُوٓا۟ إِلَى ٱلسَّمَآءِ فَوْقَهُمْ كَيْفَ بَنَيْنَٰهَا وَزَيَّنَّٰهَا وَمَا لَهَا مِن فُرُوجٍۢ
</Aside>

<Aside type="reference" title="6">
آیا آنان به آسمان بالای سرشان نگاه نکردند که چگونه ما آن را بنا کرده‌ایم، و چگونه آن را (بوسیله ستارگان) زینت بخشیده‌ایم و هیچ شکاف و شکستی در آن نیست؟!
</Aside>

<Aside type="quran" title="7">
وَٱلْأَرْضَ مَدَدْنَٰهَا وَأَلْقَيْنَا فِيهَا رَوَٰسِىَ وَأَنۢبَتْنَا فِيهَا مِن كُلِّ زَوْجٍۭ بَهِيجٍۢ
</Aside>

<Aside type="reference" title="7">
و زمین را گسترش دادیم و در آن کوه‌هائی عظیم و استوار افکندیم و از هر نوع گیاه بهجت‌انگیز در آن رویاندیم،
</Aside>

---

<Aside type="quran" title="8">

</Aside>

<Aside type="reference" title="8">

</Aside>

<Aside type="quran" title="9">

</Aside>

<Aside type="reference" title="9">

</Aside>

<Aside type="quran" title="10">

<Aside type="reference" title="10">

<Aside type="quran" title="11">

</Aside>

<Aside type="reference" title="11">

<Aside type="quran" title="12">

</Aside>

<Aside type="reference" title="12">

</Aside>

<Aside type="quran" title="13">

<Aside type="reference" title="13">

</Aside>

---

<Aside type="quran" title="14">

</Aside>

<Aside type="reference" title="14">

</Aside>

<Aside type="quran" title="15">

</Aside>

<Aside type="reference" title="15">

</Aside>

<Aside type="quran" title="16">
وَلَقَدْ خَلَقْنَا ٱلْإِنسَٰنَ وَنَعْلَمُ مَا تُوَسْوِسُ بِهِۦ نَفْسُهُۥ ۖ وَنَحْنُ أَقْرَبُ إِلَيْهِ مِنْ حَبْلِ ٱلْوَرِيدِ
</Aside>

<Aside type="reference" title="16">
ما انسان را آفریدیم و وسوسه‌های نفس او را می‌دانیم، و ما به او از رگ قلبش نزدیکتریم!
</Aside>

<Aside type="quran" title="17">

</Aside>

<Aside type="reference" title="17">
(به خاطر بیاورید) هنگامی را که دو فرشته راست و چپ که ملازم انسانند اعمال او را دریافت می‌دارند؛
</Aside>

<Aside type="quran" title="18">

</Aside>

<Aside type="reference" title="18">
انسان هیچ سخنی را بر زبان نمی‌آورد مگر اینکه همان دم، فرشته‌ای مراقب و آماده برای انجام مأموریت (و ضبط آن) است!
</Aside>

<Aside type="quran" title="19">

</Aside>

<Aside type="reference" title="19">

</Aside>

<Aside type="quran" title="20">

<Aside type="quran" title="44">
يَوْمَ تَشَقَّقُ ٱلْأَرْضُ عَنْهُمْ سِرَاعًۭا ۚ ذَٰلِكَ حَشْرٌ عَلَيْنَا يَسِيرٌۭ
</Aside>

<Aside type="reference" title="44">

</Aside>

<Aside type="reference" title="45">

</Aside>
