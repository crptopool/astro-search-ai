<Aside type="quran" title="1">
وَيْلٌۭ لِّلْمُطَفِّفِينَ
</Aside>

<Aside type="reference" title="1">
وای بر کم‌فروشان!
</Aside>

---

<Aside type="quran" title="2">
ٱلَّذِينَ إِذَا ٱكْتَالُوا۟ عَلَى ٱلنَّاسِ يَسْتَوْفُونَ
</Aside>

<Aside type="reference" title="2">
آنان که وقتی برای خود پیمانه می‌کنند، حق خود را بطور کامل می‌گیرند؛
</Aside>

---

<Aside type="quran" title="3">
وَإِذَا كَالُوهُمْ أَو وَّزَنُوهُمْ يُخْسِرُونَ
</Aside>

<Aside type="reference" title="3">
امّا هنگامی که می‌خواهند برای دیگران پیمانه یا وزن کنند، کم می‌گذارند!
</Aside>

---

<Aside type="quran" title="4">
أَلَا يَظُنُّ أُو۟لَٰٓئِكَ أَنَّهُم مَّبْعُوثُونَ
</Aside>

<Aside type="reference" title="4">
آیا آنها گمان نمی‌کنند که برانگیخته می‌شوند،
</Aside>

---

<Aside type="quran" title="5">
لِيَوْمٍ عَظِيمٍۢ
</Aside>

<Aside type="reference" title="5">
در روزی بزرگ؛
</Aside>

---

<Aside type="quran" title="6">
يَوْمَ يَقُومُ ٱلنَّاسُ لِرَبِّ ٱلْعَٰلَمِينَ
</Aside>

<Aside type="reference" title="6">
روزی که مردم در پیشگاه پروردگار جهانیان می‌ایستند.
</Aside>

---

<Aside type="quran" title="7">
كَلَّآ إِنَّ كِتَٰبَ ٱلْفُجَّارِ لَفِى سِجِّينٍۢ
</Aside>

<Aside type="reference" title="7">
چنین نیست که آنها (درباره قیامت) می‌پندارند، به یقین نامه اعمال بدکاران در «سجّین» است!
</Aside>

---

<Aside type="quran" title="8">
وَمَآ أَدْرَىٰكَ مَا سِجِّينٌۭ
</Aside>

<Aside type="reference" title="8">
تو چه می‌دانی «سجّین» چیست؟
</Aside>

---

<Aside type="quran" title="9">
كِتَٰبٌۭ مَّرْقُومٌۭ
</Aside>

<Aside type="reference" title="9">
نامه‌ای است رقم زده شده و سرنوشتی است حتمی!
</Aside>

---

<Aside type="quran" title="10">
وَيْلٌۭ يَوْمَئِذٍۢ لِّلْمُكَذِّبِينَ
</Aside>

<Aside type="reference" title="10">
وای در آن روز بر تکذیب‌کنندگان!
</Aside>

---

<Aside type="quran" title="11">
ٱلَّذِينَ يُكَذِّبُونَ بِيَوْمِ ٱلدِّينِ
</Aside>

<Aside type="reference" title="11">
همانها که روز جزا را انکار می‌کنند.
</Aside>

---

<Aside type="quran" title="12">
وَمَا يُكَذِّبُ بِهِۦٓ إِلَّا كُلُّ مُعْتَدٍ أَثِيمٍ
</Aside>

<Aside type="reference" title="12">
تنها کسی آن را انکار می‌کند که متجاوز و گنهکار است!
</Aside>

---

<Aside type="quran" title="13">
إِذَا تُتْلَىٰ عَلَيْهِ ءَايَٰتُنَا قَالَ أَسَٰطِيرُ ٱلْأَوَّلِينَ
</Aside>

<Aside type="reference" title="13">
(همان کسی که) وقتی آیات ما بر او خوانده می‌شود می‌گوید: «این افسانه‌های پیشینیان است!»
</Aside>

---

<Aside type="quran" title="14">
كَلَّا ۖ بَلْ ۜ رَانَ عَلَىٰ قُلُوبِهِم مَّا كَانُوا۟ يَكْسِبُونَ
</Aside>

<Aside type="reference" title="14">
چنین نیست که آنها می‌پندارند، بلکه اعمالشان چون زنگاری بر دلهایشان نشسته است!
</Aside>

---

<Aside type="quran" title="15">
كَلَّآ إِنَّهُمْ عَن رَّبِّهِمْ يَوْمَئِذٍۢ لَّمَحْجُوبُونَ
</Aside>

<Aside type="reference" title="15">
چنین نیست که می‌پندارند، بلکه آنها در آن روز از پروردگارشان محجوبند!
</Aside>

---

<Aside type="quran" title="16">
ثُمَّ إِنَّهُمْ لَصَالُوا۟ ٱلْجَحِيمِ
</Aside>

<Aside type="reference" title="16">
سپس آنها به یقین وارد دوزخ می‌شوند!
</Aside>

---

<Aside type="quran" title="17">
ثُمَّ يُقَالُ هَٰذَا ٱلَّذِى كُنتُم بِهِۦ تُكَذِّبُونَ
</Aside>

<Aside type="reference" title="17">
بعد به آنها گفته می‌شود: «این همان چیزی است که آن را انکار می‌کردید!»
</Aside>

---

<Aside type="quran" title="18">
كَلَّآ إِنَّ كِتَٰبَ ٱلْأَبْرَارِ لَفِى عِلِّيِّينَ
</Aside>

<Aside type="reference" title="18">
چنان نیست که آنها (درباره معاد) می‌پندارند، بلکه نامه اعمال نیکان در «علیّین» است!
</Aside>

---

<Aside type="quran" title="19">
وَمَآ أَدْرَىٰكَ مَا عِلِّيُّونَ
</Aside>

<Aside type="reference" title="19">
و تو چه می‌دانی «علیّین» چیست!
</Aside>

---

<Aside type="quran" title="20">
كِتَٰبٌۭ مَّرْقُومٌۭ
</Aside>

<Aside type="reference" title="20">
نامه‌ای است رقم‌خورده و سرنوشتی است قطعی،
</Aside>

---

<Aside type="quran" title="21">
يَشْهَدُهُ ٱلْمُقَرَّبُونَ
</Aside>

<Aside type="reference" title="21">
که مقربان شاهد آنند!
</Aside>

---

<Aside type="quran" title="22">
إِنَّ ٱلْأَبْرَارَ لَفِى نَعِيمٍ
</Aside>

<Aside type="reference" title="22">
مسلّماً نیکان در انواع نعمت‌اند:
</Aside>

---

<Aside type="quran" title="23">
عَلَى ٱلْأَرَآئِكِ يَنظُرُونَ
</Aside>

<Aside type="reference" title="23">
بر تختهای زیبای بهشتی تکیه کرده و (به زیباییهای بهشت) می‌نگرند!
</Aside>

---

<Aside type="quran" title="24">
تَعْرِفُ فِى وُجُوهِهِمْ نَضْرَةَ ٱلنَّعِيمِ
</Aside>

<Aside type="reference" title="24">
در چهره‌هایشان طراوت و نشاط نعمت را می‌بینی و می‌شناسی!
</Aside>

---

<Aside type="quran" title="25">
يُسْقَوْنَ مِن رَّحِيقٍۢ مَّخْتُومٍ
</Aside>

<Aside type="reference" title="25">
آنها از شراب (طهور) زلال دست‌نخورده و سربسته‌ای سیراب می‌شوند!
</Aside>

---

<Aside type="quran" title="26">
خِتَٰمُهُۥ مِسْكٌۭ ۚ وَفِى ذَٰلِكَ فَلْيَتَنَافَسِ ٱلْمُتَنَٰفِسُونَ
</Aside>

<Aside type="reference" title="26">
مهری که بر آن نهاده شده از مشک است؛ و در این نعمتهای بهشتی راغبان باید بر یکدیگر پیشی گیرند!
</Aside>

---

<Aside type="quran" title="27">
وَمِزَاجُهُۥ مِن تَسْنِيمٍ
</Aside>

<Aside type="reference" title="27">
این شراب (طهور) آمیخته با «تسنیم» است،
</Aside>

---

<Aside type="quran" title="28">
عَيْنًۭا يَشْرَبُ بِهَا ٱلْمُقَرَّبُونَ
</Aside>

<Aside type="reference" title="28">
همان چشمه‌ای که مقرّبان از آن می‌نوشند.
</Aside>

---

<Aside type="quran" title="29">
إِنَّ ٱلَّذِينَ أَجْرَمُوا۟ كَانُوا۟ مِنَ ٱلَّذِينَ ءَامَنُوا۟ يَضْحَكُونَ
</Aside>

<Aside type="reference" title="29">
بدکاران (در دنیا) پیوسته به مؤمنان می‌خندیدند،
</Aside>

---

<Aside type="quran" title="30">
وَإِذَا مَرُّوا۟ بِهِمْ يَتَغَامَزُونَ
</Aside>

<Aside type="reference" title="30">
و هنگامی که از کنارشان می‌گذشتند آنان را با اشاره تمسخر می‌کردند،
</Aside>

---

<Aside type="quran" title="31">
وَإِذَا ٱنقَلَبُوٓا۟ إِلَىٰٓ أَهْلِهِمُ ٱنقَلَبُوا۟ فَكِهِينَ
</Aside>

<Aside type="reference" title="31">
و چون به سوی خانواده خود بازمی‌گشتند مسرور و خندان بودند،
</Aside>

---

<Aside type="quran" title="32">
وَإِذَا رَأَوْهُمْ قَالُوٓا۟ إِنَّ هَٰٓؤُلَآءِ لَضَآلُّونَ
</Aside>

<Aside type="reference" title="32">
و هنگامی که آنها را می‌دیدند می‌گفتند: «اینها گمراهانند!»
</Aside>

---

<Aside type="quran" title="33">
وَمَآ أُرْسِلُوا۟ عَلَيْهِمْ حَٰفِظِينَ
</Aside>

<Aside type="reference" title="33">
در حالی که هرگز مأمور مراقبت و متکفّل آنان [= مؤمنان‌] نبودند!
</Aside>

---

<Aside type="quran" title="34">
فَٱلْيَوْمَ ٱلَّذِينَ ءَامَنُوا۟ مِنَ ٱلْكُفَّارِ يَضْحَكُونَ
</Aside>

<Aside type="reference" title="34">
ولی امروز مؤمنان به کفار می‌خندند،
</Aside>

---

<Aside type="quran" title="35">
عَلَى ٱلْأَرَآئِكِ يَنظُرُونَ
</Aside>

<Aside type="reference" title="35">
در حالی که بر تختهای آراسته بهشتی نشسته و (به سرنوشت شوم آنها) می‌نگرند!
</Aside>

---

<Aside type="quran" title="36">
هَلْ ثُوِّبَ ٱلْكُفَّارُ مَا كَانُوا۟ يَفْعَلُونَ
</Aside>

<Aside type="reference" title="36">
آیا (با این حال) کافران پاداش اعمال خود را گرفتند؟!
</Aside>
