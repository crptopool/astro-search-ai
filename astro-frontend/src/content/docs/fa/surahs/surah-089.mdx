<Aside type="quran" title="1">
وَٱلْفَجْرِ
</Aside>

<Aside type="reference" title="1">
به سپیده دم سوگند،
</Aside>

---

<Aside type="quran" title="2">
وَلَيَالٍ عَشْرٍۢ
</Aside>

<Aside type="reference" title="2">
و به شبهای دهگانه،
</Aside>

---

<Aside type="quran" title="3">
وَٱلشَّفْعِ وَٱلْوَتْرِ
</Aside>

<Aside type="reference" title="3">
و به زوج و فرد،
</Aside>

---

<Aside type="quran" title="4">
وَٱلَّيْلِ إِذَا يَسْرِ
</Aside>

<Aside type="reference" title="4">
و به شب، هنگامی که (به سوی روشنایی روز) حرکت می‌کند سوگند (که پروردگارت در کمین ظالمان است)!
</Aside>

---

<Aside type="quran" title="5">
هَلْ فِى ذَٰلِكَ قَسَمٌۭ لِّذِى حِجْرٍ
</Aside>

<Aside type="reference" title="5">
آیا در آنچه گفته شد، سوگند مهمّی برای صاحبان خرد نیست؟!
</Aside>

---

<Aside type="quran" title="6">
أَلَمْ تَرَ كَيْفَ فَعَلَ رَبُّكَ بِعَادٍ
</Aside>

<Aside type="reference" title="6">
آیا ندیدی پروردگارت با قوم «عاد» چه کرد؟!
</Aside>

---

<Aside type="quran" title="7">
إِرَمَ ذَاتِ ٱلْعِمَادِ
</Aside>

<Aside type="reference" title="7">
و با آن شهر «اِرَم» باعظمت،
</Aside>

---

<Aside type="quran" title="8">
ٱلَّتِى لَمْ يُخْلَقْ مِثْلُهَا فِى ٱلْبِلَٰدِ
</Aside>

<Aside type="reference" title="8">
همان شهری که مانندش در شهرها آفریده نشده بود!
</Aside>

---

<Aside type="quran" title="9">
وَثَمُودَ ٱلَّذِينَ جَابُوا۟ ٱلصَّخْرَ بِٱلْوَادِ
</Aside>

<Aside type="reference" title="9">
و قوم «ثمود» که صخره‌های عظیم را از (کنار) درّه می‌بریدند (و از آن خانه و کاخ می‌ساختند)!
</Aside>

---

<Aside type="quran" title="10">
وَفِرْعَوْنَ ذِى ٱلْأَوْتَادِ
</Aside>

<Aside type="reference" title="10">
و فرعونی که قدرتمند و شکنجه‌گر بود،
</Aside>

---

<Aside type="quran" title="11">
ٱلَّذِينَ طَغَوْا۟ فِى ٱلْبِلَٰدِ
</Aside>

<Aside type="reference" title="11">
همان اقوامی که در شهرها طغیان کردند،
</Aside>

---

<Aside type="quran" title="12">
فَأَكْثَرُوا۟ فِيهَا ٱلْفَسَادَ
</Aside>

<Aside type="reference" title="12">
و فساد فراوان در آنها به بار آوردند؛
</Aside>

---

<Aside type="quran" title="13">
فَصَبَّ عَلَيْهِمْ رَبُّكَ سَوْطَ عَذَابٍ
</Aside>

<Aside type="reference" title="13">
به همین سبب خداوند تازیانه عذاب را بر آنان فرو ریخت!
</Aside>

---

<Aside type="quran" title="14">
إِنَّ رَبَّكَ لَبِٱلْمِرْصَادِ
</Aside>

<Aside type="reference" title="14">
به یقین پروردگار تو در کمینگاه (ستمگران) است!
</Aside>

---

<Aside type="quran" title="15">
فَأَمَّا ٱلْإِنسَٰنُ إِذَا مَا ٱبْتَلَىٰهُ رَبُّهُۥ فَأَكْرَمَهُۥ وَنَعَّمَهُۥ فَيَقُولُ رَبِّىٓ أَكْرَمَنِ
</Aside>

<Aside type="reference" title="15">
امّا انسان هنگامی که پروردگارش او را برای آزمایش، اکرام می‌کند و نعمت می‌بخشد (مغرور می‌شود و) می‌گوید: «پروردگارم مرا گرامی داشته است!»
</Aside>

---

<Aside type="quran" title="16">
وَأَمَّآ إِذَا مَا ٱبْتَلَىٰهُ فَقَدَرَ عَلَيْهِ رِزْقَهُۥ فَيَقُولُ رَبِّىٓ أَهَٰنَنِ
</Aside>

<Aside type="reference" title="16">
و امّا هنگامی که برای امتحان، روزیش را بر او تنگ می‌گیرد (مأیوس می‌شود و) می‌گوید: «پروردگارم مرا خوار کرده است!»
</Aside>

---

<Aside type="quran" title="17">
كَلَّا ۖ بَل لَّا تُكْرِمُونَ ٱلْيَتِيمَ
</Aside>

<Aside type="reference" title="17">
چنان نیست که شما می‌پندارید؛ شما یتیمان را گرامی نمی‌دارید،
</Aside>

---

<Aside type="quran" title="18">
وَلَا تَحَٰٓضُّونَ عَلَىٰ طَعَامِ ٱلْمِسْكِينِ
</Aside>

<Aside type="reference" title="18">
و یکدیگر را بر اطعام مستمندان تشویق نمی‌کنید،
</Aside>

---

<Aside type="quran" title="19">
وَتَأْكُلُونَ ٱلتُّرَاثَ أَكْلًۭا لَّمًّۭا
</Aside>

<Aside type="reference" title="19">
و میراث را (از راه مشروع و نامشروع) جمع کرده می‌خورید،
</Aside>

---

<Aside type="quran" title="20">
وَتُحِبُّونَ ٱلْمَالَ حُبًّۭا جَمًّۭا
</Aside>

<Aside type="reference" title="20">
و مال و ثروت را بسیار دوست دارید (و بخاطر آن گناهان زیادی مرتکب می‌شوید)!
</Aside>

---

<Aside type="quran" title="21">
كَلَّآ إِذَا دُكَّتِ ٱلْأَرْضُ دَكًّۭا دَكًّۭا
</Aside>

<Aside type="reference" title="21">
چنان نیست که آنها می‌پندارند! در آن هنگام که زمین سخت در هم کوبیده شود،
</Aside>

---

<Aside type="quran" title="22">
وَجَآءَ رَبُّكَ وَٱلْمَلَكُ صَفًّۭا صَفًّۭا
</Aside>

<Aside type="reference" title="22">
و فرمان پروردگارت فرا رسد و فرشتگان صف در صف حاضر شوند،
</Aside>

---

<Aside type="quran" title="23">
وَجِا۟ىٓءَ يَوْمَئِذٍۭ بِجَهَنَّمَ ۚ يَوْمَئِذٍۢ يَتَذَكَّرُ ٱلْإِنسَٰنُ وَأَنَّىٰ لَهُ ٱلذِّكْرَىٰ
</Aside>

<Aside type="reference" title="23">
و در آن روز جهنم را حاضر می‌کنند؛ (آری) در آن روز انسان متذکّر می‌شود؛ امّا این تذکّر چه سودی برای او دارد؟!
</Aside>

---

<Aside type="quran" title="24">
يَقُولُ يَٰلَيْتَنِى قَدَّمْتُ لِحَيَاتِى
</Aside>

<Aside type="reference" title="24">
می‌گوید: «ای کاش برای (این) زندگیم چیزی از پیش فرستاده بودم!»
</Aside>

---

<Aside type="quran" title="25">
فَيَوْمَئِذٍۢ لَّا يُعَذِّبُ عَذَابَهُۥٓ أَحَدٌۭ
</Aside>

<Aside type="reference" title="25">
در آن روز هیچ کس همانند او [= خدا] عذاب نمی‌کند،
</Aside>

---

<Aside type="quran" title="26">
وَلَا يُوثِقُ وَثَاقَهُۥٓ أَحَدٌۭ
</Aside>

<Aside type="reference" title="26">
و هیچ کس همچون او کسی را به بند نمی‌کشد!
</Aside>

---

<Aside type="quran" title="27">
يَٰٓأَيَّتُهَا ٱلنَّفْسُ ٱلْمُطْمَئِنَّةُ
</Aside>

<Aside type="reference" title="27">
تو ای روح آرام‌یافته!
</Aside>

---

<Aside type="quran" title="28">
ٱرْجِعِىٓ إِلَىٰ رَبِّكِ رَاضِيَةًۭ مَّرْضِيَّةًۭ
</Aside>

<Aside type="reference" title="28">
به سوی پروردگارت بازگرد در حالی که هم تو از او خشنودی و هم او از تو خشنود است،
</Aside>

</Aside>
