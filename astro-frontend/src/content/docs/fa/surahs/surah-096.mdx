<Aside type="quran" title="1">
ٱقْرَأْ بِٱسْمِ رَبِّكَ ٱلَّذِى خَلَقَ
</Aside>

<Aside type="reference" title="1">
بخوان به نام پروردگارت که (جهان را) آفرید،
</Aside>

---

<Aside type="quran" title="2">
خَلَقَ ٱلْإِنسَٰنَ مِنْ عَلَقٍ
</Aside>

<Aside type="reference" title="2">
همان کس که انسان را از خون بسته‌ای خلق کرد!
</Aside>

---

<Aside type="quran" title="3">
ٱقْرَأْ وَرَبُّكَ ٱلْأَكْرَمُ
</Aside>

<Aside type="reference" title="3">
بخوان که پروردگارت (از همه) بزرگوارتر است،
</Aside>

---

<Aside type="quran" title="4">
ٱلَّذِى عَلَّمَ بِٱلْقَلَمِ
</Aside>

<Aside type="reference" title="4">
همان کسی که بوسیله قلم تعلیم نمود،
</Aside>

---

<Aside type="quran" title="5">
عَلَّمَ ٱلْإِنسَٰنَ مَا لَمْ يَعْلَمْ
</Aside>

<Aside type="reference" title="5">
و به انسان آنچه را نمی‌دانست یاد داد!
</Aside>

---

<Aside type="quran" title="6">
كَلَّآ إِنَّ ٱلْإِنسَٰنَ لَيَطْغَىٰٓ
</Aside>

<Aside type="reference" title="6">
چنین نیست (که شما می‌پندارید) به یقین انسان طغیان می‌کند،
</Aside>

---

<Aside type="quran" title="7">
أَن رَّءَاهُ ٱسْتَغْنَىٰٓ
</Aside>

<Aside type="reference" title="7">
از اینکه خود را بی‌نیاز ببیند!
</Aside>

---

<Aside type="quran" title="8">
إِنَّ إِلَىٰ رَبِّكَ ٱلرُّجْعَىٰٓ
</Aside>

<Aside type="reference" title="8">
و به یقین بازگشت (همه) به سوی پروردگار تو است!
</Aside>

---

<Aside type="quran" title="9">
أَرَءَيْتَ ٱلَّذِى يَنْهَىٰ
</Aside>

<Aside type="reference" title="9">
به من خبر ده آیا کسی که نهی می‌کند،
</Aside>

---

<Aside type="quran" title="10">
عَبْدًا إِذَا صَلَّىٰٓ
</Aside>

<Aside type="reference" title="10">
بنده‌ای را به هنگامی که نماز می‌خواند (آیا مستحق عذاب الهی نیست)؟!
</Aside>

---

<Aside type="quran" title="11">
أَرَءَيْتَ إِن كَانَ عَلَى ٱلْهُدَىٰٓ
</Aside>

<Aside type="reference" title="11">
به من خبر ده اگر این بنده به راه هدایت باشد،
</Aside>

---

<Aside type="quran" title="12">
أَوْ أَمَرَ بِٱلتَّقْوَىٰٓ
</Aside>

<Aside type="reference" title="12">
یا مردم را به تقوا فرمان دهد (آیا نهی کردن او سزاوار است)؟!
</Aside>

---

<Aside type="quran" title="13">
أَرَءَيْتَ إِن كَذَّبَ وَتَوَلَّىٰٓ
</Aside>

<Aside type="reference" title="13">
به من خبر ده اگر (این طغیانگر) حق را انکار کند و به آن پشت نماید (آیا مستحق مجازات الهی نیست)؟!
</Aside>

---

<Aside type="quran" title="14">
أَلَمْ يَعْلَم بِأَنَّ ٱللَّهَ يَرَىٰ
</Aside>

<Aside type="reference" title="14">
آیا او ندانست که خداوند (همه اعمالش را) می‌بیند؟!
</Aside>

---

<Aside type="quran" title="15">
كَلَّا لَئِن لَّمْ يَنتَهِ لَنَسْفَعًۢا بِٱلنَّاصِيَةِ
</Aside>

<Aside type="reference" title="15">
چنان نیست که او خیال می‌کند، اگر دست از کار خود برندارد، ناصیه‌اش [= موی پیش سرش‌] را گرفته (و به سوی عذاب می‌کشانیم)،
</Aside>

---

<Aside type="quran" title="16">
نَاصِيَةٍۢ كَٰذِبَةٍ خَاطِئَةٍۢ
</Aside>

<Aside type="reference" title="16">
همان ناصیه دروغگوی خطاکار را!
</Aside>

---

<Aside type="quran" title="17">
فَلْيَدْعُ نَادِيَهُۥ
</Aside>

<Aside type="reference" title="17">
سپس هر که را می‌خواهد صدا بزند (تا یاریش کند)!
</Aside>

---

<Aside type="quran" title="18">
سَنَدْعُ ٱلزَّبَانِيَةَ
</Aside>

<Aside type="reference" title="18">
ما هم بزودی مأموران دوزخ را صدا می‌زنیم (تا او را به دوزخ افکنند)!
</Aside>

---

<Aside type="quran" title="19">
كَلَّا لَا تُطِعْهُ وَٱسْجُدْ وَٱقْتَرِب ۩
</Aside>

<Aside type="reference" title="19">
چنان نیست (که آن طغیانگر می‌پندارد)؛ هرگز او را اطاعت مکن، و سجده نما و (به خدا) تقرّب جوی!
</Aside>
