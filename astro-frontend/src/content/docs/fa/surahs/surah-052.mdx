</Aside>

<Aside type="reference" title="14">

</Aside>

---

<Aside type="quran" title="15">

</Aside>

---

<Aside type="quran" title="16">
ٱصْلَوْهَا فَٱصْبِرُوٓا۟ أَوْ لَا تَصْبِرُوا۟ سَوَآءٌ عَلَيْكُمْ ۖ إِنَّمَا تُجْزَوْنَ مَا كُنتُمْ تَعْمَلُونَ
</Aside>

<Aside type="reference" title="16">
در آن وارد شوید و بسوزید؛ می‌خواهید صبر کنید یا نکنید، برای شما یکسان است؛ چرا که تنها به اعمالتان جزا داده می‌شوید!
</Aside>

---

<Aside type="quran" title="17">
إِنَّ ٱلْمُتَّقِينَ فِى جَنَّٰتٍۢ وَنَعِيمٍۢ
</Aside>

<Aside type="reference" title="17">
ولی پرهیزگاران در میان باغهای بهشت و نعمتهای فراوان جای دارند،
</Aside>

---

<Aside type="quran" title="18">
فَٰكِهِينَ بِمَآ ءَاتَىٰهُمْ رَبُّهُمْ وَوَقَىٰهُمْ رَبُّهُمْ عَذَابَ ٱلْجَحِيمِ
</Aside>

<Aside type="reference" title="18">
و از آنچه پروردگارشان به آنها داده و آنان را از عذاب دوزخ نگاه داشته است شاد و مسرورند!
</Aside>

---

<Aside type="quran" title="19">
كُلُوا۟ وَٱشْرَبُوا۟ هَنِيٓـًٔۢا بِمَا كُنتُمْ تَعْمَلُونَ
</Aside>

<Aside type="reference" title="19">
(به آنها گفته می‌شود:) بخورید و بیاشامید گوارا؛ اینها در برابر اعمالی است که انجام می‌دادید!
</Aside>

---

<Aside type="quran" title="20">
مُتَّكِـِٔينَ عَلَىٰ سُرُرٍۢ مَّصْفُوفَةٍۢ ۖ وَزَوَّجْنَٰهُم بِحُورٍ عِينٍۢ
</Aside>

<Aside type="reference" title="20">
این در حالی که بر تختهای صف‌کشیده در کنار هم تکیه می‌زنند، و «حور العین» را به همسری آنها درمی‌آوریم!
</Aside>

---

<Aside type="quran" title="21">
وَٱلَّذِينَ ءَامَنُوا۟ وَٱتَّبَعَتْهُمْ ذُرِّيَّتُهُم بِإِيمَٰنٍ أَلْحَقْنَا بِهِمْ ذُرِّيَّتَهُمْ وَمَآ أَلَتْنَٰهُم مِّنْ عَمَلِهِم مِّن شَىْءٍۢ ۚ كُلُّ ٱمْرِئٍۭ بِمَا كَسَبَ رَهِينٌۭ
</Aside>

<Aside type="reference" title="21">
کسانی که ایمان آوردند و فرزندانشان به پیروی از آنان ایمان اختیار کردند، فرزندانشان را (در بهشت) به آنان ملحق می‌کنیم؛ و از (پاداش) عملشان چیزی نمی‌کاهیم؛ و هر کس در گرو اعمال خویش است!
</Aside>

---

<Aside type="quran" title="22">
وَأَمْدَدْنَٰهُم بِفَٰكِهَةٍۢ وَلَحْمٍۢ مِّمَّا يَشْتَهُونَ
</Aside>

<Aside type="reference" title="22">
و همواره از انواع میوه‌ها و گوشتها -از هر نوع که بخواهند- در اختیارشان می‌گذاریم!
</Aside>

---

<Aside type="quran" title="23">
يَتَنَٰزَعُونَ فِيهَا كَأْسًۭا لَّا لَغْوٌۭ فِيهَا وَلَا تَأْثِيمٌۭ
</Aside>

<Aside type="reference" title="23">
آنها در بهشت جامهای پر از شراب طهور را که نه بیهوده‌گویی در آن است و نه گناه، از یکدیگر می‌گیرند!
</Aside>

---

<Aside type="quran" title="24">
۞ وَيَطُوفُ عَلَيْهِمْ غِلْمَانٌۭ لَّهُمْ كَأَنَّهُمْ لُؤْلُؤٌۭ مَّكْنُونٌۭ
</Aside>

<Aside type="reference" title="24">
و پیوسته بر گردشان نوجوانانی برای (خدمت) آنان گردش می‌کنند که همچون مرواریدهای درون صدفند!
</Aside>

---

<Aside type="quran" title="25">

</Aside>

<Aside type="reference" title="25">

</Aside>

---

<Aside type="quran" title="26">
قَالُوٓا۟ إِنَّا كُنَّا قَبْلُ فِىٓ أَهْلِنَا مُشْفِقِينَ
</Aside>

<Aside type="reference" title="26">

</Aside>

---

<Aside type="quran" title="27">

<Aside type="quran" title="28">

</Aside>

<Aside type="reference" title="28">

</Aside>

---

<Aside type="quran" title="29">

</Aside>

<Aside type="reference" title="29">

---

<Aside type="quran" title="30">

</Aside>

<Aside type="reference" title="30">

---

<Aside type="quran" title="31">

</Aside>

<Aside type="reference" title="31">

</Aside>

---

<Aside type="quran" title="32">

</Aside>

<Aside type="reference" title="32">

<Aside type="quran" title="33">

</Aside>

<Aside type="reference" title="33">

<Aside type="quran" title="34">

<Aside type="quran" title="35">

---

<Aside type="quran" title="37">

<Aside type="quran" title="38">

<Aside type="quran" title="40">

<Aside type="quran" title="43">

---

<Aside type="quran" title="44">

---

<Aside type="quran" title="45">

<Aside type="quran" title="46">

---

<Aside type="quran" title="47">

---

<Aside type="quran" title="48">
