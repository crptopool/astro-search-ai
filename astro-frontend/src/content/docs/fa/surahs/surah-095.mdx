<Aside type="quran" title="1">
بِّسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ وَٱلتِّينِ وَٱلزَّيْتُونِ
</Aside>

<Aside type="reference" title="1">
قسم به انجیر و زیتون [یا: قسم به سرزمین شام و بیت المقدّس‌]،
</Aside>

---

<Aside type="quran" title="2">
وَطُورِ سِينِينَ
</Aside>

<Aside type="reference" title="2">
و سوگند به «طور سینین»،
</Aside>

---

<Aside type="quran" title="3">
وَهَٰذَا ٱلْبَلَدِ ٱلْأَمِينِ
</Aside>

<Aside type="reference" title="3">
و قسم به این شهر امن [= مکّه‌]،
</Aside>

---

<Aside type="quran" title="4">
لَقَدْ خَلَقْنَا ٱلْإِنسَٰنَ فِىٓ أَحْسَنِ تَقْوِيمٍۢ
</Aside>

<Aside type="reference" title="4">
که ما انسان را در بهترین صورت و نظام آفریدیم،
</Aside>

---

<Aside type="quran" title="5">
ثُمَّ رَدَدْنَٰهُ أَسْفَلَ سَٰفِلِينَ
</Aside>

<Aside type="reference" title="5">
سپس او را به پایین‌ترین مرحله بازگرداندیم،
</Aside>

---

<Aside type="quran" title="6">
إِلَّا ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّٰلِحَٰتِ فَلَهُمْ أَجْرٌ غَيْرُ مَمْنُونٍۢ
</Aside>

<Aside type="reference" title="6">
مگر کسانی که ایمان آورده و اعمال صالح انجام داده‌اند که برای آنها پاداشی تمام‌نشدنی است!
</Aside>

---

<Aside type="quran" title="7">
فَمَا يُكَذِّبُكَ بَعْدُ بِٱلدِّينِ
</Aside>

<Aside type="reference" title="7">
پس چه چیز سبب می‌شود که بعد از این همه (دلایل روشن) روز جزا را انکار کنی؟!
</Aside>

---

<Aside type="quran" title="8">
أَلَيْسَ ٱللَّهُ بِأَحْكَمِ ٱلْحَٰكِمِينَ
</Aside>

<Aside type="reference" title="8">
آیا خداوند بهترین حکم‌کنندگان نیست؟!
</Aside>

---
