<Aside type="quran" title="1">
ٱلْحَمْدُ لِلَّهِ ٱلَّذِى خَلَقَ ٱلسَّمَٰوَٰتِ وَٱلْأَرْضَ وَجَعَلَ ٱلظُّلُمَٰتِ وَٱلنُّورَ ۖ ثُمَّ ٱلَّذِينَ كَفَرُوا۟ بِرَبِّهِمْ يَعْدِلُونَ
</Aside>

<Aside type="reference" title="1">
ستایش برای خداوندی است که آسمانها و زمین را آفرید، و ظلمتها و نور را پدید آورد؛ اما کافران برای پروردگار خود، شریک و شبیه قرارمی‌دهند (با اینکه دلایل توحید و یگانگی او، در آفرینش جهان آشکار است)!
</Aside>

---

<Aside type="quran" title="2">
هُوَ ٱلَّذِى خَلَقَكُم مِّن طِينٍۢ ثُمَّ قَضَىٰٓ أَجَلًۭا ۖ وَأَجَلٌۭ مُّسَمًّى عِندَهُۥ ۖ ثُمَّ أَنتُمْ تَمْتَرُونَ
</Aside>

<Aside type="reference" title="2">
او کسی است که شما را از گل آفرید؛ سپس مدتی مقرّر داشت (تا انسان تکامل یابد)؛ و اجل حتمی نزد اوست (و فقط او از آن آگاه است). با این همه، شما (مشرکان در توحید و یگانگی و قدرت او،) تردید می‌کنید!
</Aside>

---

<Aside type="quran" title="3">
وَهُوَ ٱللَّهُ فِى ٱلسَّمَٰوَٰتِ وَفِى ٱلْأَرْضِ ۖ يَعْلَمُ سِرَّكُمْ وَجَهْرَكُمْ وَيَعْلَمُ مَا تَكْسِبُونَ
</Aside>

<Aside type="reference" title="3">
اوست خداوند در آسمانها و در زمین؛ پنهان و آشکار شما را می‌داند؛ و از آنچه (انجام می‌دهید و) به دست می‌آورید، با خبر است.
</Aside>

<Aside type="quran" title="4">
وَمَا تَأْتِيهِم مِّنْ ءَايَةٍۢ مِّنْ ءَايَٰتِ رَبِّهِمْ إِلَّا كَانُوا۟ عَنْهَا مُعْرِضِينَ
</Aside>

<Aside type="reference" title="4">
هیچ نشانه و آیه‌ای از آیات پروردگارشان برای آنان نمی‌آید، مگر اینکه از آن رویگردان می‌شوند!
</Aside>

<Aside type="quran" title="5">
فَقَدْ كَذَّبُوا۟ بِٱلْحَقِّ لَمَّا جَآءَهُمْ ۖ فَسَوْفَ يَأْتِيهِمْ أَنۢبَٰٓؤُا۟ مَا كَانُوا۟ بِهِۦ يَسْتَهْزِءُونَ
</Aside>

<Aside type="reference" title="5">
آنان، حق را هنگامی که سراغشان آمد، تکذیب کردند! ولی بزودی خبر آنچه را به باد مسخره می‌گرفتند، به آنان می‌رسد؛ (و از نتایج کار خود، آگاه می‌شوند).
</Aside>

<Aside type="quran" title="6">
أَلَمْ يَرَوْا۟ كَمْ أَهْلَكْنَا مِن قَبْلِهِم مِّن قَرْنٍۢ مَّكَّنَّٰهُمْ فِى ٱلْأَرْضِ مَا لَمْ نُمَكِّن لَّكُمْ وَأَرْسَلْنَا ٱلسَّمَآءَ عَلَيْهِم مِّدْرَارًۭا وَجَعَلْنَا ٱلْأَنْهَٰرَ تَجْرِى مِن تَحْتِهِمْ فَأَهْلَكْنَٰهُم بِذُنُوبِهِمْ وَأَنشَأْنَا مِنۢ بَعْدِهِمْ قَرْنًا ءَاخَرِينَ
</Aside>

<Aside type="reference" title="6">
آیا ندیدند چقدر از اقوام پیشین را هلاک کردیم؟! اقوامی که (از شما نیرومندتر بودند؛ و) قدرتهایی به آنها داده بودیم که به شما ندادیم؛ بارانهای پی‌درپی برای آنها فرستادیم؛ و از زیر (آبادیهای) آنها، نهرها را جاری ساختیم؛ (اما هنگامی که سرکشی و طغیان کردند،) آنان را بخاطر گناهانشان نابود کردیم؛ و جمعیت دیگری بعد از آنان پدید آوردیم.
</Aside>

<Aside type="quran" title="7">
وَلَوْ نَزَّلْنَا عَلَيْكَ كِتَٰبًۭا فِى قِرْطَاسٍۢ فَلَمَسُوهُ بِأَيْدِيهِمْ لَقَالَ ٱلَّذِينَ كَفَرُوٓا۟ إِنْ هَٰذَآ إِلَّا سِحْرٌۭ مُّبِينٌۭ
</Aside>

<Aside type="reference" title="7">

</Aside>

<Aside type="quran" title="8">
وَقَالُوا۟ لَوْلَآ أُنزِلَ عَلَيْهِ مَلَكٌۭ ۖ وَلَوْ أَنزَلْنَا مَلَكًۭا لَّقُضِىَ ٱلْأَمْرُ ثُمَّ لَا يُنظَرُونَ
</Aside>

<Aside type="reference" title="8">
(از بهانه‌های آنها این بود که) گفتند: «چرا فرشته‌ای بر او نازل نشده (تا او را در دعوت مردم به سوی خدا همراهی کند؟!)» ولی اگر فرشته‌ای بفرستیم، (و موضوع، جنبه حسی و شهود پیدا کند،) کار تمام می‌شود؛ (یعنی اگر مخالفت کنند،) دیگر به آنها مهلت داده نخواهد شد (و همه هلاک می‌شوند).
</Aside>

<Aside type="quran" title="9">

</Aside>

<Aside type="reference" title="9">
و اگر او را فرشته‌ای قرارمی‌دادیم، حتماً وی را بصورت انسانی درمی‌آوردیم؛ (باز به پندار آنان،) کار را بر آنها مشتبه می‌ساختیم؛ همان‌طور که آنها کار را بر دیگران مشتبه می‌سازند!
</Aside>

<Aside type="quran" title="10">

<Aside type="reference" title="10">

</Aside>

<Aside type="quran" title="12">
