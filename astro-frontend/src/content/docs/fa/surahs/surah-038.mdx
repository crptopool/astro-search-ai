<Aside type="quran" title="1">
صٓ ۚ وَٱلْقُرْءَانِ ذِى ٱلذِّكْرِ
</Aside>

<Aside type="reference" title="1">
ص، سوگند به قرآنی که دارای ذکر است (که این کتاب، معجزه الهی است).
</Aside>

---

<Aside type="quran" title="2">
بَلِ ٱلَّذِينَ كَفَرُوا۟ فِى عِزَّةٍۢ وَشِقَاقٍۢ
</Aside>

<Aside type="reference" title="2">
ولی کافران گرفتار غرور اختلافند!
</Aside>

---

<Aside type="quran" title="3">
كَمْ أَهْلَكْنَا مِن قَبْلِهِم مِّن قَرْنٍۢ فَنَادَوا۟ وَّلَاتَ حِينَ مَنَاصٍۢ
</Aside>

<Aside type="reference" title="3">
چه بسیار اقوامی را که پیش از آنها هلاک کردیم؛ و به هنگام نزول عذاب فریاد می‌زدند (و کمک می‌خواستند) ولی وقت نجات گذشته بود!
</Aside>

---

<Aside type="quran" title="4">
وَعَجِبُوٓا۟ أَن جَآءَهُم مُّنذِرٌۭ مِّنْهُمْ ۖ وَقَالَ ٱلْكَٰفِرُونَ هَٰذَا سَٰحِرٌۭ كَذَّابٌ
</Aside>

<Aside type="reference" title="4">
آنها تعجّب کردند که پیامبر بیم‌دهنده‌ای از میان آنان به سویشان آمده؛ و کافران گفتند: این ساحر بسیار دروغگویی است!
</Aside>

---

<Aside type="quran" title="5">
أَجَعَلَ ٱلْءَالِهَةَ إِلَٰهًۭا وَٰحِدًا ۖ إِنَّ هَٰذَا لَشَىْءٌ عُجَابٌۭ
</Aside>

<Aside type="reference" title="5">
آیا او بجای اینهمه خدایان، خدای واحدی قرار داده؟! این براستی چیز عجیبی است!»
</Aside>

---

<Aside type="quran" title="6">
وَٱنطَلَقَ ٱلْمَلَأُ مِنْهُمْ أَنِ ٱمْشُوا۟ وَٱصْبِرُوا۟ عَلَىٰٓ ءَالِهَتِكُمْ ۖ إِنَّ هَٰذَا لَشَىْءٌۭ يُرَادُ
</Aside>

<Aside type="reference" title="6">
سرکردگان آنها بیرون آمدند و گفتند: «بروید و خدایانتان را محکم بچسبید، این چیزی است که خواسته‌اند (شما را گمراه کنند)!
</Aside>

---

<Aside type="quran" title="7">
مَا سَمِعْنَا بِهَٰذَا فِى ٱلْمِلَّةِ ٱلْءَاخِرَةِ إِنْ هَٰذَآ إِلَّا ٱخْتِلَٰقٌ
</Aside>

<Aside type="reference" title="7">
ما هرگز چنین چیزی در آیین دیگری نشنیده‌ایم؛ این تنها یک آئین ساختگی است!
</Aside>

---

<Aside type="quran" title="8">
أَءُنزِلَ عَلَيْهِ ٱلذِّكْرُ مِنۢ بَيْنِنَا ۚ بَلْ هُمْ فِى شَكٍّۢ مِّن ذِكْرِى ۖ بَل لَّمَّا يَذُوقُوا۟ عَذَابِ
</Aside>

<Aside type="reference" title="8">
آیا از میان همه ما، قرآن تنها بر او [= محمّد] نازل شده؟!» آنها در حقیقت در اصل وحی من تردید دارند، بلکه آنان هنوز عذاب الهی را نچشیده‌اند (که این چنین گستاخانه سخن می‌گویند)!
</Aside>

---

<Aside type="quran" title="9">
أَمْ عِندَهُمْ خَزَآئِنُ رَحْمَةِ رَبِّكَ ٱلْعَزِيزِ ٱلْوَهَّابِ
</Aside>

<Aside type="reference" title="9">
مگر خزاین رحمت پروردگار توانا و بخشنده‌ات نزد آنهاست (تا به هر کس میل دارند بدهند)؟!
</Aside>

---

<Aside type="quran" title="10">
أَمْ لَهُم مُّلْكُ ٱلسَّمَٰوَٰتِ وَٱلْأَرْضِ وَمَا بَيْنَهُمَا ۖ فَلْيَرْتَقُوا۟ فِى ٱلْأَسْبَٰبِ
</Aside>

<Aside type="reference" title="10">
یا اینکه مالکیّت و حاکمیّت آسمانها و زمین و آنچه میان این دو است از آن آنهاست؟! (اگر چنین است) با هر وسیله ممکن به آسمانها بروند (و جلو نزول وحی را بر قلب پاک محمّد بگیرند)!
</Aside>

---

<Aside type="quran" title="11">
جُندٌۭ مَّا هُنَالِكَ مَهْزُومٌۭ مِّنَ ٱلْأَحْزَابِ
</Aside>

<Aside type="reference" title="11">
(آری) اینها لشکر کوچک شکست‌خورده‌ای از احزابند!
</Aside>

---

<Aside type="quran" title="12">
كَذَّبَتْ قَبْلَهُمْ قَوْمُ نُوحٍۢ وَعَادٌۭ وَفِرْعَوْنُ ذُو ٱلْأَوْتَادِ
</Aside>

<Aside type="reference" title="12">
پیش از آنان قوم نوح و عاد و فرعون صاحب قدرت (پیامبران ما را) تکذیب کردند!
</Aside>

---

<Aside type="quran" title="13">
وَثَمُودُ وَقَوْمُ لُوطٍۢ وَأَصْحَٰبُ لْـَٔيْكَةِ ۚ أُو۟لَٰٓئِكَ ٱلْأَحْزَابُ
</Aside>

<Aside type="reference" title="13">
و (نیز) قوم ثمود و لوط و اصحاب الأیکه [= قوم شعیب‌]، اینها احزابی بودند (که به تکذیب پیامبران برخاستند)!
</Aside>

<Aside type="quran" title="14">

</Aside>

<Aside type="reference" title="14">
هر یک (از این گروه‌ها) رسولان را تکذیب کردند، و عذاب الهی درباره آنان تحقق یافت!
</Aside>

<Aside type="quran" title="15">
وَمَا يَنظُرُ هَٰٓؤُلَآءِ إِلَّا صَيْحَةًۭ وَٰحِدَةًۭ مَّا لَهَا مِن فَوَاقٍۢ
</Aside>

<Aside type="reference" title="15">
اینها (با این اعمالشان) جز یک صیحه آسمانی را انتظار نمی‌کشند که هیچ مهلت و بازگشتی برای آن وجود ندارد (و همگی را نابود می‌سازد)!
</Aside>

<Aside type="quran" title="16">

</Aside>

<Aside type="reference" title="16">
آنها (از روی خیره‌سری) گفتند: «پروردگارا! بهره ما را از عذاب هر چه زودتر قبل از روز حساب به ما ده!»
</Aside>

<Aside type="quran" title="17">

</Aside>

<Aside type="reference" title="17">
در برابر آنچه می‌گویند شکیبا باش، و به خاطر بیاور بنده ما داوود صاحب قدرت را، که او بسیار توبه‌کننده بود!
</Aside>

<Aside type="quran" title="18">

</Aside>

<Aside type="reference" title="18">

<Aside type="quran" title="19">

</Aside>

<Aside type="reference" title="19">
پرندگان را نیز دسته جمعی مسخّر او کردیم (تا همراه او تسبیح خدا گویند)؛و همه اینها بازگشت‌کننده به سوی او بودند!
</Aside>

<Aside type="quran" title="20">

</Aside>

<Aside type="reference" title="20">

<Aside type="quran" title="21">

<Aside type="reference" title="21">

<Aside type="quran" title="22">
إِذْ دَخَلُوا۟ عَلَىٰ دَاوُۥدَ فَفَزِعَ مِنْهُمْ ۖ قَالُوا۟ لَا تَخَفْ ۖ خَصْمَانِ بَغَىٰ بَعْضُنَا عَلَىٰ بَعْضٍۢ فَٱحْكُم بَيْنَنَا بِٱلْحَقِّ وَلَا تُشْطِطْ وَٱهْدِنَآ إِلَىٰ سَوَآءِ ٱلصِّرَٰطِ
</Aside>

<Aside type="reference" title="22">
در آن هنگام که (بی‌مقدمه) بر او وارد شدند و او از دیدن آنها وحشت کرد؛ گفتند: «نترس، دو نفر شاکی هستیم که یکی از ما بر دیگری ستم کرده؛ اکنون در میان ما بحق داوری کن و ستم روا مدار و ما را به راه راست هدایت کن!
</Aside>

<Aside type="quran" title="23">

</Aside>

<Aside type="reference" title="23">

</Aside>

<Aside type="quran" title="24">
قَالَ لَقَدْ ظَلَمَكَ بِسُؤَالِ نَعْجَتِكَ إِلَىٰ نِعَاجِهِۦ ۖ وَإِنَّ كَثِيرًۭا مِّنَ ٱلْخُلَطَآءِ لَيَبْغِى بَعْضُهُمْ عَلَىٰ بَعْضٍ إِلَّا ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّٰلِحَٰتِ وَقَلِيلٌۭ مَّا هُمْ ۗ وَظَنَّ دَاوُۥدُ أَنَّمَا فَتَنَّٰهُ فَٱسْتَغْفَرَ رَبَّهُۥ وَخَرَّ رَاكِعًۭا وَأَنَابَ ۩
</Aside>

<Aside type="reference" title="24">
(داوود) گفت: «مسلّماً او با درخواست یک میش تو برای افزودن آن به میشهایش، بر تو ستم نموده؛ و بسیاری از شریکان (و دوستان) به یکدیگر ستم می‌کنند، مگر کسانی که ایمان آورده و اعمال صالح انجام داده‌اند؛ امّا عدّه آنان کم است!» داوود دانست که ما او را (با این ماجرا) آزموده‌ایم، از این رو از پروردگارش طلب آمرزش نمود و به سجده افتاد و توبه کرد.
</Aside>
