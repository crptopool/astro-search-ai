<Aside type="quran" title="1">
لَمْ يَكُنِ ٱلَّذِينَ كَفَرُوا۟ مِنْ أَهْلِ ٱلْكِتَٰبِ وَٱلْمُشْرِكِينَ مُنفَكِّينَ حَتَّىٰ تَأْتِيَهُمُ ٱلْبَيِّنَةُ
</Aside>

<Aside type="reference" title="1">
کافران از اهل کتاب و مشرکان (می‌گفتند:) دست از آیین خود برنمی‌دارند تا دلیل روشنی برای آنها بیاید،
</Aside>

---

<Aside type="quran" title="2">
رَسُولٌۭ مِّنَ ٱللَّهِ يَتْلُوا۟ صُحُفًۭا مُّطَهَّرَةًۭ
</Aside>

<Aside type="reference" title="2">
پیامبری از سوی خدا (بیاید) که صحیفه‌های پاکی را (بر آنها) بخواند،
</Aside>

---

<Aside type="quran" title="3">
فِيهَا كُتُبٌۭ قَيِّمَةٌۭ
</Aside>

<Aside type="reference" title="3">
و در آن نوشته‌های صحیح و پرارزشی باشد! (ولی هنگامی که آمد ایمان نیاوردند، مانند اهل کتاب).
</Aside>

---

<Aside type="quran" title="4">
وَمَا تَفَرَّقَ ٱلَّذِينَ أُوتُوا۟ ٱلْكِتَٰبَ إِلَّا مِنۢ بَعْدِ مَا جَآءَتْهُمُ ٱلْبَيِّنَةُ
</Aside>

<Aside type="reference" title="4">
اهل کتاب (نیز در دین خدا) اختلاف نکردند مگر بعد از آنکه دلیل روشن برای آنان آمد!
</Aside>

---

<Aside type="quran" title="5">
وَمَآ أُمِرُوٓا۟ إِلَّا لِيَعْبُدُوا۟ ٱللَّهَ مُخْلِصِينَ لَهُ ٱلدِّينَ حُنَفَآءَ وَيُقِيمُوا۟ ٱلصَّلَوٰةَ وَيُؤْتُوا۟ ٱلزَّكَوٰةَ ۚ وَذَٰلِكَ دِينُ ٱلْقَيِّمَةِ
</Aside>

<Aside type="reference" title="5">
و به آنها دستوری داده نشده بود جز اینکه خدا را بپرستند در حالی که دین خود را برای او خالص کنند و از شرک به توحید بازگردند، نماز را برپا دارند و زکات را بپردازند؛ و این است آیین مستقیم و پایدار!
</Aside>

---

<Aside type="quran" title="6">
إِنَّ ٱلَّذِينَ كَفَرُوا۟ مِنْ أَهْلِ ٱلْكِتَٰبِ وَٱلْمُشْرِكِينَ فِى نَارِ جَهَنَّمَ خَٰلِدِينَ فِيهَآ ۚ أُو۟لَٰٓئِكَ هُمْ شَرُّ ٱلْبَرِيَّةِ
</Aside>

<Aside type="reference" title="6">
کافران از اهل کتاب و مشرکان در آتش دوزخند، جاودانه در آن می‌مانند؛ آنها بدترین مخلوقاتند!
</Aside>

---

<Aside type="quran" title="7">
إِنَّ ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّٰلِحَٰتِ أُو۟لَٰٓئِكَ هُمْ خَيْرُ ٱلْبَرِيَّةِ
</Aside>

<Aside type="reference" title="7">
(امّا) کسانی که ایمان آوردند و اعمال صالح انجام دادند، بهترین مخلوقات (خدا)یند!
</Aside>

---

<Aside type="quran" title="8">
جَزَآؤُهُمْ عِندَ رَبِّهِمْ جَنَّٰتُ عَدْنٍۢ تَجْرِى مِن تَحْتِهَا ٱلْأَنْهَٰرُ خَٰلِدِينَ فِيهَآ أَبَدًۭا ۖ رَّضِىَ ٱللَّهُ عَنْهُمْ وَرَضُوا۟ عَنْهُ ۚ ذَٰلِكَ لِمَنْ خَشِىَ رَبَّهُۥ
</Aside>

<Aside type="reference" title="8">
پاداش آنها نزد پروردگارشان باغهای بهشت جاویدان است که نهرها از زیر درختانش جاری است؛ همیشه در آن می‌مانند! (هم) خدا از آنها خشنود است و (هم) آنها از خدا خشنودند؛ و این (مقام والا) برای کسی است که از پروردگارش بترسد!
</Aside>
