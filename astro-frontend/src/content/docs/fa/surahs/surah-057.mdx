<Aside type="quran" title="1">
سَبَّحَ لِلَّهِ مَا فِى ٱلسَّمَٰوَٰتِ وَٱلْأَرْضِ ۖ وَهُوَ ٱلْعَزِيزُ ٱلْحَكِيمُ
</Aside>

<Aside type="reference" title="1">
آنچه در آسمانها و زمین است برای خدا تسبیح می‌گویند؛ و او عزیز و حکیم است.
</Aside>

---

<Aside type="quran" title="2">
لَهُۥ مُلْكُ ٱلسَّمَٰوَٰتِ وَٱلْأَرْضِ ۖ يُحْىِۦ وَيُمِيتُ ۖ وَهُوَ عَلَىٰ كُلِّ شَىْءٍۢ قَدِيرٌ
</Aside>

<Aside type="reference" title="2">
مالکیّت (و حاکمیّت) آسمانها و زمین از آن اوست؛ زنده می‌کند و می‌میراند؛ و او بر هر چیز توانا است!
</Aside>

---

<Aside type="quran" title="3">
هُوَ ٱلْأَوَّلُ وَٱلْءَاخِرُ وَٱلظَّٰهِرُ وَٱلْبَاطِنُ ۖ وَهُوَ بِكُلِّ شَىْءٍ عَلِيمٌ
</Aside>

<Aside type="reference" title="3">
اوّل و آخر و پیدا و پنهان اوست؛ و او به هر چیز داناست.
</Aside>

---

<Aside type="quran" title="4">
هُوَ ٱلَّذِى خَلَقَ ٱلسَّمَٰوَٰتِ وَٱلْأَرْضَ فِى سِتَّةِ أَيَّامٍۢ ثُمَّ ٱسْتَوَىٰ عَلَى ٱلْعَرْشِ ۚ يَعْلَمُ مَا يَلِجُ فِى ٱلْأَرْضِ وَمَا يَخْرُجُ مِنْهَا وَمَا يَنزِلُ مِنَ ٱلسَّمَآءِ وَمَا يَعْرُجُ فِيهَا ۖ وَهُوَ مَعَكُمْ أَيْنَ مَا كُنتُمْ ۚ وَٱللَّهُ بِمَا تَعْمَلُونَ بَصِيرٌۭ
</Aside>

<Aside type="reference" title="4">
او کسی است که آسمانها و زمین را در شش روز [= شش دوران‌] آفرید؛ سپس بر تخت قدرت قرار گرفت (و به تدبیر جهان پرداخت)؛ آنچه را در زمین فرو می‌رود می‌داند، و آنچه را از آن خارج می‌شود و آنچه از آسمان نازل می‌گردد و آنچه به آسمان بالا می‌رود؛ و هر جا باشید او با شما است، و خداوند نسبت به آنچه انجام می‌دهید بیناست!
</Aside>

---

<Aside type="quran" title="5">
لَّهُۥ مُلْكُ ٱلسَّمَٰوَٰتِ وَٱلْأَرْضِ ۚ وَإِلَى ٱللَّهِ تُرْجَعُ ٱلْأُمُورُ
</Aside>

<Aside type="reference" title="5">
مالکیّت آسمانها و زمین از آن اوست؛ و همه کارها به سوی او بازمی‌گردد.
</Aside>

---

<Aside type="quran" title="6">
يُولِجُ ٱلَّيْلَ فِى ٱلنَّهَارِ وَيُولِجُ ٱلنَّهَارَ فِى ٱلَّيْلِ ۚ وَهُوَ عَلِيمٌۢ بِذَاتِ ٱلصُّدُورِ
</Aside>

<Aside type="reference" title="6">
شب را در روز می‌کند و روز را در شب؛ و او به آنچه در دل سینه‌ها وجود دارد داناست.
</Aside>

<Aside type="quran" title="7">
ءَامِنُوا۟ بِٱللَّهِ وَرَسُولِهِۦ وَأَنفِقُوا۟ مِمَّا جَعَلَكُم مُّسْتَخْلَفِينَ فِيهِ ۖ فَٱلَّذِينَ ءَامَنُوا۟ مِنكُمْ وَأَنفَقُوا۟ لَهُمْ أَجْرٌۭ كَبِيرٌۭ
</Aside>

<Aside type="reference" title="7">
به خدا و رسولش ایمان بیاورید و از آنچه شما را جانشین و نماینده (خود) در آن قرار داده انفاق کنید؛ (زیرا) کسانی که از شما ایمان بیاورند و انفاق کنند، اجر بزرگی دارند!
</Aside>

---

<Aside type="quran" title="8">
وَمَا لَكُمْ لَا تُؤْمِنُونَ بِٱللَّهِ ۙ وَٱلرَّسُولُ يَدْعُوكُمْ لِتُؤْمِنُوا۟ بِرَبِّكُمْ وَقَدْ أَخَذَ مِيثَٰقَكُمْ إِن كُنتُم مُّؤْمِنِينَ
</Aside>

<Aside type="reference" title="8">
چرا به خدا ایمان نیاورید در حالی که رسول (او) شما را می‌خواند که به پروردگارتان ایمان بیاورید، و از شما پیمان گرفته است (پیمانی از طریق فطرت و خِرد)، اگر آماده ایمان‌آوردنید.
</Aside>

---

<Aside type="quran" title="9">
هُوَ ٱلَّذِى يُنَزِّلُ عَلَىٰ عَبْدِهِۦٓ ءَايَٰتٍۭ بَيِّنَٰتٍۢ لِّيُخْرِجَكُم مِّنَ ٱلظُّلُمَٰتِ إِلَى ٱلنُّورِ ۚ وَإِنَّ ٱللَّهَ بِكُمْ لَرَءُوفٌۭ رَّحِيمٌۭ
</Aside>

<Aside type="reference" title="9">
او کسی است که آیات روشنی بر بنده‌اش [= محمد] نازل می‌کند تا شما را از تاریکیها به سوی نور برد؛ و خداوند نسبت به شما مهربان و رحیم است.
</Aside>

---

<Aside type="quran" title="10">
وَمَا لَكُمْ أَلَّا تُنفِقُوا۟ فِى سَبِيلِ ٱللَّهِ وَلِلَّهِ مِيرَٰثُ ٱلسَّمَٰوَٰتِ وَٱلْأَرْضِ ۚ لَا يَسْتَوِى مِنكُم مَّنْ أَنفَقَ مِن قَبْلِ ٱلْفَتْحِ وَقَٰتَلَ ۚ أُو۟لَٰٓئِكَ أَعْظَمُ دَرَجَةًۭ مِّنَ ٱلَّذِينَ أَنفَقُوا۟ مِنۢ بَعْدُ وَقَٰتَلُوا۟ ۚ وَكُلًّۭا وَعَدَ ٱللَّهُ ٱلْحُسْنَىٰ ۚ وَٱللَّهُ بِمَا تَعْمَلُونَ خَبِيرٌۭ
</Aside>

<Aside type="reference" title="10">
چرا در راه خدا انفاق نکنید در حالی که میراث آسمانها و زمین همه از آن خداست (و کسی چیزی را با خود نمی‌برد)! کسانی که قبل از پیروزی انفاق کردند و جنگیدند (با کسانی که پس از پیروزی انفاق کردند) یکسان نیستند؛ آنها بلندمقامتر از کسانی هستند که بعد از فتح انفاق نمودند و جهاد کردند؛ و خداوند به هر دو وعده نیک داده؛ و خدا به آنچه انجام می‌دهید آگاه است.
</Aside>

<Aside type="quran" title="11">
مَّن ذَا ٱلَّذِى يُقْرِضُ ٱللَّهَ قَرْضًا حَسَنًۭا فَيُضَٰعِفَهُۥ لَهُۥ وَلَهُۥٓ أَجْرٌۭ كَرِيمٌۭ
</Aside>

<Aside type="reference" title="11">
کیست که به خدا وام نیکو دهد (و از اموالی که به او ارزانی داشته انفاق کند) تا خداوند آن را برای او چندین برابر کند؟ و برای او پاداش پرارزشی است!
</Aside>

<Aside type="quran" title="12">
يَوْمَ تَرَى ٱلْمُؤْمِنِينَ وَٱلْمُؤْمِنَٰتِ يَسْعَىٰ نُورُهُم بَيْنَ أَيْدِيهِمْ وَبِأَيْمَٰنِهِم بُشْرَىٰكُمُ ٱلْيَوْمَ جَنَّٰتٌۭ تَجْرِى مِن تَحْتِهَا ٱلْأَنْهَٰرُ خَٰلِدِينَ فِيهَا ۚ ذَٰلِكَ هُوَ ٱلْفَوْزُ ٱلْعَظِيمُ
</Aside>

<Aside type="reference" title="12">
(این پاداش بزرگ) در روزی است که مردان و زنان باایمان را می‌نگری که نورشان پیش‌رو و در سمت راستشان بسرعت حرکت می‌کند (و به آنها می‌گویند:) بشارت باد بر شما امروز به باغهایی از بهشت که نهرها زیر (درختان) آن جاری است؛ جاودانه در آن خواهید ماند! و این همان رستگاری بزرگ است!
</Aside>

<Aside type="quran" title="13">
يَوْمَ يَقُولُ ٱلْمُنَٰفِقُونَ وَٱلْمُنَٰفِقَٰتُ لِلَّذِينَ ءَامَنُوا۟ ٱنظُرُونَا نَقْتَبِسْ مِن نُّورِكُمْ قِيلَ ٱرْجِعُوا۟ وَرَآءَكُمْ فَٱلْتَمِسُوا۟ نُورًۭا فَضُرِبَ بَيْنَهُم بِسُورٍۢ لَّهُۥ بَابٌۢ بَاطِنُهُۥ فِيهِ ٱلرَّحْمَةُ وَظَٰهِرُهُۥ مِن قِبَلِهِ ٱلْعَذَابُ
</Aside>

<Aside type="reference" title="13">
روزی که مردان و زنان منافق به مؤمنان می‌گویند: «نظری به ما بیفکنید تا از نور شما پرتوی برگیریم!» به آنها گفته می‌شود: «به پشت سر خود بازگردید و کسب نور کنید!» در این هنگام دیواری میان آنها زده می‌شود که دری دارد، درونش رحمت است و برونش عذاب!
</Aside>

<Aside type="quran" title="14">
يُنَادُونَهُمْ أَلَمْ نَكُن مَّعَكُمْ ۖ قَالُوا۟ بَلَىٰ وَلَٰكِنَّكُمْ فَتَنتُمْ أَنفُسَكُمْ وَتَرَبَّصْتُمْ وَٱرْتَبْتُمْ وَغَرَّتْكُمُ ٱلْأَمَانِىُّ حَتَّىٰ جَآءَ أَمْرُ ٱللَّهِ وَغَرَّكُم بِٱللَّهِ ٱلْغَرُورُ
</Aside>

<Aside type="reference" title="14">
آنها را صدا می‌زنند: «مگر ما با شما نبودیم؟!» می‌گویند: «آری، ولی شما خود را به هلاکت افکندید و انتظار (مرگ پیامبر را) کشیدید، و (در همه چیز) شکّ و تردید داشتید، و آرزوهای دور و دراز شما را فریب داد تا فرمان خدا فرا رسید، و شیطان فریبکار شما را در برابر (فرمان) خداوند فریب داد!
</Aside>

<Aside type="quran" title="16">

<Aside type="quran" title="19">

<Aside type="reference" title="19">

<Aside type="quran" title="20">
ٱعْلَمُوٓا۟ أَنَّمَا ٱلْحَيَوٰةُ ٱلدُّنْيَا لَعِبٌۭ وَلَهْوٌۭ وَزِينَةٌۭ وَتَفَاخُرٌۢ بَيْنَكُمْ وَتَكَاثُرٌۭ فِى ٱلْأَمْوَٰلِ وَٱلْأَوْلَٰدِ ۖ كَمَثَلِ غَيْثٍ أَعْجَبَ ٱلْكُفَّارَ نَبَاتُهُۥ ثُمَّ يَهِيجُ فَتَرَىٰهُ مُصْفَرًّۭا ثُمَّ يَكُونُ حُطَٰمًۭا ۖ وَفِى ٱلْءَاخِرَةِ عَذَابٌۭ شَدِيدٌۭ وَمَغْفِرَةٌۭ مِّنَ ٱللَّهِ وَرِضْوَٰنٌۭ ۚ وَمَا ٱلْحَيَوٰةُ ٱلدُّنْيَآ إِلَّا مَتَٰعُ ٱلْغُرُورِ
</Aside>

<Aside type="reference" title="20">
بدانید زندگی دنیا تنها بازی و سرگرمی و تجمّل پرستی و فخرفروشی در میان شما و افزون طلبی در اموال و فرزندان است، همانند بارانی که محصولش کشاورزان را در شگفتی فرو می‌برد، سپس خشک می‌شود بگونه‌ای که آن را زردرنگ می‌بینی؛ سپس تبدیل به کاه می‌شود! و در آخرت، عذاب شدید است یا مغفرت و رضای الهی؛ و (به هر حال) زندگی دنیا چیزی جز متاع فریب نیست!
</Aside>

سَابِقُوٓا۟ إِلَىٰ مَغْفِرَةٍۢ مِّن رَّبِّكُمْ وَجَنَّةٍ عَرْضُهَا كَعَرْضِ ٱلسَّمَآءِ وَٱلْأَرْضِ أُعِدَّتْ لِلَّذِينَ ءَامَنُوا۟ بِٱللَّهِ وَرُسُلِهِۦ ۚ ذَٰلِكَ فَضْلُ ٱللَّهِ يُؤْتِيهِ مَن يَشَآءُ ۚ وَٱللَّهُ ذُو ٱلْفَضْلِ ٱلْعَظِيمِ
</Aside>

<Aside type="reference" title="21">
به پیش تازید برای رسیدن به مغفرت پروردگارتان و بهشتی که پهنه آن مانند پهنه آسمان و زمین است و برای کسانی که به خدا و رسولانش ایمان آورده‌اند؛ آماده شده است، این فضل خداوند است که به هر کس بخواهد می‌دهد؛ و خداوند صاحب فضل عظیم است!
</Aside>

<Aside type="quran" title="22">
مَآ أَصَابَ مِن مُّصِيبَةٍۢ فِى ٱلْأَرْضِ وَلَا فِىٓ أَنفُسِكُمْ إِلَّا فِى كِتَٰبٍۢ مِّن قَبْلِ أَن نَّبْرَأَهَآ ۚ إِنَّ ذَٰلِكَ عَلَى ٱللَّهِ يَسِيرٌۭ
</Aside>

<Aside type="reference" title="22">
هیچ مصیبتی (ناخواسته) در زمین و نه در وجود شما روی نمی‌دهد مگر اینکه همه آنها قبل از آنکه زمین را بیافرینیم در لوح محفوظ ثبت است؛ و این امر برای خدا آسان است!
</Aside>

<Aside type="quran" title="23">
لِّكَيْلَا تَأْسَوْا۟ عَلَىٰ مَا فَاتَكُمْ وَلَا تَفْرَحُوا۟ بِمَآ ءَاتَىٰكُمْ ۗ وَٱللَّهُ لَا يُحِبُّ كُلَّ مُخْتَالٍۢ فَخُورٍ
</Aside>

<Aside type="reference" title="23">
این بخاطر آن است که برای آنچه از دست داده‌اید تأسف نخورید، و به آنچه به شما داده است دلبسته و شادمان نباشید؛ و خداوند هیچ متکبّر فخرفروشی را دوست ندارد!
</Aside>

<Aside type="quran" title="24">
ٱلَّذِينَ يَبْخَلُونَ وَيَأْمُرُونَ ٱلنَّاسَ بِٱلْبُخْلِ ۗ وَمَن يَتَوَلَّ فَإِنَّ ٱللَّهَ هُوَ ٱلْغَنِىُّ ٱلْحَمِيدُ
</Aside>

<Aside type="reference" title="24">
همانها که بخل می‌ورزند و مردم را به بخل دعوت می‌کنند؛ و هر کس (از این فرمان) روی‌گردان شود، (به خود زیان می‌رساند نه به خدا)، چرا که خداوند بی‌نیاز و شایسته ستایش است!
</Aside>

<Aside type="quran" title="25">
لَقَدْ أَرْسَلْنَا رُسُلَنَا بِٱلْبَيِّنَٰتِ وَأَنزَلْنَا مَعَهُمُ ٱلْكِتَٰبَ وَٱلْمِيزَانَ لِيَقُومَ ٱلنَّاسُ بِٱلْقِسْطِ ۖ وَأَنزَلْنَا ٱلْحَدِيدَ فِيهِ بَأْسٌۭ شَدِيدٌۭ وَمَنَٰفِعُ لِلنَّاسِ وَلِيَعْلَمَ ٱللَّهُ مَن يَنصُرُهُۥ وَرُسُلَهُۥ بِٱلْغَيْبِ ۚ إِنَّ ٱللَّهَ قَوِىٌّ عَزِيزٌۭ
</Aside>

<Aside type="reference" title="25">
ما رسولان خود را با دلایل روشن فرستادیم، و با آنها کتاب (آسمانی) و میزان (شناسایی حقّ از باطل و قوانین عادلانه) نازل کردیم تا مردم قیام به عدالت کنند؛ و آهن را نازل کردیم که در آن نیروی شدید و منافعی برای مردم است، تا خداوند بداند چه کسی او و رسولانش را یاری می‌کند بی‌آنکه او را ببینند؛ خداوند قوّی و شکست‌ناپذیر است!
</Aside>

<Aside type="quran" title="26">
وَلَقَدْ أَرْسَلْنَا نُوحًۭا وَإِبْرَٰهِيمَ وَجَعَلْنَا فِى ذُرِّيَّتِهِمَا ٱلنُّبُوَّةَ وَٱلْكِتَٰبَ ۖ فَمِنْهُم مُّهْتَدٍۢ ۖ وَكَثِيرٌۭ مِّنْهُمْ فَٰسِقُونَ
</Aside>

<Aside type="reference" title="26">
ما نوح و ابراهیم را فرستادیم، و در دودمان آن دو نبوّت و کتاب قرار دادیم؛ بعضی از آنها هدایت یافته‌اند و بسیاری از آنها گنهکارند.
</Aside>

<Aside type="quran" title="27">
ثُمَّ قَفَّيْنَا عَلَىٰٓ ءَاثَٰرِهِم بِرُسُلِنَا وَقَفَّيْنَا بِعِيسَى ٱبْنِ مَرْيَمَ وَءَاتَيْنَٰهُ ٱلْإِنجِيلَ وَجَعَلْنَا فِى قُلُوبِ ٱلَّذِينَ ٱتَّبَعُوهُ رَأْفَةًۭ وَرَحْمَةًۭ وَرَهْبَانِيَّةً ٱبْتَدَعُوهَا مَا كَتَبْنَٰهَا عَلَيْهِمْ إِلَّا ٱبْتِغَآءَ رِضْوَٰنِ ٱللَّهِ فَمَا رَعَوْهَا حَقَّ رِعَايَتِهَا ۖ فَـَٔاتَيْنَا ٱلَّذِينَ ءَامَنُوا۟ مِنْهُمْ أَجْرَهُمْ ۖ وَكَثِيرٌۭ مِّنْهُمْ فَٰسِقُونَ
</Aside>

<Aside type="reference" title="27">
سپس در پی آنان رسولان دیگر خود را فرستادیم، و بعد از آنان عیسی بن مریم را مبعوث کردیم و به او انجیل عطا کردیم، و در دل کسانی که از او پیروی کردند رأفت و رحمت قرار دادیم؛ و رهبانیّتی را که ابداع کرده بودند، ما بر آنان مقرّر نداشته بودیم؛ گرچه هدفشان جلب خشنودی خدا بود، ولی حقّ آن را رعایت نکردند؛ از این‌رو ما به کسانی از آنها که ایمان آوردند پاداششان را دادیم؛ و بسیاری از آنها فاسقند!
</Aside>

<Aside type="quran" title="28">
يَٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوا۟ ٱتَّقُوا۟ ٱللَّهَ وَءَامِنُوا۟ بِرَسُولِهِۦ يُؤْتِكُمْ كِفْلَيْنِ مِن رَّحْمَتِهِۦ وَيَجْعَل لَّكُمْ نُورًۭا تَمْشُونَ بِهِۦ وَيَغْفِرْ لَكُمْ ۚ وَٱللَّهُ غَفُورٌۭ رَّحِيمٌۭ
</Aside>

<Aside type="reference" title="28">
ای کسانی که ایمان آورده‌اید! تقوای الهی پیشه کنید و به رسولش ایمان بیاورید تا دو سهم از رحمتش به شما ببخشد و برای شما نوری قرار دهد که با آن (در میان مردم و در مسیر زندگی خود) راه بروید و گناهان شما را ببخشد؛ و خداوند غفور و رحیم است.
</Aside>
