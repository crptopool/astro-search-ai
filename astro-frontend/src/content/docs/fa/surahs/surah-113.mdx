<Aside type="quran" title="1">
قُلْ أَعُوذُ بِرَبِّ ٱلْفَلَقِ
</Aside>

<Aside type="reference" title="1">
بگو: پناه می‌برم به پروردگار سپیده صبح،
</Aside>

---

<Aside type="quran" title="2">
مِن شَرِّ مَا خَلَقَ
</Aside>

<Aside type="reference" title="2">
از شرّ تمام آنچه آفریده است؛
</Aside>

---

<Aside type="quran" title="3">
وَمِن شَرِّ غَاسِقٍ إِذَا وَقَبَ
</Aside>

<Aside type="reference" title="3">
و از شرّ هر موجود شرور هنگامی که شبانه وارد می‌شود؛
</Aside>

---

<Aside type="quran" title="4">
وَمِن شَرِّ ٱلنَّفَّٰثَٰتِ فِى ٱلْعُقَدِ
</Aside>

<Aside type="reference" title="4">
و از شرّ آنها که با افسون در گره‌ها می‌دمند (و هر تصمیمی را سست می‌کنند)؛
</Aside>

---

<Aside type="quran" title="5">
وَمِن شَرِّ حَاسِدٍ إِذَا حَسَدَ
</Aside>

<Aside type="reference" title="5">
و از شرّ هر حسودی هنگامی که حسد می‌ورزد!
</Aside>
