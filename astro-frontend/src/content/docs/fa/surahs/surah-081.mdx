إِذَا ٱلشَّمْسُ كُوِّرَتْ
</Aside>

<Aside type="reference" title="1">
در آن هنگام که خورشید در هم پیچیده شود،
</Aside>

<Aside type="quran" title="2">
وَإِذَا ٱلنُّجُومُ ٱنكَدَرَتْ
</Aside>

<Aside type="reference" title="2">
و در آن هنگام که ستارگان بی‌فروغ شوند،
</Aside>

وَإِذَا ٱلْجِبَالُ سُيِّرَتْ
</Aside>

<Aside type="reference" title="3">
و در آن هنگام که کوه‌ها به حرکت درآیند،
</Aside>

---

<Aside type="quran" title="4">
وَإِذَا ٱلْعِشَارُ عُطِّلَتْ
</Aside>

<Aside type="reference" title="4">
و در آن هنگام که باارزش‌ترین اموال به دست فراموشی سپرده شود،
</Aside>

---

<Aside type="quran" title="5">
وَإِذَا ٱلْوُحُوشُ حُشِرَتْ
</Aside>

<Aside type="reference" title="5">
و در آن هنگام که وحوش جمع شوند،
</Aside>

---

<Aside type="quran" title="6">
وَإِذَا ٱلْبِحَارُ سُجِّرَتْ
</Aside>

<Aside type="reference" title="6">
و در آن هنگام که دریاها برافروخته شوند،
</Aside>

---

<Aside type="quran" title="7">
وَإِذَا ٱلنُّفُوسُ زُوِّجَتْ
</Aside>

<Aside type="reference" title="7">
و در آن هنگام که هر کس با همسان خود قرین گردد،
</Aside>

---

وَإِذَا ٱلْمَوْءُۥدَةُ سُئِلَتْ
</Aside>

<Aside type="reference" title="8">
و در آن هنگام که از دختران زنده به گور شده سؤال شود:
</Aside>

<Aside type="quran" title="9">
بِأَىِّ ذَنۢبٍۢ قُتِلَتْ
</Aside>

<Aside type="reference" title="9">
به کدامین گناه کشته شدند؟!
</Aside>

<Aside type="quran" title="10">
وَإِذَا ٱلصُّحُفُ نُشِرَتْ
</Aside>

<Aside type="reference" title="10">
و در آن هنگام که نامه‌های اعمال گشوده شود،
</Aside>

---

<Aside type="quran" title="11">
وَإِذَا ٱلسَّمَآءُ كُشِطَتْ
</Aside>

<Aside type="reference" title="11">
و در آن هنگام که پرده از روی آسمان برگرفته شود،
</Aside>

---

<Aside type="quran" title="12">
وَإِذَا ٱلْجَحِيمُ سُعِّرَتْ
</Aside>

<Aside type="reference" title="12">
و در آن هنگام که دوزخ شعله‌ور گردد،
</Aside>

<Aside type="quran" title="13">
وَإِذَا ٱلْجَنَّةُ أُزْلِفَتْ
</Aside>

<Aside type="reference" title="13">
و در آن هنگام که بهشت نزدیک شود،
</Aside>

<Aside type="quran" title="14">

</Aside>

<Aside type="reference" title="14">

</Aside>

فَلَآ أُقْسِمُ بِٱلْخُنَّسِ
</Aside>

<Aside type="reference" title="15">
سوگند به ستارگانی که بازمی‌گردند،
</Aside>

---

<Aside type="quran" title="16">
ٱلْجَوَارِ ٱلْكُنَّسِ
</Aside>

<Aside type="reference" title="16">
حرکت می‌کنند و از دیده‌ها پنهان می‌شوند،
</Aside>

---

<Aside type="quran" title="17">
وَٱلَّيْلِ إِذَا عَسْعَسَ
</Aside>

<Aside type="reference" title="17">
و قسم به شب، هنگامی که پشت کند و به آخر رسد،
</Aside>

---

<Aside type="quran" title="18">
وَٱلصُّبْحِ إِذَا تَنَفَّسَ
</Aside>

<Aside type="reference" title="18">
و به صبح، هنگامی که تنفّس کند،
</Aside>

---

<Aside type="quran" title="19">

</Aside>

<Aside type="reference" title="19">
که این (قرآن) کلام فرستاده بزرگواری است [= جبرئیل امین‌]
</Aside>

---

<Aside type="quran" title="20">
ذِى قُوَّةٍ عِندَ ذِى ٱلْعَرْشِ مَكِينٍۢ
</Aside>

<Aside type="reference" title="20">
که صاحب قدرت است و نزد (خداوند) صاحب عرش، مقام والائی دارد!
</Aside>

---

<Aside type="quran" title="21">
مُّطَاعٍۢ ثَمَّ أَمِينٍۢ
</Aside>

<Aside type="reference" title="21">
در آسمانها مورد اطاعت (فرشتگان) و امین است!
</Aside>

---

<Aside type="quran" title="22">
وَمَا صَاحِبُكُم بِمَجْنُونٍۢ
</Aside>

<Aside type="reference" title="22">
و مصاحب شما [= پیامبر] دیوانه نیست!
</Aside>

---

<Aside type="quran" title="23">
وَلَقَدْ رَءَاهُ بِٱلْأُفُقِ ٱلْمُبِينِ
</Aside>

<Aside type="reference" title="23">
او (جبرئیل) را در اُفق روشن دیده است!
</Aside>

---

<Aside type="quran" title="24">
وَمَا هُوَ عَلَى ٱلْغَيْبِ بِضَنِينٍۢ
</Aside>

<Aside type="reference" title="24">
و او نسبت به آنچه از طریق وحی دریافت داشته بخل ندارد!
</Aside>

---

<Aside type="quran" title="25">

</Aside>

<Aside type="reference" title="25">
این (قرآن) گفته شیطان رجیم نیست!
</Aside>

---

<Aside type="quran" title="26">
فَأَيْنَ تَذْهَبُونَ
</Aside>

<Aside type="reference" title="26">

</Aside>

---

<Aside type="quran" title="27">

</Aside>

<Aside type="reference" title="27">
این قرآن چیزی جز تذکّری برای جهانیان نیست،
</Aside>

<Aside type="quran" title="28">

</Aside>

<Aside type="reference" title="28">

</Aside>

</Aside>

<Aside type="reference" title="29">
