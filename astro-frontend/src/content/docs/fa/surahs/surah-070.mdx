<Aside type="reference" title="1">

</Aside>

---

<Aside type="quran" title="2">
لِّلْكَٰفِرِينَ لَيْسَ لَهُۥ دَافِعٌۭ
</Aside>

<Aside type="reference" title="2">

</Aside>

---

<Aside type="quran" title="3">
مِّنَ ٱللَّهِ ذِى ٱلْمَعَارِجِ
</Aside>

<Aside type="reference" title="3">
از سوی خداوند ذی المعارج [= خداوندی که فرشتگانش بر آسمانها صعود و عروج می‌کنند]!
</Aside>

---

<Aside type="quran" title="4">
تَعْرُجُ ٱلْمَلَٰٓئِكَةُ وَٱلرُّوحُ إِلَيْهِ فِى يَوْمٍۢ كَانَ مِقْدَارُهُۥ خَمْسِينَ أَلْفَ سَنَةٍۢ
</Aside>

<Aside type="reference" title="4">
فرشتگان و روح [= فرشته مقرّب خداوند] بسوی او عروج می‌کنند در آن روزی که مقدارش پنجاه هزار سال است!
</Aside>

---

<Aside type="quran" title="5">
فَٱصْبِرْ صَبْرًۭا جَمِيلًا
</Aside>

<Aside type="reference" title="5">
پس صبر جمیل پیشه کن،
</Aside>

---

<Aside type="quran" title="6">
إِنَّهُمْ يَرَوْنَهُۥ بَعِيدًۭا
</Aside>

<Aside type="reference" title="6">
زیرا آنها آن روز را دور می‌بینند،
</Aside>

---

<Aside type="quran" title="7">
وَنَرَىٰهُ قَرِيبًۭا
</Aside>

<Aside type="reference" title="7">
و ما آن را نزدیک می‌بینیم!
</Aside>

---

<Aside type="quran" title="8">
يَوْمَ تَكُونُ ٱلسَّمَآءُ كَٱلْمُهْلِ
</Aside>

<Aside type="reference" title="8">
همان روز که آسمان همچون فلز گداخته می‌شود،
</Aside>

---

<Aside type="quran" title="9">
وَتَكُونُ ٱلْجِبَالُ كَٱلْعِهْنِ
</Aside>

<Aside type="reference" title="9">
و کوه‌ها مانند پشم رنگین متلاشی خواهد بود،
</Aside>

---

<Aside type="quran" title="10">
وَلَا يَسْـَٔلُ حَمِيمٌ حَمِيمًۭا
</Aside>

<Aside type="reference" title="10">
و هیچ دوست صمیمی سراغ دوستش را نمی‌گیرد!
</Aside>

---

<Aside type="quran" title="11">
يُبَصَّرُونَهُمْ ۚ يَوَدُّ ٱلْمُجْرِمُ لَوْ يَفْتَدِى مِنْ عَذَابِ يَوْمِئِذٍۭ بِبَنِيهِ
</Aside>

<Aside type="reference" title="11">
آنها را نشانشان می‌دهند (ولی هر کس گرفتار کار خویشتن است)، چنان است که گنهکار دوست می‌دارد فرزندان خود را در برابر عذاب آن روز فدا کند،
</Aside>

---

<Aside type="quran" title="12">
وَصَٰحِبَتِهِۦ وَأَخِيهِ
</Aside>

<Aside type="reference" title="12">
و همسر و برادرش را،
</Aside>

---

<Aside type="quran" title="13">
وَفَصِيلَتِهِ ٱلَّتِى تُـْٔوِيهِ
</Aside>

<Aside type="reference" title="13">
و قبیله‌اش را که همیشه از او حمایت می‌کرد،
</Aside>

---

<Aside type="quran" title="14">
وَمَن فِى ٱلْأَرْضِ جَمِيعًۭا ثُمَّ يُنجِيهِ
</Aside>

<Aside type="reference" title="14">
و همه مردم روی زمین را تا مایه نجاتش گردند؛
</Aside>

---

<Aside type="quran" title="15">
كَلَّآ ۖ إِنَّهَا لَظَىٰ
</Aside>

<Aside type="reference" title="15">
امّا هرگز چنین نیست (که با اینها بتوان نجات یافت، آری) شعله‌های سوزان آتش است،
</Aside>

---

<Aside type="quran" title="16">
نَزَّاعَةًۭ لِّلشَّوَىٰ
</Aside>

<Aside type="reference" title="16">
دست و پا و پوست سر را می‌کند و می‌برد!
</Aside>

---

<Aside type="quran" title="17">
تَدْعُوا۟ مَنْ أَدْبَرَ وَتَوَلَّىٰ
</Aside>

<Aside type="reference" title="17">
و کسانی را که به فرمان خدا پشت کردند صدا می‌زند،
</Aside>

---

<Aside type="quran" title="18">
وَجَمَعَ فَأَوْعَىٰٓ
</Aside>

<Aside type="reference" title="18">
و (همچنین آنها که) اموال را جمع و ذخیره کردند!
</Aside>

---

<Aside type="quran" title="19">
۞ إِنَّ ٱلْإِنسَٰنَ خُلِقَ هَلُوعًا
</Aside>

<Aside type="reference" title="19">
به یقین انسان حریص و کم‌طاقت آفریده شده است،
</Aside>

---

<Aside type="quran" title="20">
إِذَا مَسَّهُ ٱلشَّرُّ جَزُوعًۭا
</Aside>

<Aside type="reference" title="20">
هنگامی که بدی به او رسد بیتابی می‌کند،
</Aside>

---

<Aside type="quran" title="21">
وَإِذَا مَسَّهُ ٱلْخَيْرُ مَنُوعًا
</Aside>

<Aside type="reference" title="21">
و هنگامی که خوبی به او رسد مانع دیگران می‌شود (و بخل می‌ورزد)،
</Aside>

---

<Aside type="quran" title="22">
إِلَّا ٱلْمُصَلِّينَ
</Aside>

<Aside type="reference" title="22">
مگر نمازگزاران،
</Aside>

---

<Aside type="quran" title="23">
ٱلَّذِينَ هُمْ عَلَىٰ صَلَاتِهِمْ دَآئِمُونَ
</Aside>

<Aside type="reference" title="23">
آنها که نمازها را پیوسته بجا می‌آورند،
</Aside>

---

<Aside type="quran" title="24">
وَٱلَّذِينَ فِىٓ أَمْوَٰلِهِمْ حَقٌّۭ مَّعْلُومٌۭ
</Aside>

<Aside type="reference" title="24">
و آنها که در اموالشان حق معلومی است...
</Aside>

---

<Aside type="quran" title="25">
لِّلسَّآئِلِ وَٱلْمَحْرُومِ
</Aside>

<Aside type="reference" title="25">
برای تقاضاکننده و محروم،
</Aside>

---

<Aside type="quran" title="26">
وَٱلَّذِينَ يُصَدِّقُونَ بِيَوْمِ ٱلدِّينِ
</Aside>

<Aside type="reference" title="26">
و آنها که به روز جزا ایمان دارند،
</Aside>

---

<Aside type="quran" title="27">
وَٱلَّذِينَ هُم مِّنْ عَذَابِ رَبِّهِم مُّشْفِقُونَ
</Aside>

<Aside type="reference" title="27">
و آنها که از عذاب پروردگارشان بیمناکند،
</Aside>

---

<Aside type="quran" title="28">
إِنَّ عَذَابَ رَبِّهِمْ غَيْرُ مَأْمُونٍۢ
</Aside>

<Aside type="reference" title="28">
چرا که هیچ کس از عذاب پروردگارش در امان نیست،
</Aside>

---

<Aside type="quran" title="29">
وَٱلَّذِينَ هُمْ لِفُرُوجِهِمْ حَٰفِظُونَ
</Aside>

<Aside type="reference" title="29">
و آنها که دامان خویش را (از بی‌عفّتی) حفظ می‌کنند،
</Aside>

---

<Aside type="quran" title="30">
إِلَّا عَلَىٰٓ أَزْوَٰجِهِمْ أَوْ مَا مَلَكَتْ أَيْمَٰنُهُمْ فَإِنَّهُمْ غَيْرُ مَلُومِينَ
</Aside>

<Aside type="reference" title="30">
جز با همسران و کنیزان (که در حکم همسرند آمیزش ندارند)، چرا که در بهره‌گیری از اینها مورد سرزنش نخواهند بود!
</Aside>

---

<Aside type="quran" title="31">
فَمَنِ ٱبْتَغَىٰ وَرَآءَ ذَٰلِكَ فَأُو۟لَٰٓئِكَ هُمُ ٱلْعَادُونَ
</Aside>

<Aside type="reference" title="31">
و هر کس جز اینها را طلب کند، متجاوز است!
</Aside>

---

<Aside type="quran" title="32">
وَٱلَّذِينَ هُمْ لِأَمَٰنَٰتِهِمْ وَعَهْدِهِمْ رَٰعُونَ
</Aside>

<Aside type="reference" title="32">
و آنها که امانتها و عهد خود را رعایت می‌کنند،
</Aside>

---

<Aside type="quran" title="33">
وَٱلَّذِينَ هُم بِشَهَٰدَٰتِهِمْ قَآئِمُونَ
</Aside>

<Aside type="reference" title="33">

</Aside>

---

<Aside type="quran" title="34">
وَٱلَّذِينَ هُمْ عَلَىٰ صَلَاتِهِمْ يُحَافِظُونَ
</Aside>

<Aside type="reference" title="34">
و آنها که بر نماز مواظبت دارند،
</Aside>

---

<Aside type="quran" title="35">
أُو۟لَٰٓئِكَ فِى جَنَّٰتٍۢ مُّكْرَمُونَ
</Aside>

<Aside type="reference" title="35">
آنان در باغهای بهشتی (پذیرایی و) گرامی داشته می‌شوند.
</Aside>

---

<Aside type="quran" title="36">
فَمَالِ ٱلَّذِينَ كَفَرُوا۟ قِبَلَكَ مُهْطِعِينَ
</Aside>

<Aside type="reference" title="36">
این کافران را چه می‌شود که با سرعت نزد تو می‌آیند...
</Aside>

---

<Aside type="quran" title="37">
عَنِ ٱلْيَمِينِ وَعَنِ ٱلشِّمَالِ عِزِينَ
</Aside>

<Aside type="reference" title="37">
از راست و چپ، گروه گروه (و آرزوی بهشت دارند)!
</Aside>

---

<Aside type="quran" title="38">
أَيَطْمَعُ كُلُّ ٱمْرِئٍۢ مِّنْهُمْ أَن يُدْخَلَ جَنَّةَ نَعِيمٍۢ
</Aside>

<Aside type="reference" title="38">
آیا هر یک از آنها (با این اعمال زشتش) طمع دارد که او را در بهشت پر نعمت الهی وارد کنند؟!
</Aside>

---

<Aside type="quran" title="39">
كَلَّآ ۖ إِنَّا خَلَقْنَٰهُم مِّمَّا يَعْلَمُونَ
</Aside>

<Aside type="reference" title="39">
هرگز چنین نیست؛ ما آنها را از آنچه خودشان می‌دانند آفریده‌ایم!
</Aside>

---

<Aside type="quran" title="40">
فَلَآ أُقْسِمُ بِرَبِّ ٱلْمَشَٰرِقِ وَٱلْمَغَٰرِبِ إِنَّا لَقَٰدِرُونَ
</Aside>

<Aside type="reference" title="40">
سوگند به پروردگار مشرقها و مغربها که ما قادریم...
</Aside>

---

<Aside type="quran" title="41">
عَلَىٰٓ أَن نُّبَدِّلَ خَيْرًۭا مِّنْهُمْ وَمَا نَحْنُ بِمَسْبُوقِينَ
</Aside>

<Aside type="reference" title="41">
که جای آنان را به کسانی بدهیم که از آنها بهترند؛ و ما هرگز مغلوب نخواهیم شد!
</Aside>

---

<Aside type="quran" title="42">
فَذَرْهُمْ يَخُوضُوا۟ وَيَلْعَبُوا۟ حَتَّىٰ يُلَٰقُوا۟ يَوْمَهُمُ ٱلَّذِى يُوعَدُونَ
</Aside>

<Aside type="reference" title="42">
آنان را به حال خود واگذار تا در باطل خود فروروند و بازی کنند تا زمانی که روز موعود خود را ملاقات نمایند!
</Aside>

---

<Aside type="quran" title="43">
يَوْمَ يَخْرُجُونَ مِنَ ٱلْأَجْدَاثِ سِرَاعًۭا كَأَنَّهُمْ إِلَىٰ نُصُبٍۢ يُوفِضُونَ
</Aside>

<Aside type="reference" title="43">

</Aside>

---

<Aside type="quran" title="44">

</Aside>

<Aside type="reference" title="44">

</Aside>
