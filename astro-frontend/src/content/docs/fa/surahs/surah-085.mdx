<Aside type="quran" title="1">
وَٱلسَّمَآءِ ذَاتِ ٱلْبُرُوجِ
</Aside>

<Aside type="reference" title="1">
سوگند به آسمان که دارای برجهای بسیار است،
</Aside>

---

<Aside type="quran" title="2">
وَٱلْيَوْمِ ٱلْمَوْعُودِ
</Aside>

<Aside type="reference" title="2">
و سوگند به آن روز موعود،
</Aside>

---

<Aside type="quran" title="3">
وَشَاهِدٍۢ وَمَشْهُودٍۢ
</Aside>

<Aside type="reference" title="3">
و سوگند به «شاهد» و «مشهود»! [«شاهد»: پیامبر و گواهان اعمال، و «مشهود» : اعمال امّت است‌]
</Aside>

---

<Aside type="quran" title="4">
قُتِلَ أَصْحَٰبُ ٱلْأُخْدُودِ
</Aside>

<Aside type="reference" title="4">
مرگ بر شکنجه‌گران صاحب گودال (آتش)،
</Aside>

---

<Aside type="quran" title="5">
ٱلنَّارِ ذَاتِ ٱلْوَقُودِ
</Aside>

<Aside type="reference" title="5">
آتشی عظیم و شعله‌ور!
</Aside>

---

<Aside type="quran" title="6">
إِذْ هُمْ عَلَيْهَا قُعُودٌۭ
</Aside>

<Aside type="reference" title="6">
هنگامی که در کنار آن نشسته بودند،
</Aside>

---

<Aside type="quran" title="7">
وَهُمْ عَلَىٰ مَا يَفْعَلُونَ بِٱلْمُؤْمِنِينَ شُهُودٌۭ
</Aside>

<Aside type="reference" title="7">
و آنچه را با مؤمنان انجام می‌دادند (با خونسردی و قساوت) تماشا می‌کردند!
</Aside>

---

<Aside type="quran" title="8">
وَمَا نَقَمُوا۟ مِنْهُمْ إِلَّآ أَن يُؤْمِنُوا۟ بِٱللَّهِ ٱلْعَزِيزِ ٱلْحَمِيدِ
</Aside>

<Aside type="reference" title="8">
آنها هیچ ایرادی بر مؤمنان نداشتند جز اینکه به خداوند عزیز و حمید ایمان آورده بودند؛
</Aside>

---

<Aside type="quran" title="9">
ٱلَّذِى لَهُۥ مُلْكُ ٱلسَّمَٰوَٰتِ وَٱلْأَرْضِ ۚ وَٱللَّهُ عَلَىٰ كُلِّ شَىْءٍۢ شَهِيدٌ
</Aside>

<Aside type="reference" title="9">
همان کسی که حکومت آسمانها و زمین از آن اوست و خداوند بر همه چیز گواه است!
</Aside>

---

<Aside type="quran" title="10">
إِنَّ ٱلَّذِينَ فَتَنُوا۟ ٱلْمُؤْمِنِينَ وَٱلْمُؤْمِنَٰتِ ثُمَّ لَمْ يَتُوبُوا۟ فَلَهُمْ عَذَابُ جَهَنَّمَ وَلَهُمْ عَذَابُ ٱلْحَرِيقِ
</Aside>

<Aside type="reference" title="10">
کسانی که مردان و زنان باایمان را شکنجه دادند سپس توبه نکردند، برای آنها عذاب دوزخ و عذاب آتش سوزان است!
</Aside>

---

<Aside type="quran" title="11">
إِنَّ ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّٰلِحَٰتِ لَهُمْ جَنَّٰتٌۭ تَجْرِى مِن تَحْتِهَا ٱلْأَنْهَٰرُ ۚ ذَٰلِكَ ٱلْفَوْزُ ٱلْكَبِيرُ
</Aside>

<Aside type="reference" title="11">
و برای کسانی که ایمان آوردند و اعمال شایسته انجام دادند، باغهایی از بهشت است که نهرها زیر درختانش جاری است؛ و این نجات و پیروزی بزرگ است!
</Aside>

---

<Aside type="quran" title="12">
إِنَّ بَطْشَ رَبِّكَ لَشَدِيدٌ
</Aside>

<Aside type="reference" title="12">
گرفتن قهرآمیز و مجازات پروردگارت به یقین بسیار شدید است!
</Aside>

---

<Aside type="quran" title="13">
إِنَّهُۥ هُوَ يُبْدِئُ وَيُعِيدُ
</Aside>

<Aside type="reference" title="13">
اوست که آفرینش را آغاز می‌کند و بازمی‌گرداند،
</Aside>

---

<Aside type="quran" title="14">
وَهُوَ ٱلْغَفُورُ ٱلْوَدُودُ
</Aside>

<Aside type="reference" title="14">
و او آمرزنده و دوستدار (مؤمنان) است،
</Aside>

---

<Aside type="quran" title="15">
ذُو ٱلْعَرْشِ ٱلْمَجِيدُ
</Aside>

<Aside type="reference" title="15">
صاحب عرش و دارای مجد و عظمت است،
</Aside>

---

<Aside type="quran" title="16">
فَعَّالٌۭ لِّمَا يُرِيدُ
</Aside>

<Aside type="reference" title="16">
و آنچه را می‌خواهد انجام می‌دهد!
</Aside>

---

<Aside type="quran" title="17">
هَلْ أَتَىٰكَ حَدِيثُ ٱلْجُنُودِ
</Aside>

<Aside type="reference" title="17">
آیا داستان لشکرها به تو رسیده است،
</Aside>

---

<Aside type="quran" title="18">
فِرْعَوْنَ وَثَمُودَ
</Aside>

<Aside type="reference" title="18">
لشکریان فرعون و ثمود؟!
</Aside>

---

<Aside type="quran" title="19">
بَلِ ٱلَّذِينَ كَفَرُوا۟ فِى تَكْذِيبٍۢ
</Aside>

<Aside type="reference" title="19">
ولی کافران پیوسته در تکذیب حقند،
</Aside>

---

<Aside type="quran" title="20">
وَٱللَّهُ مِن وَرَآئِهِم مُّحِيطٌۢ

<Aside type="reference" title="20">
