<Aside type="quran" title="1">
وَٱلْعَٰدِيَٰتِ ضَبْحًۭا
</Aside>

<Aside type="reference" title="1">
سوگند به اسبان دونده (مجاهدان) در حالی که نفس‌زنان به پیش می‌رفتند،
</Aside>

---

<Aside type="quran" title="2">
فَٱلْمُورِيَٰتِ قَدْحًۭا
</Aside>

<Aside type="reference" title="2">
و سوگند به افروزندگان جرقه آتش (در برخورد سمهایشان با سنگهای بیابان)،
</Aside>

---

<Aside type="quran" title="3">
فَٱلْمُغِيرَٰتِ صُبْحًۭا
</Aside>

<Aside type="reference" title="3">
و سوگند به هجوم آوران سپیده دم‌
</Aside>

---

<Aside type="quran" title="4">
فَأَثَرْنَ بِهِۦ نَقْعًۭا
</Aside>

<Aside type="reference" title="4">
که گرد و غبار به هر سو پراکندند،
</Aside>

---

<Aside type="quran" title="5">
فَوَسَطْنَ بِهِۦ جَمْعًا
</Aside>

<Aside type="reference" title="5">
و (ناگهان) در میان دشمن ظاهر شدند،
</Aside>

---

<Aside type="quran" title="6">
إِنَّ ٱلْإِنسَٰنَ لِرَبِّهِۦ لَكَنُودٌۭ
</Aside>

<Aside type="reference" title="6">
که انسان در برابر نعمتهای پروردگارش بسیار ناسپاس و بخیل است؛
</Aside>

---

<Aside type="quran" title="7">
وَإِنَّهُۥ عَلَىٰ ذَٰلِكَ لَشَهِيدٌۭ
</Aside>

<Aside type="reference" title="7">
و او خود (نیز) بر این معنی گواه است!
</Aside>

---

<Aside type="quran" title="8">
وَإِنَّهُۥ لِحُبِّ ٱلْخَيْرِ لَشَدِيدٌ
</Aside>

<Aside type="reference" title="8">
و او علاقه شدید به مال دارد!
</Aside>

---

<Aside type="quran" title="9">
۞ أَفَلَا يَعْلَمُ إِذَا بُعْثِرَ مَا فِى ٱلْقُبُورِ
</Aside>

<Aside type="reference" title="9">
آیا نمی‌داند در آن روز که تمام کسانی که در قبرها هستند برانگیخته می‌شوند،
</Aside>

---

<Aside type="quran" title="10">
وَحُصِّلَ مَا فِى ٱلصُّدُورِ
</Aside>

<Aside type="reference" title="10">
و آنچه در درون سینه‌هاست آشکار می‌گردد،
</Aside>

---

<Aside type="quran" title="11">
إِنَّ رَبَّهُم بِهِمْ يَوْمَئِذٍۢ لَّخَبِيرٌۢ
</Aside>

<Aside type="reference" title="11">
در آن روز پروردگارشان از آنها کاملاً باخبر است!
</Aside>
