<Aside type="quran" title="1">
لَآ أُقْسِمُ بِهَٰذَا ٱلْبَلَدِ
</Aside>

<Aside type="reference" title="1">
قسم به این شهر مقدّس [= مکّه‌]،
</Aside>

---

<Aside type="quran" title="2">
وَأَنتَ حِلٌّۢ بِهَٰذَا ٱلْبَلَدِ
</Aside>

<Aside type="reference" title="2">
شهری که تو در آن ساکنی،
</Aside>

---

<Aside type="quran" title="3">
وَوَالِدٍۢ وَمَا وَلَدَ
</Aside>

<Aside type="reference" title="3">
و قسم به پدر و فرزندش [= ابراهیم خلیل و فرزندش اسماعیل ذبیح‌]،
</Aside>

---

<Aside type="quran" title="4">
لَقَدْ خَلَقْنَا ٱلْإِنسَٰنَ فِى كَبَدٍ
</Aside>

<Aside type="reference" title="4">
که ما انسان را در رنج آفریدیم (و زندگی او پر از رنجهاست)!
</Aside>

---

<Aside type="quran" title="5">
أَيَحْسَبُ أَن لَّن يَقْدِرَ عَلَيْهِ أَحَدٌۭ
</Aside>

<Aside type="reference" title="5">
آیا او گمان می‌کند که هیچ کس نمی‌تواند بر او دست یابد؟!
</Aside>

---

<Aside type="quran" title="6">
يَقُولُ أَهْلَكْتُ مَالًۭا لُّبَدًا
</Aside>

<Aside type="reference" title="6">
می‌گوید: «مال زیادی را (در کارهای خیر) نابود کرده‌ام!»
</Aside>

---

<Aside type="quran" title="7">
أَيَحْسَبُ أَن لَّمْ يَرَهُۥٓ أَحَدٌ
</Aside>

<Aside type="reference" title="7">
آیا (انسان) گمان می‌کند هیچ کس او را ندیده (که عمل خیری انجام نداده) است؟!
</Aside>

---

<Aside type="quran" title="8">
أَلَمْ نَجْعَل لَّهُۥ عَيْنَيْنِ
</Aside>

<Aside type="reference" title="8">
آیا برای او دو چشم قرار ندادیم،
</Aside>

---

<Aside type="quran" title="9">
وَلِسَانًۭا وَشَفَتَيْنِ
</Aside>

<Aside type="reference" title="9">
و یک زبان و دو لب؟!
</Aside>

---

<Aside type="quran" title="10">
وَهَدَيْنَٰهُ ٱلنَّجْدَيْنِ
</Aside>

<Aside type="reference" title="10">
و او را به راه خیر و شرّ هدایت کردیم!
</Aside>

---

<Aside type="quran" title="11">
فَلَا ٱقْتَحَمَ ٱلْعَقَبَةَ
</Aside>

<Aside type="reference" title="11">
ولی او از آن گردنه مهمّ نگذشت!
</Aside>

---

<Aside type="quran" title="12">
وَمَآ أَدْرَىٰكَ مَا ٱلْعَقَبَةُ
</Aside>

<Aside type="reference" title="12">
و تو نمی‌دانی آن گردنه چیست!
</Aside>

---

<Aside type="quran" title="13">
فَكُّ رَقَبَةٍ
</Aside>

<Aside type="reference" title="13">
آزادکردن برده‌ای،
</Aside>

---

<Aside type="quran" title="14">
أَوْ إِطْعَٰمٌۭ فِى يَوْمٍۢ ذِى مَسْغَبَةٍۢ
</Aside>

<Aside type="reference" title="14">
یا غذا دادن در روز گرسنگی...
</Aside>

---

<Aside type="quran" title="15">
يَتِيمًۭا ذَا مَقْرَبَةٍ
</Aside>

<Aside type="reference" title="15">
یتیمی از خویشاوندان،
</Aside>

---

<Aside type="quran" title="16">
أَوْ مِسْكِينًۭا ذَا مَتْرَبَةٍۢ
</Aside>

<Aside type="reference" title="16">
یا مستمندی خاک‌نشین را،
</Aside>

---

<Aside type="quran" title="17">
ثُمَّ كَانَ مِنَ ٱلَّذِينَ ءَامَنُوا۟ وَتَوَاصَوْا۟ بِٱلصَّبْرِ وَتَوَاصَوْا۟ بِٱلْمَرْحَمَةِ
</Aside>

<Aside type="reference" title="17">
سپس از کسانی باشد که ایمان آورده و یکدیگر را به شکیبایی و رحمت توصیه می‌کنند!
</Aside>

---

<Aside type="quran" title="18">
أُو۟لَٰٓئِكَ أَصْحَٰبُ ٱلْمَيْمَنَةِ
</Aside>

<Aside type="reference" title="18">
آنها «اصحاب الیمین»اند (که نامه اعمالشان را به دست راستشان می‌دهند)!
</Aside>

---

<Aside type="quran" title="19">
وَٱلَّذِينَ كَفَرُوا۟ بِـَٔايَٰتِنَا هُمْ أَصْحَٰبُ ٱلْمَشْـَٔمَةِ
</Aside>

<Aside type="reference" title="19">
و کسانی که آیات ما را انکار کرده‌اند افرادی شومند (که نامه اعمالشان به دست چپشان داده می‌شود).
</Aside>

---

<Aside type="quran" title="20">
عَلَيْهِمْ نَارٌۭ مُّؤْصَدَةٌۢ
</Aside>

<Aside type="reference" title="20">
بر آنها آتشی است فروبسته (که راه فراری از آن نیست)!
</Aside>
