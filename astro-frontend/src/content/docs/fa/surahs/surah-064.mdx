<Aside type="quran" title="1">
يُسَبِّحُ لِلَّهِ مَا فِى ٱلسَّمَٰوَٰتِ وَمَا فِى ٱلْأَرْضِ ۖ لَهُ ٱلْمُلْكُ وَلَهُ ٱلْحَمْدُ ۖ وَهُوَ عَلَىٰ كُلِّ شَىْءٍۢ قَدِيرٌ
</Aside>

<Aside type="reference" title="1">
آنچه در آسمانها و آنچه در زمین است برای خدا تسبیح می‌گویند؛ مالکیت و حکومت از آن اوست و ستایش از آن او؛ و او بر همه چیز تواناست!
</Aside>

---

<Aside type="quran" title="2">
هُوَ ٱلَّذِى خَلَقَكُمْ فَمِنكُمْ كَافِرٌۭ وَمِنكُم مُّؤْمِنٌۭ ۚ وَٱللَّهُ بِمَا تَعْمَلُونَ بَصِيرٌ
</Aside>

<Aside type="reference" title="2">
او کسی است که شما را آفرید (و به شما آزادی و اختیار داد)؛ گروهی از شما کافرید و گروهی مؤمن؛ و خداوند به آنچه انجام می‌دهید بیناست!
</Aside>

---

<Aside type="quran" title="3">
خَلَقَ ٱلسَّمَٰوَٰتِ وَٱلْأَرْضَ بِٱلْحَقِّ وَصَوَّرَكُمْ فَأَحْسَنَ صُوَرَكُمْ ۖ وَإِلَيْهِ ٱلْمَصِيرُ
</Aside>

<Aside type="reference" title="3">
آسمانها و زمین را بحق آفرید؛ و شما را (در عالم جنین) تصویر کرد، تصویری زیبا و دلپذیر؛ و سرانجام (همه) بسوی اوست.
</Aside>

---

<Aside type="quran" title="4">
يَعْلَمُ مَا فِى ٱلسَّمَٰوَٰتِ وَٱلْأَرْضِ وَيَعْلَمُ مَا تُسِرُّونَ وَمَا تُعْلِنُونَ ۚ وَٱللَّهُ عَلِيمٌۢ بِذَاتِ ٱلصُّدُورِ
</Aside>

<Aside type="reference" title="4">
آنچه را در آسمانها و زمین است می‌داند، و از آنچه پنهان یا آشکار می‌کنید با خبر است؛ و خداوند از آنچه در درون سینه‌هاست آگاه است.
</Aside>

---

<Aside type="quran" title="5">
أَلَمْ يَأْتِكُمْ نَبَؤُا۟ ٱلَّذِينَ كَفَرُوا۟ مِن قَبْلُ فَذَاقُوا۟ وَبَالَ أَمْرِهِمْ وَلَهُمْ عَذَابٌ أَلِيمٌۭ
</Aside>

<Aside type="reference" title="5">
آیا خبر کسانی که پیش از این کافر شدند به شما نرسیده است؟! (آری) آنها طعم کیفر گناهان بزرگ خود را چشیدند؛ و عذاب دردناک برای آنهاست!
</Aside>

---

<Aside type="quran" title="6">
ذَٰلِكَ بِأَنَّهُۥ كَانَت تَّأْتِيهِمْ رُسُلُهُم بِٱلْبَيِّنَٰتِ فَقَالُوٓا۟ أَبَشَرٌۭ يَهْدُونَنَا فَكَفَرُوا۟ وَتَوَلَّوا۟ ۚ وَّٱسْتَغْنَى ٱللَّهُ ۚ وَٱللَّهُ غَنِىٌّ حَمِيدٌۭ
</Aside>

<Aside type="reference" title="6">
این بخاطر آن است که رسولان آنها (پیوسته) با دلایل روشن به سراغشان می‌آمدند، ولی آنها (از روی کبر و غرور) گفتند: «آیا بشرهایی (مثل ما) می‌خواهند ما را هدایت کنند؟!» از این رو کافر شدند و روی برگرداندند؛ و خداوند (از ایمان و طاعتشان) بی نیاز بود، و خدا غنی و شایسته ستایش است!
</Aside>

---

<Aside type="quran" title="7">
زَعَمَ ٱلَّذِينَ كَفَرُوٓا۟ أَن لَّن يُبْعَثُوا۟ ۚ قُلْ بَلَىٰ وَرَبِّى لَتُبْعَثُنَّ ثُمَّ لَتُنَبَّؤُنَّ بِمَا عَمِلْتُمْ ۚ وَذَٰلِكَ عَلَى ٱللَّهِ يَسِيرٌۭ
</Aside>

<Aside type="reference" title="7">
کافران پنداشتند که هرگز برانگیخته نخواهند شد، بگو: «آری به پروردگارم سوگند که همه شما (در قیامت) برانگیخته خواهید شد، سپس آنچه را عمل می‌کردید به شما خبر داده می‌شود، و این برای خداوند آسان است!»
</Aside>

---

<Aside type="quran" title="8">
فَـَٔامِنُوا۟ بِٱللَّهِ وَرَسُولِهِۦ وَٱلنُّورِ ٱلَّذِىٓ أَنزَلْنَا ۚ وَٱللَّهُ بِمَا تَعْمَلُونَ خَبِيرٌۭ
</Aside>

<Aside type="reference" title="8">
حال که چنین است، به خدا و رسول او و نوری که نازل کرده‌ایم ایمان بیاورید؛ و بدانید خدا به آنچه انجام می‌دهید آگاه است!
</Aside>

---

<Aside type="quran" title="9">
يَوْمَ يَجْمَعُكُمْ لِيَوْمِ ٱلْجَمْعِ ۖ ذَٰلِكَ يَوْمُ ٱلتَّغَابُنِ ۗ وَمَن يُؤْمِنۢ بِٱللَّهِ وَيَعْمَلْ صَٰلِحًۭا يُكَفِّرْ عَنْهُ سَيِّـَٔاتِهِۦ وَيُدْخِلْهُ جَنَّٰتٍۢ تَجْرِى مِن تَحْتِهَا ٱلْأَنْهَٰرُ خَٰلِدِينَ فِيهَآ أَبَدًۭا ۚ ذَٰلِكَ ٱلْفَوْزُ ٱلْعَظِيمُ
</Aside>

<Aside type="reference" title="9">
این در زمانی خواهد بود که همه شما را در روز اجتماع [= روز رستاخیز] گردآوری می‌کند؛ آن روز روز تغابن است (روز احساس خسارت و پشیمانی)! و هر کس به خدا ایمان بیاورد و عمل صالح انجام دهد، گناهان او را می‌بخشد و او را در باغهایی از بهشت که نهرها از زیر درختانش جاری است وارد می‌کند، جاودانه در آن می مانند؛ و این پیروزی بزرگ است!
</Aside>

---

<Aside type="quran" title="10">
وَٱلَّذِينَ كَفَرُوا۟ وَكَذَّبُوا۟ بِـَٔايَٰتِنَآ أُو۟لَٰٓئِكَ أَصْحَٰبُ ٱلنَّارِ خَٰلِدِينَ فِيهَا ۖ وَبِئْسَ ٱلْمَصِيرُ
</Aside>

<Aside type="reference" title="10">
اما کسانی که کافر شدند و آیات ما را تکذیب کردند اصحاب دوزخند، جاودانه در آن می‌مانند، و (سرانجام آنها) سرانجام بدی است!
</Aside>

---

<Aside type="quran" title="11">
مَآ أَصَابَ مِن مُّصِيبَةٍ إِلَّا بِإِذْنِ ٱللَّهِ ۗ وَمَن يُؤْمِنۢ بِٱللَّهِ يَهْدِ قَلْبَهُۥ ۚ وَٱللَّهُ بِكُلِّ شَىْءٍ عَلِيمٌۭ
</Aside>

<Aside type="reference" title="11">
هیچ مصیبتی رخ نمی‌دهد مگر به اذن خدا! و هر کس به خدا ایمان آورد، خداوند قلبش را هدایت می‌کند؛ و خدا به هر چیز داناست!
</Aside>

---

<Aside type="quran" title="12">
وَأَطِيعُوا۟ ٱللَّهَ وَأَطِيعُوا۟ ٱلرَّسُولَ ۚ فَإِن تَوَلَّيْتُمْ فَإِنَّمَا عَلَىٰ رَسُولِنَا ٱلْبَلَٰغُ ٱلْمُبِينُ
</Aside>

<Aside type="reference" title="12">
اطاعت کنید خدا را، و اطاعت کنید پیامبر را؛ و اگر روی‌گردان شوید، رسول ما جز ابلاغ آشکار وظیفه‌ای ندارد!
</Aside>

---

<Aside type="quran" title="13">
ٱللَّهُ لَآ إِلَٰهَ إِلَّا هُوَ ۚ وَعَلَى ٱللَّهِ فَلْيَتَوَكَّلِ ٱلْمُؤْمِنُونَ
</Aside>

<Aside type="reference" title="13">
خداوند کسی است که هیچ معبودی جز او نیست، و مؤمنان باید فقط بر او توکل کنند!
</Aside>

---

<Aside type="quran" title="14">
يَٰٓأَيُّهَا ٱلَّذِينَ ءَامَنُوٓا۟ إِنَّ مِنْ أَزْوَٰجِكُمْ وَأَوْلَٰدِكُمْ عَدُوًّۭا لَّكُمْ فَٱحْذَرُوهُمْ ۚ وَإِن تَعْفُوا۟ وَتَصْفَحُوا۟ وَتَغْفِرُوا۟ فَإِنَّ ٱللَّهَ غَفُورٌۭ رَّحِيمٌ
</Aside>

<Aside type="reference" title="14">
ای کسانی که ایمان آورده‌اید! بعضی از همسران و فرزندانتان دشمنان شما هستند، از آنها برحذر باشید؛ و اگر عفو کنید و چشم بپوشید و ببخشید، (خدا شما را می‌بخشد)؛ چرا که خداوند بخشنده و مهربان است!
</Aside>

---

<Aside type="quran" title="15">
إِنَّمَآ أَمْوَٰلُكُمْ وَأَوْلَٰدُكُمْ فِتْنَةٌۭ ۚ وَٱللَّهُ عِندَهُۥٓ أَجْرٌ عَظِيمٌۭ
</Aside>

<Aside type="reference" title="15">
اموال و فرزندانتان فقط وسیله آزمایش شما هستند؛ و خداست که پاداش عظیم نزد اوست!
</Aside>

---

<Aside type="quran" title="16">
فَٱتَّقُوا۟ ٱللَّهَ مَا ٱسْتَطَعْتُمْ وَٱسْمَعُوا۟ وَأَطِيعُوا۟ وَأَنفِقُوا۟ خَيْرًۭا لِّأَنفُسِكُمْ ۗ وَمَن يُوقَ شُحَّ نَفْسِهِۦ فَأُو۟لَٰٓئِكَ هُمُ ٱلْمُفْلِحُونَ
</Aside>

<Aside type="reference" title="16">
پس تا می‌توانید تقوای الهی پیشه کنید و گوش دهید و اطاعت نمایید و انفاق کنید که برای شما بهتر است؛ و کسانی که از بخل و حرص خویشتن مصون بمانند رستگارانند!
</Aside>

---

<Aside type="quran" title="17">
إِن تُقْرِضُوا۟ ٱللَّهَ قَرْضًا حَسَنًۭا يُضَٰعِفْهُ لَكُمْ وَيَغْفِرْ لَكُمْ ۚ وَٱللَّهُ شَكُورٌ حَلِيمٌ
</Aside>

<Aside type="reference" title="17">
اگر به خدا قرض‌الحسنه دهید، آن را برای شما مضاعف می‌سازد و شما را می‌بخشد؛ و خداوند شکرکننده و بردبار است!
</Aside>

---

<Aside type="quran" title="18">
عَٰلِمُ ٱلْغَيْبِ وَٱلشَّهَٰدَةِ ٱلْعَزِيزُ ٱلْحَكِيمُ
</Aside>

<Aside type="reference" title="18">
او دانای پنهان و آشکار است؛ و او عزیز و حکیم است!
</Aside>
