<Aside type="quran" title="3">
لَعَلَّكَ بَٰخِعٌۭ نَّفْسَكَ أَلَّا يَكُونُوا۟ مُؤْمِنِينَ
</Aside>

<Aside type="reference" title="3">
گویی می‌خواهی جان خود را از شدّت اندوه از دست دهی بخاطر اینکه آنها ایمان نمی‌آورند!
</Aside>

---

<Aside type="quran" title="4">
إِن نَّشَأْ نُنَزِّلْ عَلَيْهِم مِّنَ ٱلسَّمَآءِ ءَايَةًۭ فَظَلَّتْ أَعْنَٰقُهُمْ لَهَا خَٰضِعِينَ
</Aside>

<Aside type="reference" title="4">
اگر ما اراده کنیم، از آسمان بر آنان آیه‌ای نازل می‌کنیم که گردنهایشان در برابر آن خاضع گردد!
</Aside>

---

<Aside type="quran" title="5">
وَمَا يَأْتِيهِم مِّن ذِكْرٍۢ مِّنَ ٱلرَّحْمَٰنِ مُحْدَثٍ إِلَّا كَانُوا۟ عَنْهُ مُعْرِضِينَ
</Aside>

<Aside type="reference" title="5">
و هیچ ذکر تازه‌ای از سوی خداوند مهربان برای آنها نمی‌آید مگر اینکه از آن روی‌گردان می‌شوند!
</Aside>

---

<Aside type="quran" title="6">
فَقَدْ كَذَّبُوا۟ فَسَيَأْتِيهِمْ أَنۢبَٰٓؤُا۟ مَا كَانُوا۟ بِهِۦ يَسْتَهْزِءُونَ
</Aside>

<Aside type="reference" title="6">
آنان تکذیب کردند؛ امّا بزودی اخبار (کیفر) آنچه را استهزا می‌کردند به آنان می‌رسد!
</Aside>

---

<Aside type="quran" title="7">
أَوَلَمْ يَرَوْا۟ إِلَى ٱلْأَرْضِ كَمْ أَنۢبَتْنَا فِيهَا مِن كُلِّ زَوْجٍۢ كَرِيمٍ
</Aside>

<Aside type="reference" title="7">
آیا آنان به زمین نگاه نکردند که چقدر از انواع گیاهان پرارزش در آن رویاندیم؟!
</Aside>

---

<Aside type="quran" title="8">
إِنَّ فِى ذَٰلِكَ لَءَايَةًۭ ۖ وَمَا كَانَ أَكْثَرُهُم مُّؤْمِنِينَ
</Aside>

<Aside type="reference" title="8">
در این، نشانه روشنی است (بر وجود خدا)؛ ولی بیشترشان هرگز مؤمن نبوده‌اند!
</Aside>

---

<Aside type="quran" title="9">
وَإِنَّ رَبَّكَ لَهُوَ ٱلْعَزِيزُ ٱلرَّحِيمُ
</Aside>

<Aside type="reference" title="9">
و پروردگار تو عزیز و رحیم است!
</Aside>

---

<Aside type="quran" title="10">
وَإِذْ نَادَىٰ رَبُّكَ مُوسَىٰٓ أَنِ ٱئْتِ ٱلْقَوْمَ ٱلظَّٰلِمِينَ
</Aside>

<Aside type="reference" title="10">
(به خاطر بیاور) هنگامی را که پروردگارت موسی را ندا داد که به سراغ قوم ستمگر برو...
</Aside>

---

<Aside type="quran" title="11">
قَوْمَ فِرْعَوْنَ ۚ أَلَا يَتَّقُونَ
</Aside>

<Aside type="reference" title="11">
قوم فرعون، آیا آنان (از مخالفت فرمان پروردگار) پرهیز نمی‌کنند؟!
</Aside>

---

<Aside type="quran" title="12">
قَالَ رَبِّ إِنِّىٓ أَخَافُ أَن يُكَذِّبُونِ
</Aside>

<Aside type="reference" title="12">
(موسی) عرض کرد: «پروردگارا! از آن بیم دارم که مرا تکذیب کنند،
</Aside>

---

<Aside type="quran" title="13">
وَيَضِيقُ صَدْرِى وَلَا يَنطَلِقُ لِسَانِى فَأَرْسِلْ إِلَىٰ هَٰرُونَ
</Aside>

<Aside type="reference" title="13">
و سینه‌ام تنگ شود، و زبانم بقدر کافی گویا نیست؛ (برادرم) هارون را نیز رسالت ده (تا مرا یاری کند)!
</Aside>

---

<Aside type="quran" title="14">
وَلَهُمْ عَلَىَّ ذَنۢبٌۭ فَأَخَافُ أَن يَقْتُلُونِ
</Aside>

<Aside type="reference" title="14">
و آنان (به اعتقاد خودشان) بر گردن من گناهی دارند؛ می‌ترسم مرا بکشند (و این رسالت به پایان نرسد)!
</Aside>

---

<Aside type="quran" title="15">
قَالَ كَلَّا ۖ فَٱذْهَبَا بِـَٔايَٰتِنَآ ۖ إِنَّا مَعَكُم مُّسْتَمِعُونَ
</Aside>

<Aside type="reference" title="15">
فرمود: «چنین نیست، (آنان کاری نمی‌توانند انجام دهند)! شما هر دو با آیات ما (برای هدایتشان) بروید؛ ما با شما هستیم و (سخنانتان را) می‌شنویم!
</Aside>

---

<Aside type="quran" title="16">
فَأْتِيَا فِرْعَوْنَ فَقُولَآ إِنَّا رَسُولُ رَبِّ ٱلْعَٰلَمِينَ
</Aside>

<Aside type="reference" title="16">
به سراغ فرعون بروید و بگویید: ما فرستاده پروردگار جهانیان هستیم؛
</Aside>

---

<Aside type="quran" title="17">
أَنْ أَرْسِلْ مَعَنَا بَنِىٓ إِسْرَٰٓءِيلَ
</Aside>

<Aside type="reference" title="17">
بنی اسرائیل را با ما بفرست!» (آنها به سراغ فرعون آمدند)؛
</Aside>

---

<Aside type="quran" title="18">
قَالَ أَلَمْ نُرَبِّكَ فِينَا وَلِيدًۭا وَلَبِثْتَ فِينَا مِنْ عُمُرِكَ سِنِينَ
</Aside>

<Aside type="reference" title="18">
(فرعون) گفت: «آیا ما تو را در کودکی در میان خود پرورش ندادیم، و سالهایی از زندگیت را در میان ما نبودی؟!
</Aside>

---

<Aside type="quran" title="19">
وَفَعَلْتَ فَعْلَتَكَ ٱلَّتِى فَعَلْتَ وَأَنتَ مِنَ ٱلْكَٰفِرِينَ
</Aside>

<Aside type="reference" title="19">
و سرانجام، آن کارت را (که نمی‌بایست انجام دهی) انجام دادی (و یک نفر از ما را کشتی)، و تو از ناسپاسانی!»
</Aside>

---

<Aside type="quran" title="20">
قَالَ فَعَلْتُهَآ إِذًۭا وَأَنَا۠ مِنَ ٱلضَّآلِّينَ
</Aside>

<Aside type="reference" title="20">
(موسی) گفت: «من آن کار را انجام دادم در حالی که از بی‌خبران بودم!
</Aside>

---

<Aside type="quran" title="21">
فَفَرَرْتُ مِنكُمْ لَمَّا خِفْتُكُمْ فَوَهَبَ لِى رَبِّى حُكْمًۭا وَجَعَلَنِى مِنَ ٱلْمُرْسَلِينَ
</Aside>

<Aside type="reference" title="21">
پس هنگامی که از شما ترسیدم فرار کردم؛ و پروردگارم به من حکمت و دانش بخشید، و مرا از پیامبران قرار داد!
</Aside>

---

<Aside type="quran" title="22">
وَتِلْكَ نِعْمَةٌۭ تَمُنُّهَا عَلَىَّ أَنْ عَبَّدتَّ بَنِىٓ إِسْرَٰٓءِيلَ
</Aside>

<Aside type="reference" title="22">
آیا این منتّی است که تو بر من می‌گذاری که بنی اسرائیل را برده خود ساخته‌ای؟!»
</Aside>

---

<Aside type="quran" title="23">
قَالَ فِرْعَوْنُ وَمَا رَبُّ ٱلْعَٰلَمِينَ
</Aside>

<Aside type="reference" title="23">

</Aside>

---

<Aside type="quran" title="24">
قَالَ رَبُّ ٱلسَّمَٰوَٰتِ وَٱلْأَرْضِ وَمَا بَيْنَهُمَآ ۖ إِن كُنتُم مُّوقِنِينَ
</Aside>

<Aside type="reference" title="24">
(موسی) گفت: «پروردگار آسمانها و زمین و آنچه میان آن دو است، اگر اهل یقین هستید!».
</Aside>

---

<Aside type="quran" title="25">
قَالَ لِمَنْ حَوْلَهُۥٓ أَلَا تَسْتَمِعُونَ
</Aside>

<Aside type="reference" title="25">

</Aside>

---

<Aside type="quran" title="26">
قَالَ رَبُّكُمْ وَرَبُّ ءَابَآئِكُمُ ٱلْأَوَّلِينَ
</Aside>

<Aside type="reference" title="26">
(موسی) گفت: «او پروردگار شما و پروردگار نیاکان شماست!»
</Aside>

---

<Aside type="quran" title="27">
قَالَ إِنَّ رَسُولَكُمُ ٱلَّذِىٓ أُرْسِلَ إِلَيْكُمْ لَمَجْنُونٌۭ
</Aside>

<Aside type="reference" title="27">

</Aside>

---

<Aside type="quran" title="28">
قَالَ رَبُّ ٱلْمَشْرِقِ وَٱلْمَغْرِبِ وَمَا بَيْنَهُمَآ ۖ إِن كُنتُمْ تَعْقِلُونَ
</Aside>

<Aside type="reference" title="28">
(موسی) گفت: «او پروردگار مشرق و مغرب و آنچه میان آن دو است می‌باشد، اگر شما عقل و اندیشه خود را به کار می‌گرفتید!»
</Aside>

---

<Aside type="quran" title="29">
قَالَ لَئِنِ ٱتَّخَذْتَ إِلَٰهًا غَيْرِى لَأَجْعَلَنَّكَ مِنَ ٱلْمَسْجُونِينَ
</Aside>

<Aside type="reference" title="29">
(فرعون خشمگین شد و) گفت: «اگر معبودی غیر از من برگزینی، تو را از زندانیان قرار خواهم داد!»
</Aside>

---

<Aside type="quran" title="30">

</Aside>

<Aside type="reference" title="30">

</Aside>

---

<Aside type="quran" title="31">

</Aside>

<Aside type="reference" title="31">

<Aside type="quran" title="32">

</Aside>

<Aside type="reference" title="32">

<Aside type="quran" title="33">

<Aside type="reference" title="33">

<Aside type="quran" title="34">

<Aside type="quran" title="35">
