<Aside type="quran" title="1">
عَمَّ يَتَسَآءَلُونَ
</Aside>

<Aside type="reference" title="1">
آنها از چه چیز از یکدیگر سؤال می‌کنند؟!
</Aside>

---

<Aside type="quran" title="2">
عَنِ ٱلنَّبَإِ ٱلْعَظِيمِ
</Aside>

<Aside type="reference" title="2">
از خبر بزرگ و پراهمیّت (رستاخیز)!
</Aside>

---

<Aside type="quran" title="3">
ٱلَّذِى هُمْ فِيهِ مُخْتَلِفُونَ
</Aside>

<Aside type="reference" title="3">
همان خبری که پیوسته در آن اختلاف دارند!
</Aside>

---

<Aside type="quran" title="4">
كَلَّا سَيَعْلَمُونَ
</Aside>

<Aside type="reference" title="4">
چنین نیست که آنها فکر می‌کنند، و بزودی می‌فهمند!
</Aside>

---

<Aside type="quran" title="5">
ثُمَّ كَلَّا سَيَعْلَمُونَ
</Aside>

<Aside type="reference" title="5">
باز هم چنین نیست که آنها می‌پندارند، و بزودی می‌فهمند (که قیامت حق است)!
</Aside>

---

<Aside type="quran" title="6">
أَلَمْ نَجْعَلِ ٱلْأَرْضَ مِهَٰدًۭا
</Aside>

<Aside type="reference" title="6">
آیا زمین را محل آرامش (شما) قرار ندادیم؟!
</Aside>

---

<Aside type="quran" title="7">
وَٱلْجِبَالَ أَوْتَادًۭا
</Aside>

<Aside type="reference" title="7">
و کوه‌ها را میخهای زمین؟!
</Aside>

---

<Aside type="quran" title="8">
وَخَلَقْنَٰكُمْ أَزْوَٰجًۭا
</Aside>

<Aside type="reference" title="8">
و شما را بصورت زوجها آفریدیم!
</Aside>

---

<Aside type="quran" title="9">
وَجَعَلْنَا نَوْمَكُمْ سُبَاتًۭا
</Aside>

<Aside type="reference" title="9">
و خواب شما را مایه آرامشتان قرار دادیم،
</Aside>

---

<Aside type="quran" title="10">
وَجَعَلْنَا ٱلَّيْلَ لِبَاسًۭا
</Aside>

<Aside type="reference" title="10">
و شب را پوششی (برای شما)،
</Aside>

---

<Aside type="quran" title="11">
وَجَعَلْنَا ٱلنَّهَارَ مَعَاشًۭا
</Aside>

<Aside type="reference" title="11">
و روز را وسیله‌ای برای زندگی و معاش!
</Aside>

---

<Aside type="quran" title="12">
وَبَنَيْنَا فَوْقَكُمْ سَبْعًۭا شِدَادًۭا
</Aside>

<Aside type="reference" title="12">
و بر فراز شما هفت (آسمان) محکم بنا کردیم!
</Aside>

---

<Aside type="quran" title="13">
وَجَعَلْنَا سِرَاجًۭا وَهَّاجًۭا
</Aside>

<Aside type="reference" title="13">
و چراغی روشن و حرارت‌بخش آفریدیم!
</Aside>

---

<Aside type="quran" title="14">
وَأَنزَلْنَا مِنَ ٱلْمُعْصِرَٰتِ مَآءًۭ ثَجَّاجًۭا
</Aside>

<Aside type="reference" title="14">
و از ابرهای باران‌زا آبی فراوان نازل کردیم،
</Aside>

---

<Aside type="quran" title="15">
لِّنُخْرِجَ بِهِۦ حَبًّۭا وَنَبَاتًۭا
</Aside>

<Aside type="reference" title="15">
تا بوسیله آن دانه و گیاه بسیار برویانیم،
</Aside>

---

<Aside type="quran" title="16">
وَجَنَّٰتٍ أَلْفَافًا
</Aside>

<Aside type="reference" title="16">
و باغهایی پردرخت!
</Aside>

---

<Aside type="quran" title="17">
إِنَّ يَوْمَ ٱلْفَصْلِ كَانَ مِيقَٰتًۭا
</Aside>

<Aside type="reference" title="17">
(آری) روز جدایی، میعاد همگان است!
</Aside>

---

<Aside type="quran" title="18">
يَوْمَ يُنفَخُ فِى ٱلصُّورِ فَتَأْتُونَ أَفْوَاجًۭا
</Aside>

<Aside type="reference" title="18">
روزی که در «صور» دمیده می‌شود و شما فوج فوج (به محشر) می‌آیید!
</Aside>

---

<Aside type="quran" title="19">
وَفُتِحَتِ ٱلسَّمَآءُ فَكَانَتْ أَبْوَٰبًۭا
</Aside>

<Aside type="reference" title="19">
و آسمان گشوده می‌شود و بصورت درهای متعددی درمی‌آید!
</Aside>

---

<Aside type="quran" title="20">
وَسُيِّرَتِ ٱلْجِبَالُ فَكَانَتْ سَرَابًا
</Aside>

<Aside type="reference" title="20">
و کوه‌ها به حرکت درمی‌آید و بصورت سرابی می‌شود!
</Aside>

---

<Aside type="quran" title="21">
إِنَّ جَهَنَّمَ كَانَتْ مِرْصَادًۭا
</Aside>

<Aside type="reference" title="21">
مسلّماً (در آن روز) جهنّم کمینگاهی است بزرگ،
</Aside>

---

<Aside type="quran" title="22">
لِّلطَّٰغِينَ مَـَٔابًۭا
</Aside>

<Aside type="reference" title="22">
و محل بازگشتی برای طغیانگران!
</Aside>

---

<Aside type="quran" title="23">
لَّٰبِثِينَ فِيهَآ أَحْقَابًۭا
</Aside>

<Aside type="reference" title="23">
مدّتهای طولانی در آن می‌مانند!
</Aside>

---

<Aside type="quran" title="24">
لَّا يَذُوقُونَ فِيهَا بَرْدًۭا وَلَا شَرَابًا
</Aside>

<Aside type="reference" title="24">
در آنجا نه چیز خنکی می‌چشند و نه نوشیدنی گوارایی،
</Aside>

---

<Aside type="quran" title="25">
إِلَّا حَمِيمًۭا وَغَسَّاقًۭا
</Aside>

<Aside type="reference" title="25">
جز آبی سوزان و مایعی از چرک و خون!
</Aside>

---

<Aside type="quran" title="26">
جَزَآءًۭ وِفَاقًا
</Aside>

<Aside type="reference" title="26">
این مجازاتی است موافق و مناسب (اعمالشان)!
</Aside>

---

<Aside type="quran" title="27">
إِنَّهُمْ كَانُوا۟ لَا يَرْجُونَ حِسَابًۭا
</Aside>

<Aside type="reference" title="27">
چرا که آنها هیچ امیدی به حساب نداشتند،
</Aside>

---

<Aside type="quran" title="28">
وَكَذَّبُوا۟ بِـَٔايَٰتِنَا كِذَّابًۭا
</Aside>

<Aside type="reference" title="28">
و آیات ما را بکلی تکذیب کردند!
</Aside>

---

<Aside type="quran" title="29">
وَكُلَّ شَىْءٍ أَحْصَيْنَٰهُ كِتَٰبًۭا
</Aside>

<Aside type="reference" title="29">
و ما همه چیز را شمارش و ثبت کرده‌ایم!
</Aside>

---

<Aside type="quran" title="30">
فَذُوقُوا۟ فَلَن نَّزِيدَكُمْ إِلَّا عَذَابًا
</Aside>

<Aside type="reference" title="30">
پس بچشید که چیزی جز عذاب بر شما نمی‌افزاییم!
</Aside>

---

<Aside type="quran" title="31">
إِنَّ لِلْمُتَّقِينَ مَفَازًا
</Aside>

<Aside type="reference" title="31">
مسلّماً برای پرهیزگاران نجات و پیروزی بزرگی است:
</Aside>

---

<Aside type="quran" title="32">
حَدَآئِقَ وَأَعْنَٰبًۭا
</Aside>

<Aside type="reference" title="32">
باغهایی سرسبز، و انواع انگورها،
</Aside>

---

<Aside type="quran" title="33">
وَكَوَاعِبَ أَتْرَابًۭا
</Aside>

<Aside type="reference" title="33">
و حوریانی بسیار جوان و هم‌سن و سال،
</Aside>

---

<Aside type="quran" title="34">
وَكَأْسًۭا دِهَاقًۭا
</Aside>

<Aside type="reference" title="34">
و جامهایی لبریز و پیاپی (از شراب طهور)!
</Aside>

---

<Aside type="quran" title="35">
لَّا يَسْمَعُونَ فِيهَا لَغْوًۭا وَلَا كِذَّٰبًۭا
</Aside>

<Aside type="reference" title="35">
در آنجا نه سخن لغو و بیهوده‌ای می‌شنوند و نه دروغی!
</Aside>

---

<Aside type="quran" title="36">
جَزَآءًۭ مِّن رَّبِّكَ عَطَآءً حِسَابًۭا
</Aside>

<Aside type="reference" title="36">
این کیفری است از سوی پروردگارت و عطیه‌ای است کافی!
</Aside>

---

<Aside type="quran" title="37">
رَّبِّ ٱلسَّمَٰوَٰتِ وَٱلْأَرْضِ وَمَا بَيْنَهُمَا ٱلرَّحْمَٰنِ ۖ لَا يَمْلِكُونَ مِنْهُ خِطَابًۭا
</Aside>

<Aside type="reference" title="37">
همان پروردگار آسمانها و زمین و آنچه در میان آن دو است، پروردگار رحمان! و (در آن روز) هیچ کس حق ندارد بی اجازه او سخنی بگوید (یا شفاعتی کند)!
</Aside>

---

<Aside type="quran" title="38">
يَوْمَ يَقُومُ ٱلرُّوحُ وَٱلْمَلَٰٓئِكَةُ صَفًّۭا ۖ لَّا يَتَكَلَّمُونَ إِلَّا مَنْ أَذِنَ لَهُ ٱلرَّحْمَٰنُ وَقَالَ صَوَابًۭا
</Aside>

<Aside type="reference" title="38">
روزی که «روح» و «ملائکه» در یک صف می‌ایستند و هیچ یک، جز به اذن خداوند رحمان، سخن نمی‌گویند، و (آنگاه که می‌گویند) درست می‌گویند!
</Aside>

---

<Aside type="quran" title="39">
ذَٰلِكَ ٱلْيَوْمُ ٱلْحَقُّ ۖ فَمَن شَآءَ ٱتَّخَذَ إِلَىٰ رَبِّهِۦ مَـَٔابًا
</Aside>

<Aside type="reference" title="39">
آن روز حق است؛ هر کس بخواهد راهی به سوی پروردگارش برمی‌گزیند!
</Aside>

---

<Aside type="quran" title="40">
إِنَّآ أَنذَرْنَٰكُمْ عَذَابًۭا قَرِيبًۭا يَوْمَ يَنظُرُ ٱلْمَرْءُ مَا قَدَّمَتْ يَدَاهُ وَيَقُولُ ٱلْكَافِرُ يَٰلَيْتَنِى كُنتُ تُرَٰبًۢا
</Aside>

<Aside type="reference" title="40">
و ما شما را از عذاب نزدیکی بیم دادیم! این عذاب در روزی خواهد بود که انسان آنچه را از قبل با دستهای خود فرستاده می‌بیند، و کافر می‌گوید: «ای کاش خاک بودم (و گرفتار عذاب نمی‌شدم)!»
</Aside>
