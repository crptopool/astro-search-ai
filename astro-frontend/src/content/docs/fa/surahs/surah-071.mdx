<Aside type="quran" title="1">
إِنَّآ أَرْسَلْنَا نُوحًا إِلَىٰ قَوْمِهِۦٓ أَنْ أَنذِرْ قَوْمَكَ مِن قَبْلِ أَن يَأْتِيَهُمْ عَذَابٌ أَلِيمٌۭ
</Aside>

<Aside type="reference" title="1">
ما نوح را به سوی قومش فرستادیم و گفتیم: «قوم خود را انذار کن پیش از آنکه عذاب دردناک به سراغشان آید!»
</Aside>

---

<Aside type="quran" title="2">
قَالَ يَٰقَوْمِ إِنِّى لَكُمْ نَذِيرٌۭ مُّبِينٌ
</Aside>

<Aside type="reference" title="2">
گفت: «ای قوم! من برای شما بیم‌دهنده آشکاری هستم،
</Aside>

---

<Aside type="quran" title="3">
أَنِ ٱعْبُدُوا۟ ٱللَّهَ وَٱتَّقُوهُ وَأَطِيعُونِ
</Aside>

<Aside type="reference" title="3">
که خدا را پرستش کنید و از مخالفت او بپرهیزید و مرا اطاعت نمایید!
</Aside>

---

<Aside type="quran" title="4">
يَغْفِرْ لَكُم مِّن ذُنُوبِكُمْ وَيُؤَخِّرْكُمْ إِلَىٰٓ أَجَلٍۢ مُّسَمًّى ۚ إِنَّ أَجَلَ ٱللَّهِ إِذَا جَآءَ لَا يُؤَخَّرُ ۖ لَوْ كُنتُمْ تَعْلَمُونَ
</Aside>

<Aside type="reference" title="4">
اگر چنین کنید، خدا گناهانتان را می‌آمرزد و تا زمان معیّنی شما را عمر می‌دهد؛ زیرا هنگامی که اجل الهی فرا رسد، تأخیری نخواهد داشت اگر می‌دانستید!»
</Aside>

---

<Aside type="quran" title="5">
قَالَ رَبِّ إِنِّى دَعَوْتُ قَوْمِى لَيْلًۭا وَنَهَارًۭا
</Aside>

<Aside type="reference" title="5">
(نوح) گفت: «پروردگارا! من قوم خود را شب و روز (بسوی تو) دعوت کردم،
</Aside>

---

<Aside type="quran" title="6">
فَلَمْ يَزِدْهُمْ دُعَآءِىٓ إِلَّا فِرَارًۭا
</Aside>

<Aside type="reference" title="6">
امّا دعوت من چیزی جز فرار از حقّ بر آنان نیفزود!
</Aside>

---

<Aside type="quran" title="7">
وَإِنِّى كُلَّمَا دَعَوْتُهُمْ لِتَغْفِرَ لَهُمْ جَعَلُوٓا۟ أَصَٰبِعَهُمْ فِىٓ ءَاذَانِهِمْ وَٱسْتَغْشَوْا۟ ثِيَابَهُمْ وَأَصَرُّوا۟ وَٱسْتَكْبَرُوا۟ ٱسْتِكْبَارًۭا
</Aside>

<Aside type="reference" title="7">
و من هر زمان آنها را دعوت کردم که (ایمان بیاورند و) تو آنها را بیامرزی، انگشتان خویش را در گوشهایشان قرار داده و لباسهایشان را بر خود پیچیدند، و در مخالفت اصرار ورزیدند و به شدّت استکبار کردند!
</Aside>

---

<Aside type="quran" title="8">
ثُمَّ إِنِّى دَعَوْتُهُمْ جِهَارًۭا
</Aside>

<Aside type="reference" title="8">
سپس من آنها را با صدای بلند (به اطاعت فرمان تو) دعوت کردم،
</Aside>

---

<Aside type="quran" title="9">
ثُمَّ إِنِّىٓ أَعْلَنتُ لَهُمْ وَأَسْرَرْتُ لَهُمْ إِسْرَارًۭا
</Aside>

<Aside type="reference" title="9">
سپس آشکارا و نهان (حقیقت توحید و ایمان را) برای آنان بیان داشتم!
</Aside>

---

<Aside type="quran" title="10">
فَقُلْتُ ٱسْتَغْفِرُوا۟ رَبَّكُمْ إِنَّهُۥ كَانَ غَفَّارًۭا
</Aside>

<Aside type="reference" title="10">
به آنها گفتم: «از پروردگار خویش آمرزش بطلبید که او بسیار آمرزنده است...
</Aside>

---

<Aside type="quran" title="11">
يُرْسِلِ ٱلسَّمَآءَ عَلَيْكُم مِّدْرَارًۭا
</Aside>

<Aside type="reference" title="11">
تا بارانهای پربرکت آسمان را پی در پی بر شما فرستد،
</Aside>

---

<Aside type="quran" title="12">
وَيُمْدِدْكُم بِأَمْوَٰلٍۢ وَبَنِينَ وَيَجْعَل لَّكُمْ جَنَّٰتٍۢ وَيَجْعَل لَّكُمْ أَنْهَٰرًۭا
</Aside>

<Aside type="reference" title="12">
و شما را با اموال و فرزندان فراوان کمک کند و باغهای سرسبز و نهرهای جاری در اختیارتان قرار دهد!
</Aside>

---

<Aside type="quran" title="13">
مَّا لَكُمْ لَا تَرْجُونَ لِلَّهِ وَقَارًۭا
</Aside>

<Aside type="reference" title="13">
چرا شما برای خدا عظمت قائل نیستید؟!
</Aside>

---

<Aside type="quran" title="14">
وَقَدْ خَلَقَكُمْ أَطْوَارًا
</Aside>

<Aside type="reference" title="14">
در حالی که شما را در مراحل مختلف آفرید (تا از نطفه به انسان کامل رسیدید)!
</Aside>

---

<Aside type="quran" title="15">
أَلَمْ تَرَوْا۟ كَيْفَ خَلَقَ ٱللَّهُ سَبْعَ سَمَٰوَٰتٍۢ طِبَاقًۭا
</Aside>

<Aside type="reference" title="15">
آیا نمی‌دانید چگونه خداوند هفت آسمان را یکی بالای دیگری آفریده است،
</Aside>

---

<Aside type="quran" title="16">
وَجَعَلَ ٱلْقَمَرَ فِيهِنَّ نُورًۭا وَجَعَلَ ٱلشَّمْسَ سِرَاجًۭا
</Aside>

<Aside type="reference" title="16">
و ماه را در میان آسمانها مایه روشنایی، و خورشید را چراغ فروزانی قرار داده است؟!
</Aside>

---

<Aside type="quran" title="17">
وَٱللَّهُ أَنۢبَتَكُم مِّنَ ٱلْأَرْضِ نَبَاتًۭا
</Aside>

<Aside type="reference" title="17">
و خداوند شما را همچون گیاهی از زمین رویانید،
</Aside>

---

<Aside type="quran" title="18">
ثُمَّ يُعِيدُكُمْ فِيهَا وَيُخْرِجُكُمْ إِخْرَاجًۭا
</Aside>

<Aside type="reference" title="18">
سپس شما را به همان زمین بازمی‌گرداند، و بار دیگر شما را خارج می‌سازد!
</Aside>

---

<Aside type="quran" title="19">
وَٱللَّهُ جَعَلَ لَكُمُ ٱلْأَرْضَ بِسَاطًۭا
</Aside>

<Aside type="reference" title="19">
و خداوند زمین را برای شما فرش گسترده‌ای قرار داد...
</Aside>

---

<Aside type="quran" title="20">
لِّتَسْلُكُوا۟ مِنْهَا سُبُلًۭا فِجَاجًۭا
</Aside>

<Aside type="reference" title="20">
تا از راههای وسیع و دره‌های آن بگذرید (و به هر جا می‌خواهید بروید)!»
</Aside>

---

<Aside type="quran" title="21">
قَالَ نُوحٌۭ رَّبِّ إِنَّهُمْ عَصَوْنِى وَٱتَّبَعُوا۟ مَن لَّمْ يَزِدْهُ مَالُهُۥ وَوَلَدُهُۥٓ إِلَّا خَسَارًۭا
</Aside>

<Aside type="reference" title="21">
نوح (بعد از نومیدی از هدایت آنان) گفت: «پروردگارا! آنها نافرمانی من کردند و از کسانی پیروی نمودند که اموال و فرزندانشان چیزی جز زیانکاری بر آنها نیفزوده است!
</Aside>

---

<Aside type="quran" title="22">
وَمَكَرُوا۟ مَكْرًۭا كُبَّارًۭا
</Aside>

<Aside type="reference" title="22">
و (این رهبران گمراه) مکر عظیمی به کار بردند...
</Aside>

---

<Aside type="quran" title="23">
وَقَالُوا۟ لَا تَذَرُنَّ ءَالِهَتَكُمْ وَلَا تَذَرُنَّ وَدًّۭا وَلَا سُوَاعًۭا وَلَا يَغُوثَ وَيَعُوقَ وَنَسْرًۭا
</Aside>

<Aside type="reference" title="23">
و گفتند: دست از خدایان و بتهای خود برندارید (به خصوص) بتهای «وَد»، «سواع»، «یغوث»، «یعوق» و «نسر» را رها نکنید!
</Aside>

---

<Aside type="quran" title="24">
وَقَدْ أَضَلُّوا۟ كَثِيرًۭا ۖ وَلَا تَزِدِ ٱلظَّٰلِمِينَ إِلَّا ضَلَٰلًۭا
</Aside>

<Aside type="reference" title="24">
و آنها گروه بسیاری را گمراه کردند! خداوندا، ظالمان را جز ضلالت میفزا!»
</Aside>

---

<Aside type="quran" title="25">
مِّمَّا خَطِيٓـَٰٔتِهِمْ أُغْرِقُوا۟ فَأُدْخِلُوا۟ نَارًۭا فَلَمْ يَجِدُوا۟ لَهُم مِّن دُونِ ٱللَّهِ أَنصَارًۭا
</Aside>

<Aside type="reference" title="25">
(آری، سرانجام) همگی بخاطر گناهانشان غرق شدند و در آتش دوزخ وارد گشتند، و جز خدا یاورانی برای خود نیافتند!
</Aside>

---

<Aside type="quran" title="26">
وَقَالَ نُوحٌۭ رَّبِّ لَا تَذَرْ عَلَى ٱلْأَرْضِ مِنَ ٱلْكَٰفِرِينَ دَيَّارًا
</Aside>

<Aside type="reference" title="26">
نوح گفت: «پروردگارا! هیچ یک از کافران را بر روی زمین باقی مگذار!
</Aside>

---

<Aside type="quran" title="27">
إِنَّكَ إِن تَذَرْهُمْ يُضِلُّوا۟ عِبَادَكَ وَلَا يَلِدُوٓا۟ إِلَّا فَاجِرًۭا كَفَّارًۭا
</Aside>

<Aside type="reference" title="27">
چرا که اگر آنها را باقی بگذاری، بندگانت را گمراه می‌کنند و جز نسلی فاجر و کافر به وجود نمی‌آورند!
</Aside>

---

<Aside type="quran" title="28">
رَّبِّ ٱغْفِرْ لِى وَلِوَٰلِدَىَّ وَلِمَن دَخَلَ بَيْتِىَ مُؤْمِنًۭا وَلِلْمُؤْمِنِينَ وَٱلْمُؤْمِنَٰتِ وَلَا تَزِدِ ٱلظَّٰلِمِينَ إِلَّا تَبَارًۢا
</Aside>

<Aside type="reference" title="28">
پروردگارا! مرا، و پدر و مادرم و تمام کسانی را که با ایمان وارد خانه من شدند، و جمیع مردان و زنان باایمان را بیامرز؛ و ظالمان را جز هلاکت میفزا!»
</Aside>
