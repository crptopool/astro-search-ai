<Aside type="quran" title="1">
إِذَا ٱلسَّمَآءُ ٱنشَقَّتْ
</Aside>

<Aside type="reference" title="1">
در آن هنگام که آسمان [= کرات آسمانی‌] شکافته شود،
</Aside>

---

<Aside type="quran" title="2">
وَأَذِنَتْ لِرَبِّهَا وَحُقَّتْ
</Aside>

<Aside type="reference" title="2">
و تسلیم فرمان پروردگارش شود -و سزاوار است چنین باشد-
</Aside>

---

<Aside type="quran" title="3">
وَإِذَا ٱلْأَرْضُ مُدَّتْ
</Aside>

<Aside type="reference" title="3">
و در آن هنگام که زمین گسترده شود،
</Aside>

---

<Aside type="quran" title="4">
وَأَلْقَتْ مَا فِيهَا وَتَخَلَّتْ
</Aside>

<Aside type="reference" title="4">
و آنچه در درون دارد بیرون افکنده و خالی شود،
</Aside>

---

<Aside type="quran" title="5">
وَأَذِنَتْ لِرَبِّهَا وَحُقَّتْ
</Aside>

<Aside type="reference" title="5">
و تسلیم فرمان پروردگارش گردد -و شایسته است که چنین باشد-
</Aside>

---

<Aside type="quran" title="6">
يَٰٓأَيُّهَا ٱلْإِنسَٰنُ إِنَّكَ كَادِحٌ إِلَىٰ رَبِّكَ كَدْحًۭا فَمُلَٰقِيهِ
</Aside>

<Aside type="reference" title="6">
ای انسان! تو با تلاش و رنج بسوی پروردگارت می‌روی و او را ملاقات خواهی کرد!
</Aside>

---

<Aside type="quran" title="7">
فَأَمَّا مَنْ أُوتِىَ كِتَٰبَهُۥ بِيَمِينِهِۦ
</Aside>

<Aside type="reference" title="7">
پس کسی که نامه اعمالش به دست راستش داده شود،
</Aside>

---

<Aside type="quran" title="8">
فَسَوْفَ يُحَاسَبُ حِسَابًۭا يَسِيرًۭا
</Aside>

<Aside type="reference" title="8">
بزودی حساب آسانی برای او می‌شود،
</Aside>

---

<Aside type="quran" title="9">
وَيَنقَلِبُ إِلَىٰٓ أَهْلِهِۦ مَسْرُورًۭا
</Aside>

<Aside type="reference" title="9">
و خوشحال به اهل و خانواده‌اش بازمی‌گردد.
</Aside>

---

<Aside type="quran" title="10">
وَأَمَّا مَنْ أُوتِىَ كِتَٰبَهُۥ وَرَآءَ ظَهْرِهِۦ
</Aside>

<Aside type="reference" title="10">
و امّا کسی که نامه اعمالش به پشت سرش داده شود،
</Aside>

---

<Aside type="quran" title="11">
فَسَوْفَ يَدْعُوا۟ ثُبُورًۭا
</Aside>

<Aside type="reference" title="11">
بزودی فریاد می‌زند وای بر من که هلاک شدم!
</Aside>

---

<Aside type="quran" title="12">
وَيَصْلَىٰ سَعِيرًا
</Aside>

<Aside type="reference" title="12">
و در شعله‌های سوزان آتش می‌سوزد.
</Aside>

---

<Aside type="quran" title="13">
إِنَّهُۥ كَانَ فِىٓ أَهْلِهِۦ مَسْرُورًا
</Aside>

<Aside type="reference" title="13">
چرا که او در میان خانواده‌اش پیوسته (از کفر و گناه خود) مسرور بود!
</Aside>

---

<Aside type="quran" title="14">
إِنَّهُۥ ظَنَّ أَن لَّن يَحُورَ
</Aside>

<Aside type="reference" title="14">
او گمان می‌کرد هرگز بازگشت نمی‌کند!
</Aside>

---

<Aside type="quran" title="15">
بَلَىٰٓ إِنَّ رَبَّهُۥ كَانَ بِهِۦ بَصِيرًۭا
</Aside>

<Aside type="reference" title="15">
آری، پروردگارش نسبت به او بینا بود (و اعمالش را برای حساب ثبت کرد)!
</Aside>

---

<Aside type="quran" title="16">
فَلَآ أُقْسِمُ بِٱلشَّفَقِ
</Aside>

<Aside type="reference" title="16">
سوگند به شفق،
</Aside>

---

<Aside type="quran" title="17">
وَٱلَّيْلِ وَمَا وَسَقَ
</Aside>

<Aside type="reference" title="17">
و سوگند به شب و آنچه را جمع‌آوری می‌کند،
</Aside>

---

<Aside type="quran" title="18">
وَٱلْقَمَرِ إِذَا ٱتَّسَقَ
</Aside>

<Aside type="reference" title="18">
و سوگند به ماه آنگاه که بَدر کامل می‌شود،
</Aside>

---

<Aside type="quran" title="19">
لَتَرْكَبُنَّ طَبَقًا عَن طَبَقٍۢ
</Aside>

<Aside type="reference" title="19">
که همه شما پیوسته از حالی به حال دیگر منتقل می‌شوید (تا به کمال برسید).
</Aside>

---

<Aside type="quran" title="20">
فَمَا لَهُمْ لَا يُؤْمِنُونَ
</Aside>

<Aside type="reference" title="20">
پس چرا آنان ایمان نمی‌آورند؟!
</Aside>

---

<Aside type="quran" title="21">
وَإِذَا قُرِئَ عَلَيْهِمُ ٱلْقُرْءَانُ لَا يَسْجُدُونَ ۩
</Aside>

<Aside type="reference" title="21">
و هنگامی که قرآن بر آنها خوانده می‌شود سجده نمی‌کنند؟!
</Aside>

---

<Aside type="quran" title="22">
بَلِ ٱلَّذِينَ كَفَرُوا۟ يُكَذِّبُونَ
</Aside>

<Aside type="reference" title="22">
بلکه کافران پیوسته آیات الهی را انکار می‌کنند!
</Aside>

---

<Aside type="quran" title="23">
وَٱللَّهُ أَعْلَمُ بِمَا يُوعُونَ
</Aside>

<Aside type="reference" title="23">
و خداوند آنچه را در دل پنهان می‌دارند بخوبی می‌داند!
</Aside>

---

<Aside type="quran" title="24">
فَبَشِّرْهُم بِعَذَابٍ أَلِيمٍ
</Aside>

<Aside type="reference" title="24">
پس آنها را به عذابی دردناک بشارت ده!
</Aside>

---

<Aside type="quran" title="25">
إِلَّا ٱلَّذِينَ ءَامَنُوا۟ وَعَمِلُوا۟ ٱلصَّٰلِحَٰتِ لَهُمْ أَجْرٌ غَيْرُ مَمْنُونٍۭ
</Aside>

<Aside type="reference" title="25">
مگر کسانی که ایمان آورده و اعمال صالح انجام داده‌اند، که برای آنان پاداشی است قطع‌نشدنی!
</Aside>
