<Aside type="quran" title="1">
ٱقْتَرَبَ لِلنَّاسِ حِسَابُهُمْ وَهُمْ فِى غَفْلَةٍۢ مُّعْرِضُونَ
</Aside>

<Aside type="reference" title="1">
لوگوں کا حساب (اعمال کا وقت) نزدیک آپہنچا ہے اور وہ غفلت میں (پڑے اس سے) منہ پھیر رہے ہیں
</Aside>

---

<Aside type="quran" title="2">
مَا يَأْتِيهِم مِّن ذِكْرٍۢ مِّن رَّبِّهِم مُّحْدَثٍ إِلَّا ٱسْتَمَعُوهُ وَهُمْ يَلْعَبُونَ
</Aside>

<Aside type="reference" title="2">
ان کے پاس کوئی نئی نصیحت ان کے پروردگار کی طرف سے نہیں آتی مگر وہ اسے کھیلتے ہوئے سنتے ہیں
</Aside>

---

<Aside type="quran" title="3">
لَاهِيَةًۭ قُلُوبُهُمْ ۗ وَأَسَرُّوا۟ ٱلنَّجْوَى ٱلَّذِينَ ظَلَمُوا۟ هَلْ هَٰذَآ إِلَّا بَشَرٌۭ مِّثْلُكُمْ ۖ أَفَتَأْتُونَ ٱلسِّحْرَ وَأَنتُمْ تُبْصِرُونَ
</Aside>

<Aside type="reference" title="3">
ان کے دل غفلت میں پڑے ہوئے ہیں اور ظالم لوگ (آپس میں) چپکے چپکے باتیں کرتے ہیں کہ یہ (شخص کچھ بھی) نہیں مگر تمہارے جیسا آدمی ہے۔ تو تم آنکھوں دیکھتے جادو (کی لپیٹ) میں کیوں آتے ہو
</Aside>

---

<Aside type="quran" title="4">
قَالَ رَبِّى يَعْلَمُ ٱلْقَوْلَ فِى ٱلسَّمَآءِ وَٱلْأَرْضِ ۖ وَهُوَ ٱلسَّمِيعُ ٱلْعَلِيمُ
</Aside>

<Aside type="reference" title="4">
(پیغمبر نے) کہا کہ جو بات آسمان اور زمین میں (کہی جاتی) ہے میرا پروردگار اسے جانتا ہے۔ اور وہ سننے والا (اور) جاننے والا ہے
</Aside>

---

<Aside type="quran" title="5">
بَلْ قَالُوٓا۟ أَضْغَٰثُ أَحْلَٰمٍۭ بَلِ ٱفْتَرَىٰهُ بَلْ هُوَ شَاعِرٌۭ فَلْيَأْتِنَا بِـَٔايَةٍۢ كَمَآ أُرْسِلَ ٱلْأَوَّلُونَ
</Aside>

<Aside type="reference" title="5">
بلکہ (ظالم) کہنے لگے کہ (یہ قرآن) پریشان (باتیں ہیں جو) خواب (میں دیکھ لی) ہیں۔ (نہیں) بلکہ اس نے اس کو اپنی طرف سے بنا لیا ہے (نہیں) بلکہ (یہ شعر ہے جو اس) شاعر (کا نتیجہٴ طبع) ہے۔ تو جیسے پہلے (پیغمبر نشانیاں دے کر) بھیجے گئے تھے (اسی طرح) یہ بھی ہمارے پاس کوئی نشانی لائے
</Aside>

<Aside type="quran" title="6">

</Aside>

<Aside type="reference" title="6">

</Aside>

</Aside>

<Aside type="reference" title="7">
اور ہم نے تم سے پہلے مرد ہی (پیغمبر بنا کر) بھیجے جن کی طرف ہم وحی بھیجتے تھے۔ اگر تم نہیں جانتے تو جو یاد رکھتے ہیں ان سے پوچھ لو
</Aside>

<Aside type="quran" title="8">

</Aside>

<Aside type="reference" title="8">

</Aside>

<Aside type="quran" title="9">

</Aside>

<Aside type="reference" title="10">

</Aside>

<Aside type="reference" title="11">

<Aside type="reference" title="12">

<Aside type="reference" title="13">

<Aside type="reference" title="15">

<Aside type="reference" title="16">
