<Aside type="quran" title="1">
بِّسْمِ ٱللَّهِ ٱلرَّحْمَٰنِ ٱلرَّحِيمِ إِنَّآ أَنزَلْنَٰهُ فِى لَيْلَةِ ٱلْقَدْرِ
</Aside>

<Aside type="reference" title="1">
ہم نے اس( قرآن) کو شب قدر میں نازل (کرنا شروع) کیا
</Aside>

---

<Aside type="quran" title="2">
وَمَآ أَدْرَىٰكَ مَا لَيْلَةُ ٱلْقَدْرِ
</Aside>

<Aside type="reference" title="2">
اور تمہیں کیا معلوم کہ شب قدر کیا ہے؟
</Aside>

---

<Aside type="quran" title="3">
لَيْلَةُ ٱلْقَدْرِ خَيْرٌۭ مِّنْ أَلْفِ شَهْرٍۢ
</Aside>

<Aside type="reference" title="3">
شب قدر ہزار مہینے سے بہتر ہے
</Aside>

---

<Aside type="quran" title="4">
تَنَزَّلُ ٱلْمَلَٰٓئِكَةُ وَٱلرُّوحُ فِيهَا بِإِذْنِ رَبِّهِم مِّن كُلِّ أَمْرٍۢ
</Aside>

<Aside type="reference" title="4">
اس میں روح (الامین) اور فرشتے ہر کام کے (انتظام کے) لیے اپنے پروردگار کے حکم سے اترتے ہیں
</Aside>

<Aside type="quran" title="5">
سَلَٰمٌ هِىَ حَتَّىٰ مَطْلَعِ ٱلْفَجْرِ
</Aside>

<Aside type="reference" title="5">
یہ( رات) طلوع صبح تک (امان اور) سلامتی ہے
