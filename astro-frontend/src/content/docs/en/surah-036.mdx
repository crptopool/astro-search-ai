            قَالُوا۟ رَبُّنَا يَعْلَمُ إِنَّآ إِلَيْكُمْ لَمُرْسَلُونَ

            </Aside>

        ---
        
            <Aside type="quran" title="42">

            and created for them similar things to ride in.
            </Aside>

        ---
        
            <Aside type="quran" title="43">

            </Aside>

            </Aside>

        ---
        
            <Aside type="quran" title="44">

            </Aside>

            </Aside>
